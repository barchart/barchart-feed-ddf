133 	@Override
134 	public final void decodeXML(final Element tag) {
135 
136 		xmlCheckTagName(tag, TAG);
137 
138 		final DDF_Exchange exch = getExchange();
139 		final DDF_Fraction frac = getFraction();
140 
141 		//
142 
143 		final String indCode = xmlStringDecode(tag, SESSION_INDICATOR, XML_STOP);
144 		setIndicator(DDF_Indicator.fromCode(indCode));
145 
146 		final byte dayCode = xmlByteDecode(tag, TRADE_DAY, XML_PASS);
147 		setTradeDay(DDF_TradeDay.fromCode(dayCode));
148 
149 		//
150 
151 		// last trade today
152 		final byte typeCode = xmlByteDecode(tag, SESSION, XML_PASS);
153 		setSession(DDF_Session.fromPair(exch.ord, typeCode));
154 		priceLast = xmlDecimalDecode(frac, tag, PRICE_LAST, XML_PASS);
155 		sizeLast = xmlLongDecode(tag, SIZE_LAST, XML_PASS);
156 		timeLast = xmlTimeDecode(exch.kind.time.zone, tag, TIME_LAST, XML_PASS);
157 
158 		// settle yesterday
159 		priceLastPrevious = xmlDecimalDecode(frac, tag, PRICE_LAST_PREV,
160 				XML_PASS);
161 
162 		//
163 
164 		priceOpen = xmlDecimalDecode(frac, tag, PRICE_OPEN, XML_PASS);
165 		priceHigh = xmlDecimalDecode(frac, tag, PRICE_HIGH, XML_PASS);
166 		priceLow = xmlDecimalDecode(frac, tag, PRICE_LOW, XML_PASS);
167 		priceSettle = xmlDecimalDecode(frac, tag, PRICE_SETTLE, XML_PASS);
168 
169 		sizeVolume = xmlLongDecode(tag, SIZE_VOLUME, XML_PASS);
170 		sizeInterest = xmlLongDecode(tag, SIZE_INTEREST, XML_PASS);
171 
172 		//
173 
174 		millisUTC = xmlTimeDecode(exch.kind.time.zone, tag, TIME_UPDATE,
175 				XML_PASS);
176 
177 		// log.debug("### priceLast : {}", priceLast);
178 
179 	}

184 	@Override
185 	public final void encodeXML(final Element tag) {
186 
187 		xmlCheckTagName(tag, TAG);
188 
189 		final DDF_Exchange exch = getExchange();
190 		final DDF_Fraction frac = getFraction();
191 
192 		//
193 
194 		xmlStringEncode(getIndicator().code, tag, SESSION_INDICATOR);
195 
196 		xmlByteEncode(getTradeDay().code, tag, TRADE_DAY);
197 
198 		//
199 
200 		final DDF_Session session = getSession();
201 
202 		//
203 
204 		xmlByteEncode(session.code, tag, SESSION);
205 		xmlDecimalEncode(priceLast, frac, tag, PRICE_LAST);
206 		xmlLongEncode(sizeLast, tag, SIZE_LAST);
207 		xmlTimeEncode(timeLast, exch.kind.time.zone, tag, TIME_LAST);
208 
209 		//
210 
211 		xmlDecimalEncode(priceLastPrevious, frac, tag, PRICE_LAST_PREV);
212 
213 		//
214 
215 		xmlDecimalEncode(priceOpen, frac, tag, PRICE_OPEN);
216 		xmlDecimalEncode(priceHigh, frac, tag, PRICE_HIGH);
217 		xmlDecimalEncode(priceLow, frac, tag, PRICE_LOW);
218 		xmlDecimalEncode(priceSettle, frac, tag, PRICE_SETTLE);
219 
220 		xmlLongEncode(sizeVolume, tag, SIZE_VOLUME);
221 		xmlLongEncode(sizeInterest, tag, SIZE_INTEREST);
222 
223 		//
224 
225 		xmlTimeEncode(millisUTC, exch.kind.time.zone, tag, TIME_UPDATE);
226 
227 	}
