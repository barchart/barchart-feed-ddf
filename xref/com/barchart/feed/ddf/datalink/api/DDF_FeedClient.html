8   package com.barchart.feed.ddf.datalink.api;

10  import java.util.Set;
11  import java.util.concurrent.Future;
12  
13  import com.barchart.feed.ddf.datalink.enums.DDF_FeedEvent;

15  /**
16   * Client responsible for lowest level connectivity to the data server.
17   * <p>
18   * Implementation should handle all communications from server asynchronously.
19   * Permits only nonblocking commands.
20   * <p>
21   * User is required to bind a StateListener to the client, creating a callback
22   * for login state changes for the client.
23   * <p>
24   */
25  public interface DDF_FeedClient extends DDF_FeedClientBase {
26  
27  	/**
28  	 * Handles multiple subscription requests.
29  	 * <p>
30  	 * If the client already has a subscription for any instrument of the set
31  	 * then this will overwrite it.
32  	 * <p>
33  	 * If called while the client is offline, registers the subscriptions and
34  	 * returns a future which immediately succeeds.
35  	 * 
36  	 * @param subscription
37  	 *            The set of subscriptions to subscribe.
38  	 * @return A Future which returns true if successful.
39  	 */
40  	Future<Boolean> subscribe(Set<Subscription> subscriptions);
41  
42  	/**
43  	 * Handles a subscription request.
44  	 * <p>
45  	 * If the client already has a subscription for the instrument then this
46  	 * will overwrite it.
47  	 * <p>
48  	 * If called while the client is offline, registers the subscription and
49  	 * returns a future which immediately succeeds.
50  	 * 
51  	 * @param subscription
52  	 *            The subscription to subscribe.
53  	 * @return A Future which returns true if successful.
54  	 */
55  	Future<Boolean> subscribe(Subscription subscription);
56  
57  	/**
58  	 * Handles multiple unsubscription requests.
59  	 * <p>
60  	 * If called while the client is offline, unregisters the subscriptions and
61  	 * returns a future which immediately succeeds.
62  	 * 
63  	 * @param subscription
64  	 *            The set of subscriptions to unsubscribe.
65  	 * @return A Future which returns true if successful.
66  	 */
67  	Future<Boolean> unsubscribe(Set<Subscription> subscriptions);
68  
69  	/**
70  	 * Handles an unsubscription request.
71  	 * <p>
72  	 * If called while the client is offline, unregisters the subscription and
73  	 * returns a future which immediately succeeds.
74  	 * 
75  	 * @param subscription
76  	 *            The subscription to unsubscribe.
77  	 * @return A Future which returns true if successful.
78  	 */
79  	Future<Boolean> unsubscribe(Subscription subscription);
80  
81  	/**
82  	 * Sets the event policy for a specific feed event. Default policies are set
83  	 * by the constructor to handle disconnects and login success.
84  	 * 
85  	 * @param event
86  	 *            The feed event on which to enact the policy.
87  	 * @param policy
88  	 *            The even policy to register.
89  	 */
90  	void setPolicy(DDF_FeedEvent event, EventPolicy policy);
91  
92  }
