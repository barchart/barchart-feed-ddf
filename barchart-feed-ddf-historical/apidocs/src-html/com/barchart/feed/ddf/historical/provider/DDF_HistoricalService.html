111            /**
112             * New result.
113             *
114             * @param <E> the element type
115             * @param settings the settings
116             * @param query the query
117             * @param listener the listener
118             * @return the dD f_ result
119             * @throws RuntimeException the runtime exception
120             */
121            @SuppressWarnings("unchecked")
122            public static final <E extends DDF_Entry> DDF_Result<E> newResult(
123                            final DDF_Settings settings, /* local */DDF_Query<E> query,
124                            final DDF_ResultListener listener) throws RuntimeException {
125    
126                    checkNull(query, "query is null");
127                    checkNull(query.type, "query type is null");
128    
129                    // detach from source
130                    query = query.clone();
131    
132                    final String stringURL = urlQuery(settings, query);
133                    log.debug("stringURL : \n\t {}", stringURL);
134    
135                    final Builder builder = Builder.from(query.type);
136    
137                    final DDF_Instrument instrument = query.instrument;
138    
139                    final E entryReference = (E) builder.newEntry(0, null, instrument);
140    
141                    final Result<E> result = new Result<E>(query, entryReference, listener);
142    
143                    String firstLine = "";
144                    int index = 0;
145    
146                    try {
147    
148                            final ZipReader reader = ZipReader.fromURL(stringURL);
149    
150                            while (true) {
151                                    final String inputLine = reader.readLine();
152                                    if (inputLine == null) {
153                                            break;
154                                    }
155                                    if (inputLine.length() == 0) {
156                                            continue;
157                                    }
158                                    // log.info("inputLine : {}", inputLine);
159                                    if (index == 0) {
160                                            firstLine = inputLine;
161                                            if (firstLine.contains(KEYWORD_ERROR)) {
162                                                    index++;
163                                                    break;
164                                            }
165                                    }
166                                    final E entry = (E) builder.newEntry(index, inputLine,
167                                                    instrument);
168                                    result.add(index, entry);
169                                    index++;
170                            }
171    
172                            reader.close();
173    
174                            if (index == 0) {
175                                    result.status = DDF_ResultStatus.SUCCESS;
176                                    result.statusComment = STATUS_EMPTY;
177                            } else if (firstLine.contains(KEYWORD_ERROR)) {
178                                    result.status = DDF_ResultStatus.ERROR;
179                                    result.statusComment = firstLine;
180                            } else {
181                                    result.status = DDF_ResultStatus.SUCCESS;
182                                    result.statusComment = STATUS_COUNT + index;
183                            }
184    
185                    } catch (final DDF_ResultInterruptedException e) {
186    
187                            log.debug("query lookup interrupted; query={}", query);
188    
189                            result.status = DDF_ResultStatus.INTERRUPTED;
190                            result.statusComment = e.getMessage() + " at " + index;
191    
192                    } catch (final DDF_ResultEmptyException e) {
193    
194                            log.debug("got an empty server page; query={}", query);
195    
196                            result.status = DDF_ResultStatus.SUCCESS;
197                            result.statusComment = STATUS_COUNT + 0;
198    
199                    } catch (final Exception e) {
200    
201                            log.debug("query lookup failed; query={}", query);
202                            log.error("query lookup failed", e);
203    
204                            result.status = DDF_ResultStatus.ERROR;
205                            result.statusComment = e.getMessage();
206    
207                    }
208    
209                    return result;
210    
211            }
