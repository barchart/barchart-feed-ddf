008    package com.barchart.feed.ddf.datalink.enums;
009    
010    import java.util.EnumSet;
011    import java.util.Set;
012    
013    import com.barchart.feed.base.market.enums.MarketEvent;
014    import com.barchart.feed.ddf.instrument.api.DDF_Instrument;
015    import com.barchart.feed.ddf.instrument.enums.DDF_InstrumentField;
016    import com.barchart.feed.ddf.util.FeedDDF;
017    
018    /**
019     * Flags for ddf jerq "go" command to activate specific information channels in
020     * the feed.
021     */
022    public enum DDF_FeedInterest {
023    
024            UNKNOWN('?'), //
025    
026            /** A one-time snapshot of a symbol's full book */
027            BOOK_SNAPSHOT('b'), //
028            /** A stream of the symbol's top of book */
029            BOOK_UPDATE('B'), // only book top
030    
031            //
032            /** A one-time cumulative volume snapshot */
033            CUVOL_SNAPSHOT('c'), //
034    
035            //
036            /** A one-time snapshot quote for a symbol */
037            QUOTE_SNAPSHOT('s'), //
038            /** A stream of quote updates for a symbol */
039            QUOTE_UPDATE('S'), //
040    
041            ;
042    
043            private static final DDF_FeedInterest[] ENUM_VALS = values();
044    
045            public static final int size() {
046                    return ENUM_VALS.length;
047            }
048    
049            /** The code. */
050            public final char code;
051    
052            private DDF_FeedInterest(final char code) {
053                    this.code = code;
054            }
055    
056            public static final DDF_FeedInterest of(final char code) {
057                    for (final DDF_FeedInterest known : values()) {
058                            if (known.code == code) {
059                                    return known;
060                            }
061                    }
062                    return UNKNOWN;
063            }
064    
065            private static final String NONE = "";
066    
067            /**
068             * Parses a set of MarketEvents to a String of FeedInterest names.
069             * 
070             * @param eventSet
071             *            The set of events to parse.
072             * @return a String of FeedInterest names.
073             */
074            public static final String from(final Set<MarketEvent> eventSet) {
075    
076                    if (eventSet == null || eventSet.isEmpty()) {
077                            return NONE;
078                    }
079    
080                    final Set<DDF_FeedInterest> result = EnumSet
081                                    .noneOf(DDF_FeedInterest.class);
082    
083                    for (final MarketEvent event : eventSet) {
084                            switch (event) {
085    
086                            case NEW_BOOK_ERROR:
087                                    // debug use only
088                                    continue;
089    
090                            case NEW_BOOK_SNAPSHOT:
091                                    result.add(BOOK_SNAPSHOT);
092                                    continue;
093    
094                            case NEW_BOOK_UPDATE:
095                            case NEW_BOOK_TOP:
096                                    result.add(BOOK_UPDATE);
097                                    continue;
098    
099                            case NEW_CUVOL_SNAPSHOT:
100                                    result.add(CUVOL_SNAPSHOT);
101                                    continue;
102    
103                            case NEW_BAR_CURRENT_NET:
104                            case NEW_BAR_CURRENT_PIT:
105                            case NEW_BAR_CURRENT_EXT:
106                            case NEW_BAR_CURRENT:
107                            case NEW_BAR_PREVIOUS:
108                                    result.add(QUOTE_SNAPSHOT);
109                                    continue;
110    
111                            default:
112                                    result.add(QUOTE_UPDATE);
113                                    continue;
114                            }
115                    }
116    
117                    if (result.isEmpty()) {
118                            return NONE;
119                    }
120    
121                    final StringBuilder text = new StringBuilder(size());
122    
123                    for (final DDF_FeedInterest interest : result) {
124                            text.append(interest.code);
125                    }
126    
127                    return text.toString();
128    
129            }
130    
131            /**
132             * Command.
133             * 
134             * @param instrumentDDF
135             *            the instrument ddf
136             * @param eventSet
137             *            the event set
138             * @return the char sequence
139             */
140            public static CharSequence command(final DDF_Instrument instrumentDDF,
141                            final Set<MarketEvent> eventSet) {
142    
143                    final CharSequence symbol = instrumentDDF
144                                    .get(DDF_InstrumentField.DDF_SYMBOL_REALTIME);
145    
146                    final CharSequence interest = DDF_FeedInterest.from(eventSet);
147    
148                    final CharSequence command = FeedDDF.tcpGo(symbol, interest);
149    
150                    return command;
151    
152            }
153
