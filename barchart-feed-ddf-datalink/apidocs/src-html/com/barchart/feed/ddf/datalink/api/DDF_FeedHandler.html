008    package com.barchart.feed.ddf.datalink.api;
009    
010    import com.barchart.feed.ddf.datalink.enums.DDF_FeedEvent;
011    import com.barchart.feed.ddf.message.api.DDF_BaseMessage;
012    
013    /**
014     * Specifies the handling of feed events and data messages from the server.
015     * <p>
016     * All calls are handles asynchronously by the feed client which the feed
017     * handler is bound to.
018     * 
019     */
020    public interface DDF_FeedHandler {
021    
022            /**
023             * Asynchronous server event callback.
024             * 
025             * @param event
026             *            The event to be handled.
027             */
028            void handleEvent(DDF_FeedEvent event);
029    
030            /**
031             * Asynchronous data receipt callback.
032             * 
033             * @param message
034             *            The data message produced by server.
035             */
036            void handleMessage(DDF_BaseMessage message);
037    
038    }
