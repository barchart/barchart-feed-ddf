030            static ChannelBuffer decode(final String text) throws Exception {
031    
032                    final byte[] array = text.getBytes();
033    
034                    final ChannelBuffer source = ChannelBuffers.copiedBuffer(array);
035    
036                    final MsgDeframerDDF framer = new MsgDeframerDDF();
037    
038                    final ChannelBuffer target = (ChannelBuffer) framer.decode(null, null,
039                                    source);
040    
041                    return target;
042    
043            }
044    
045            static List<String> parse(final String text) throws Exception {
046    
047                    final byte[] array = text.getBytes();
048    
049                    final ChannelBuffer buffer = ChannelBuffers.copiedBuffer(array);
050    
051                    final MsgDeframerDDF framer = new MsgDeframerDDF();
052    
053                    final List<String> list = new LinkedList<String>();
054    
055                    while (true) {
056    
057                            final ChannelBuffer frame = (ChannelBuffer) framer.decode(null,
058                                            null, buffer);
059    
060                            if (frame == null) {
061                                    break;
062                            }
063    
064                            /* parent frame decoder does this sometimes */
065                            // buffer.discardReadBytes();
066    
067                            list.add(new String(frame.array()));
068    
069                    }
070    
071                    return list;
072            }
073    
074            static void log(final ChannelBuffer buffer) {
075    
076                    log.debug("target : {}", new String(buffer.array()));
077    
078            }

115                    final String msg3 = " ddfplus  buffer \n";

117                    final String text = msg0 + msg1 + msg2 + msg3;
