072             */
073            @Test
074            public void testDecode_1() {
075    
076                    final DDF_TradeDay todayUTC = DDF_TradeDay.fromMillisUTC(System
077                                    .currentTimeMillis());
078    
079                    final DF_3B_Book msg = new DF_3B_Book();
080    
081                    final ByteBuffer buffer = ByteBuffer.wrap(ba3B_1);
082    
083                    msg.decodeDDF(buffer);
084    
085                    assertEquals(msg.getExchange(), DDF_Exchange.CME_Main);
086                    assertEquals(msg.getFraction(), DDF_Fraction.N2);
087                    assertEquals(msg.getMessageType(), DDF_MessageType.BOOK_SNAP);
088                    assertEquals(msg.getTradeDay(), todayUTC);
089                    assertEquals(msg.getSession(), DDF_Session.FUTURE_COMBO);
090                    assertEquals(msg.getId(), "ESZ1");
091                    assertEquals(msg.getSpreadType(), DDF_SpreadType.UNKNOWN);
092                    assertEquals(msg.getDelay(), 0);
093    
094                    final long t1 = msg.getTime().asMillisUTC();
095                    final long t2 = System.currentTimeMillis();
096                    assertTrue(t2 - t1 < 100);
097    
098            }

191                            final long mant, final int exp, final int size) {
