008    package com.barchart.feed.ddf.message.api;
009    
010    import com.barchart.feed.ddf.message.enums.DDF_Condition;
011    import com.barchart.feed.ddf.message.enums.DDF_QuoteMode;
012    import com.barchart.feed.ddf.message.enums.DDF_QuoteState;
013    import com.barchart.util.anno.NotMutable;
014    import com.barchart.util.values.api.PriceValue;
015    import com.barchart.util.values.api.TextValue;
016    
017    /**
018     * represents xml "market quote"; a collection of sessions.
019     */
020    @NotMutable
021    public interface DDF_MarketQuote extends DDF_MarketBookTop {
022    
023            // instrument fields
024    
025            /** ddf feed symbol name; such "Microsoft Corporation" */
026            TextValue getSymbolName();
027    
028            /** a.k.a tick increment */
029            PriceValue getPriceStep();
030    
031            /** future contract / stock share */
032            PriceValue getPointValue();
033    
034            // intrinsic market state
035    
036            /** market active / halted; etc */
037            DDF_Condition getCondition();
038    
039            /** market date roll over detection */
040            DDF_QuoteState getState();
041    
042            /** realtime vs delayed; etc */
043            DDF_QuoteMode getMode();
044    
045            // market component sessions
046    
047            /**
048             * @return session array; can be empty; supported session types: CURRENT
049             *         (default or combo), PREVIOUS (default or combo)
050             **/
051            DDF_MarketSession[] sessions();
052    
053    }
