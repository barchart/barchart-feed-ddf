008    package com.barchart.feed.ddf.settings.example;
009    
010    import org.slf4j.Logger;
011    import org.slf4j.LoggerFactory;
012    
013    import com.barchart.feed.ddf.settings.api.DDF_Settings;
014    import com.barchart.feed.ddf.settings.provider.DDF_SettingsService;
015    
016    // TODO: Auto-generated Javadoc
017    /**
018     * The Class SettingsExample.
019     */
020    
021    public class SettingsExample {
022    
023            private static final Logger log = LoggerFactory
024                            .getLogger(SettingsExample.class);
025    
026            /**
027             * The main method.
028             *
029             * @param args the arguments
030             */
031            public final static void main(String[] args) {
032    
033                    final String username = System.getProperty("barchart.username");
034                    final String password = System.getProperty("barchart.password");
035    
036                    final DDF_Settings settings = DDF_SettingsService.newSettings(username,
037                                    password);
038    
039                    if (settings.isValidLogin()) {
040                            log.info("login success : {}", settings);
041                    } else {
042                            log.error("login failure : {}", settings);
043                    }
044    
045            }
046    
047    }
