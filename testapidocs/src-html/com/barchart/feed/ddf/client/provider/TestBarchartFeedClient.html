004    package com.barchart.feed.ddf.client.provider;
005    
006    import com.barchart.feed.base.instrument.values.MarketInstrument;
007    import com.barchart.feed.base.market.api.Market;
008    import com.barchart.feed.base.market.api.MarketTaker;
009    import com.barchart.feed.base.market.enums.MarketEvent;
010    import com.barchart.feed.base.market.enums.MarketField;
011    import com.barchart.feed.client.api.FeedStateListener;
012    import com.barchart.feed.client.enums.FeedState;
013    import com.barchart.feed.client.provider.BarchartFeedClient;
014    
015    /**
016     * 
017     * Stress test to try and break the login/logout lifecycle
018     */
019    public class TestBarchartFeedClient {
020    
021            /**
022             * @param args
023             */
024            public static void main(final String[] args) throws Exception {
025    
026                    final String username = System.getProperty("barchart.username");
027                    final String password = System.getProperty("barchart.password");
028    
029                    final BarchartFeedClient client = new BarchartFeedClient();
030    
031                    final String symbol = "GOOG";
032    
033                    final MarketInstrument instrument = client.lookup(symbol);
034    
035                    System.out.println(instrument.toString());
036    
037                    final FeedStateListener feedListener = new FeedStateListener() {
038    
039                            @Override
040                            public void stateUpdate(final FeedState state) {
041    
042                                    if (state == FeedState.LOGGED_IN) {
043                                            client.addTaker(TakerFactory.makeFactory(instrument));
044                                    }
045    
046                            }
047    
048                    };
049    
050                    for (int i = 0; i < 100; i++) {
051    
052                            if (Math.random() < 0.5) {
053    
054                                    client.login(username, password);
055    
056                                    client.bindFeedStateListener(feedListener);
057    
058                            } else {
059                                    client.shutdown();
060                            }
061    
062                            Thread.sleep((long) (Math.random() * 3000));
063    
064                    }
065    
066                    client.shutdown();
067    
068            }
069    
070            private static class TakerFactory {
071    
072                    static MarketTaker<Market>
073                                    makeFactory(final MarketInstrument instrument) {
074                            return new MarketTaker<Market>() {
075    
076                                    @Override
077                                    public MarketField<Market> bindField() {
078    
079                                            return MarketField.MARKET;
080    
081                                    }
082    
083                                    @Override
084                                    public MarketEvent[] bindEvents() {
085    
086                                            return MarketEvent.in(MarketEvent.values());
087    
088                                    }
089    
090                                    @Override
091                                    public MarketInstrument[] bindInstruments() {
092    
093                                            return new MarketInstrument[] { instrument };
094    
095                                    }
096    
097                                    @Override
098                                    public void onMarketEvent(final MarketEvent event,
099                                                    final MarketInstrument instrument, final Market value) {
100    
101                                            System.out.println(value.get(MarketField.BOOK_TOP)
102                                                            .toString());
103    
104                                    }
105    
106                            };
107    
108                    }
109            }
110    
111    }
