061            static void decodeMessage(final byte[] array) {
062    
063                    final DDF_BaseMessage message;
064                    try {
065                            message = DDF_MessageService.decode(array);
066                    } catch (Exception e) {
067                            log.error("decode failed", e);
068                            return;
069                    }
070    
071                    log.info("message : {}", message.toString());
072    
073                    log.info("message fields : {}", message.toStringFields());
074    
075            }
