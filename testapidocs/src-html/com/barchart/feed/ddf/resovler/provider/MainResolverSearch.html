021    /**
022     * The Class MainResolverSearch.
023     */
024    public class MainResolverSearch {
025    
026            private static Logger log = LoggerFactory
027                            .getLogger(MainResolverSearch.class);
028    
029            static DDF_Resolver resolver;
030    
031            static void search() throws Exception {
032    
033                    log.debug("start");
034    
035                    // List<DDF_Instrument> result =
036                    // resolver.searchSimple("mini nasd 100 08");
037    
038                    // List<DDF_Instrument> result = resolver.searchSimple("es*1 *450c");
039                    final List<DDF_Instrument> result = resolver.searchSimple("msft");
040    
041                    log.debug("finish");
042    
043                    log.debug("result.size : {}", result.size());
044    
045                    for (final DDF_Instrument inst : result) {
046    
047                            log.debug("instrument {} {}", inst.get(InstrumentField.ID),
048                                            inst.get(InstrumentField.DESCRIPTION));
049    
050                    }
051    
052            }
053    
054            /**
055             * The main method.
056             * 
057             * @param args
058             *            the arguments
059             * @throws Exception
060             *             the exception
061             */
062            public static void main(final String... args) throws Exception {
063    
064                    final String folder = MainConst.FOLDER;
065                    final int limit = MainConst.LIMIT;
066    
067                    resolver = DDF_ResolverProvider.newInstance(null, folder, limit);
068    
069                    while (true) {
070    
071                            search();
072    
073                            Thread.sleep(3 * 1000);
074    
075                    }
076    
077            }
078    
079    }
