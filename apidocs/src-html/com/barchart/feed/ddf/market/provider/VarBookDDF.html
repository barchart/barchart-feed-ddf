017

035    @Mutable
036    @ThreadSafe(rule = "use in runSafe() only")
037    public final class VarBookDDF extends ValueFreezer<MarketBook> implements
038                    MarketDoBook {
039    
040            @SuppressWarnings("serial")
041            private static class EntryMap extends TreeMap<PriceValue, MarketBookEntry> {
042                    public EntryMap(final Comparator<PriceValue> comp) {
043                            super(comp);
044                    }
045            }
046    
047            private long millisUTC;
048    
049            private static final Comparator<PriceValue> CMP_ASC = new Comparator<PriceValue>() {
050                    @Override
051                    public int compare(final PriceValue o1, final PriceValue o2) {
052                            return o1.compareTo(o2);
053                    }
054            };
055    
056            private static final Comparator<PriceValue> CMP_DES = new Comparator<PriceValue>() {
057                    @Override
058                    public int compare(final PriceValue o1, final PriceValue o2) {
059                            return o2.compareTo(o1);
060                    }
061            };
062    
063            private final EntryMap bids = new EntryMap(CMP_ASC);
064            private final EntryMap asks = new EntryMap(CMP_DES);
065    
066            private MarketBookEntry topBid;
067            private MarketBookEntry topAsk;
068    
069            VarBookDDF(final MarketBookType type, final SizeValue size,
070                            final PriceValue step) {
071                    // XXX
072            }
073    
074            // #####################################
075    
076            @Override
077            public final UniBookResult setEntry(final MarketDoBookEntry entry) {
078    
079                    final MarketBookSide side = entry.side();
080    
081                    final int place = entry.place();
082    
083                    //
084    
085                    final EntryMap map = map(side);
086    
087                    switch (side) {
088                    case BID:
089                            if (place == ENTRY_TOP) {
090                                    topBid = entry;
091                            }
092                            break;
093                    case ASK:
094                            if (place == ENTRY_TOP) {
095                                    topAsk = entry;
096                            }
097                            break;
098                    default:
099                            return UniBookResult.ERROR;
100                    }
101    
102                    if (place == ENTRY_TOP) {
103                            return UniBookResult.TOP;
104                    } else {
105                            return UniBookResult.NORMAL;
106                    }
107    
108            }
109    
110            private EntryMap map(final MarketBookSide side) {
111                    switch (side) {
112                    default:
113                    case BID:
114                            return bids;
115                    case ASK:
116                            return asks;
117                    }
118            }
119    
120            @Override
121            public final MarketBookEntry[] entries(final MarketBookSide side) {
122    
123                    final EntryMap map = map(side);
124    
125                    final Collection<MarketBookEntry> values = map.values();
126    
127                    final int size = values.size();
128    
129                    final MarketBookEntry[] array = new MarketBookEntry[size];
130    
131                    int index = 0;
132    
133                    for (final MarketBookEntry entry : values) {
134                            array[index] = entry;
135                            index++;
136                    }
137    
138                    return array;
139    
140            }
141    
142            @Override
143            public final DefBook freeze() {
144                    return new DefBook(time(), entries(BID), entries(ASK), null, null);
145            }
146    
147            @Override
148            public TimeValue time() {
149                    return ValueBuilder.newTime(millisUTC);
150            }
151    
152            @Override
153            public final void setTime(final TimeValue time) {
154                    millisUTC = time.asMillisUTC();
155            }
156    
157            @Override
158            public final boolean isFrozen() {
159                    return false;
160            }
161    
162            @Override
163            public final MarketBookEntry last() {
164    
165                    final DefBookEntry entry = null; // XXX
166    
167                    return entry == null ? NULL_BOOK_ENTRY : entry;
168    
169            }
170    
171            @Override
172            public final MarketBookEntry top(final MarketBookSide side) {
173    
174                    final MarketBookEntry entry;
175                    switch (side) {
176                    case BID:
177                            entry = topBid;
178                            break;
179                    case ASK:
180                            entry = topAsk;
181                            break;
182                    default:
183                            entry = null;
184                            break;
185                    }
186    
187                    // System.err.println("### TOP : " + side + " ### " + entry);
188    
189                    return entry == null ? NULL_BOOK_ENTRY : entry;
190    
191            }
192    
193            /* #################################### */
194    
195            @Override
196            public PriceValue priceGap() {
197                    throw new UnsupportedOperationException("UNUSED");
198            }
199    
200            @Override
201            public final SizeValue[] sizes(final MarketBookSide side) {
202                    throw new UnsupportedOperationException("UNUSED");
203            }
204    
205            @Override
206            public PriceValue priceTop(final MarketBookSide side) {
207                    throw new UnsupportedOperationException("UNUSED");
208            }
209    
210            @Override
211            public SizeValue sizeTop(final MarketBookSide side) {
212                    throw new UnsupportedOperationException("UNUSED");
213            }
214    
215            /* #################################### */
216    
217            @Override
218            public void clear() {
219    
220                    bids.clear();
221                    asks.clear();
222    
223            }
224    
225            @Override
226            public UniBookResult setSnapshot(final MarketDoBookEntry[] entries) {
227    
228                    clear();
229    
230                    for (final MarketDoBookEntry entry : entries) {
231    
232                            final EntryMap map = map(entry.side());
233    
234                            map.put(entry.price(), entry);
235    
236                    }
237    
238                    return UniBookResult.NORMAL;
239    
240            }
241    
242    }
