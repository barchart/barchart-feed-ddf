004    package com.barchart.feed.ddf.datalink.api;
005    
006    import com.barchart.feed.client.api.FeedStateListener;
007    import com.barchart.util.anno.UsedOnce;
008    
009    /**
010     * Base interface for DDF feed clients. Classes directly implementing this
011     * interface should be connectionless listeners. Feeds requiring 2 way
012     * communication and a managed connection should implement DDF_FeedClient.
013     */
014    public interface DDF_FeedClientBase {
015    
016            /**
017             * Binds the feed client to a port or other data source and begins
018             * listening.
019             */
020            void startup();
021    
022            /**
023             * Stops listening to the data source.
024             */
025            void shutdown();
026    
027            /**
028             * Attaches a message listener to the client for data consumption.
029             */
030            @UsedOnce
031            void bindMessageListener(DDF_MessageListener msgListener);
032    
033            /**
034             * Attach single feed state listener to the client.
035             */
036            @UsedOnce
037            void bindStateListener(FeedStateListener stateListener);
038    
039    }
