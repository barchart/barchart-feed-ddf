008    package com.barchart.feed.ddf.datalink.provider;
009    
010    import java.util.concurrent.Executor;
011    
012    import org.slf4j.Logger;
013    import org.slf4j.LoggerFactory;
014    
015    import com.barchart.feed.ddf.datalink.api.DDF_FeedClient;
016    import com.barchart.feed.ddf.settings.enums.DDF_ServerType;
017    
018    /**
019     * Factory class for building FeedClientDDF.
020     * 
021     * @author g-litchfield
022     */
023    public class DDF_FeedClientFactory {
024    
025            private static final Logger log = LoggerFactory
026                            .getLogger(DDF_FeedClientFactory.class);
027    
028            private DDF_FeedClientFactory() {
029                    //
030            }
031    
032            /**
033             * Default factory method for DDF_FeedClient.
034             * <p>
035             * Defaults server type to STREAM.
036             * <p>
037             * Defaults channel executor to "create new thread and run."
038             * 
039             * @return The DDF_FeedClient
040             */
041            public static DDF_FeedClient newInstance() {
042    
043                    return new FeedClientDDF(DDF_ServerType.STREAM, new Executor() {
044                            @Override
045                            public void execute(final Runnable task) {
046                                    new Thread(task).start();
047                            }
048                    });
049            }
050    
051            /**
052             * Factory which defaults the DDF_ServerType to STREAM.
053             * 
054             * @param runner
055             *            The executor used by the ClientSocketChannel as both the boss
056             *            and worker executor. See
057             *            org.jboss.netty.channel.socket.nio.NioClientSocketChannelFactory
058             * @return the DDF_FeedClient
059             */
060            public static DDF_FeedClient newInstance(final Executor runner) {
061    
062                    log.debug("Built new DDF_FeedClient, default to DDF_ServerType.STREAM");
063    
064                    return new FeedClientDDF(DDF_ServerType.STREAM, runner);
065    
066            }
067    
068            /**
069             * Factory which allows user specified DDF_ServerType.
070             * 
071             * @param serverType
072             *            The server type
073             * @param runner
074             *            The executor used by the ClientSocketChannel as both the boss
075             *            and worker executor. See
076             *            org.jboss.netty.channel.socket.nio.NioClientSocketChannelFactory
077             * @return the DDF_FeedClient
078             */
079            public static DDF_FeedClient newInstance(final DDF_ServerType serverType,
080                            final Executor runner) {
081    
082                    log.debug("Built new DDF_FeedClient, DDF_ServerType.{}",
083                                    serverType.name());
084    
085                    return new FeedClientDDF(serverType, runner);
086    
087            }
088    
089    }
