284            /** The COM p_ code. */
285            public static Comparator<DDF_Session> COMP_CODE = new Comparator<DDF_Session>() {
286                    @Override
287                    public int compare(final DDF_Session o1, final DDF_Session o2) {
288                            if (o1 == null || o2 == null) {
289                                    return 0;
290                            }
291                            if (o1.code > o2.code) {
292                                    return +1;
293                            }
294                            if (o1.code < o2.code) {
295                                    return -1;
296                            }
297                            return 0;
298                    }
299            };

390            private static DDF_Session resolve(final DDF_Session known,
391                            final byte exchOrd) {
392    
393                    if (known.code == _G_) {
394                            /* duplicate session code in future and stock (different meaning) */
395                            final DDF_Exchange exch = DDF_Exchange.fromOrd(exchOrd);
396                            if (exch.kind == DDF_ExchangeKind.FUTURE) {
397                                    // electronic session for future
398                                    return $_G_NET;
399                            } else {
400                                    // default session for stock
401                                    return $_G_COM;
402                            }
403                    }
404    
405                    return known;
406    
407            }
408    
409            /**
410             * NOTE: ORD vs CODE.
411             *
412             * @param exchOrd the exch ord
413             * @param sessCode the sess code
414             * @return the dD f_ session
415             */
416            public final static DDF_Session fromPair(final byte exchOrd,
417                            final byte sessCode) {
418    
419                    for (final DDF_Session known : ENUM_VALS) {
420                            if (known.code == sessCode) {
421                                    return resolve(known, exchOrd);
422                            }
423                    }
424    
425                    // log.debug("### UNKNOWN session code : {}", sessCode);
426    
427                    return UNKNOWN;
428    
429            }
