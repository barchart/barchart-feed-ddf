001    /**
002     * 
003     * The core DDF class which encapsulates all the core functionality a new user
004     * will need to get started.
005     * <p>
006     * Instances are created using the public constructor. An optional parameter
007     * can provide an executor.
008     * <p>
009     * The price feed is started and stopped using the startup() and shutdown()
010     * methods. Note that these are non-blocking calls. Applications requiring
011     * actions upon successful login should instantiate and bind a
012     * FeedStatusListener to the client.  Note that UDP listeners (the default)
013     * will not fire any feed state events, they will just begin receiving data.
014     * <p>
015     * 
016     */
017    package com.barchart.feed.client.provider;
018    
019    import java.util.concurrent.Executor;
020    import java.util.concurrent.atomic.AtomicLong;
021    
022    import org.slf4j.Logger;
023    import org.slf4j.LoggerFactory;
024    
025    import com.barchart.feed.ddf.datalink.provider.DDF_FeedClientFactory;
026    
027    /**
028     * The entry point for Barchart data feed services.
029     */
030    public class BarchartFeedReceiver extends BarchartFeedClientBase {
031    
032            private static final Logger log = LoggerFactory
033                            .getLogger(BarchartFeedReceiver.class);
034    
035            private Executor executor = null;
036    
037            public BarchartFeedReceiver() {
038    
039                    this(new Executor() {
040    
041                            private final AtomicLong counter = new AtomicLong(0);
042    
043                            final String name = "# DDF Receiver - " + counter.getAndIncrement();
044    
045                            @Override
046                            public void execute(final Runnable task) {
047                                    new Thread(task, name).start();
048                            }
049    
050                    });
051    
052            }
053    
054            public BarchartFeedReceiver(final Executor ex) {
055                    executor = ex;
056            }
057    
058            /**
059             * Starts a stateless connection to the specified port. If the user is
060             * already logged in, this call will end the previous connection and reset
061             * all registered market takers.
062             * 
063             * @param socketAddress
064             */
065            public void listen(final int socketAddress) {
066    
067                    startListener(socketAddress);
068    
069            }
070    
071            /**
072             * Starts a stateless connection to the specified port. If the user is
073             * already logged in, this call will end the previous connection and reset
074             * all registered market takers.
075             * 
076             * @param socketAddress
077             */
078            private void startListener(final int socketAddress) {
079    
080                    setClient(DDF_FeedClientFactory.newStatelessListenerClient(
081                                    socketAddress, executor));
082    
083            }
084    
085    }
