001    /**
002     * 
003     * The core DDF class which encapsulates all the core functionality a new user
004     * will need to get started.
005     * <p>
006     * Instances are created using the public constructor and require a
007     * valid user name and password. Optional parameters include specifying the
008     * transport protocol and providing an executor.
009     * <p>
010     * The price feed is started and stopped using the startup() and shutdown()
011     * methods. Note that these are non-blocking calls. Applications requiring
012     * actions upon successful login should instantiate and bind a
013     * FeedStatusListener to the client.
014     * <p>
015     * 
016     * 
017     */
018    package com.barchart.feed.client.provider;
019    
020    import java.util.ArrayList;
021    import java.util.List;
022    import java.util.concurrent.CopyOnWriteArrayList;
023    
024    import org.slf4j.Logger;
025    import org.slf4j.LoggerFactory;
026    
027    import com.barchart.feed.base.instrument.values.MarketInstrument;
028    import com.barchart.feed.base.market.api.MarketTaker;
029    import com.barchart.feed.base.market.enums.MarketField;
030    import com.barchart.feed.client.api.FeedStateListener;
031    import com.barchart.feed.client.api.TimestampListener;
032    import com.barchart.feed.ddf.datalink.api.DDF_FeedClientBase;
033    import com.barchart.feed.ddf.datalink.api.DDF_MessageListener;
034    import com.barchart.feed.ddf.instrument.api.DDF_Instrument;
035    import com.barchart.feed.ddf.instrument.provider.DDF_InstrumentProvider;
036    import com.barchart.feed.ddf.market.api.DDF_MarketProvider;
037    import com.barchart.feed.ddf.market.provider.DDF_MarketService;
038    import com.barchart.feed.ddf.message.api.DDF_BaseMessage;
039    import com.barchart.feed.ddf.message.api.DDF_ControlTimestamp;
040    import com.barchart.feed.ddf.message.api.DDF_MarketBase;
041    import com.barchart.util.values.api.Value;
042    
043    /**
044     * The entry point for Barchart data feed services.
045     */
046    public abstract class BarchartFeedClientBase {
047    
048            private static final Logger log = LoggerFactory
049                            .getLogger(BarchartFeedClientBase.class);
050    
051            private volatile DDF_FeedClientBase feed = null;
052    
053            protected final DDF_MarketProvider maker = DDF_MarketService.newInstance();
054    
055            private final CopyOnWriteArrayList<TimestampListener> timeStampListeners = new CopyOnWriteArrayList<TimestampListener>();
056    
057            private FeedStateListener stateListener;
058    
059            public BarchartFeedClientBase() {
060            }
061    
062            /*
063             * Handles login. Non-blocking.
064             */
065            protected void setClient(final DDF_FeedClientBase client) {
066    
067                    maker.clearAll();
068    
069                    if (feed != null) {
070                            feed.shutdown();
071                    }
072    
073                    feed = client;
074    
075                    feed.bindMessageListener(msgListener);
076    
077                    if (stateListener != null) {
078                            feed.bindStateListener(stateListener);
079                    }
080    
081                    feed.startup();
082    
083            }
084    
085            /**
086             * Shuts down the data feed and clears all registered market takers.
087             */
088            public void shutdown() {
089    
090                    maker.clearAll();
091    
092                    if (feed != null) {
093                            feed.shutdown();
094                            feed = null;
095                    }
096    
097            }
098    
099            /*
100             * This is the default message listener. Users wishing to handle raw
101             * messages will need to implement their own feed client.
102             */
103            private final DDF_MessageListener msgListener = new DDF_MessageListener() {
104    
105                    @Override
106                    public void handleMessage(final DDF_BaseMessage message) {
107    
108                            if (message instanceof DDF_ControlTimestamp) {
109                                    for (final TimestampListener listener : timeStampListeners) {
110                                            listener.handleTimestamp(((DDF_ControlTimestamp) message)
111                                                            .getStampUTC());
112                                    }
113                            }
114    
115                            if (message instanceof DDF_MarketBase) {
116                                    final DDF_MarketBase marketMessage = (DDF_MarketBase) message;
117                                    maker.make(marketMessage);
118                            }
119    
120                    }
121    
122            };
123    
124            /**
125             * Applications which need to react to the connectivity state of the feed
126             * instantiate a FeedStateListener and bind it to the client.
127             * 
128             * @param listener
129             *            The listener to be bound.
130             */
131            public void bindFeedStateListener(final FeedStateListener listener) {
132    
133                    stateListener = listener;
134    
135                    if (feed != null) {
136                            feed.bindStateListener(listener);
137                    }
138    
139            }
140    
141            /**
142             * Applications which require time-stamp or heart-beat messages from the
143             * data server instantiate a DDF_TimestampListener and bind it to the
144             * client.
145             * 
146             * @param listener
147             */
148            public void bindTimestampListener(final TimestampListener listener) {
149                    timeStampListeners.add(listener);
150            }
151    
152            /**
153             * Adds a market taker to the client. This performs instrument registration
154             * with the market maker as well as subscribing to the required data from
155             * the feed.
156             * 
157             * @param taker
158             *            The market taker to be added.
159             * @return True if the taker was successfully added.
160             */
161            public boolean isTakerRegistered(final MarketTaker<?> taker) {
162                    return maker.isRegistered(taker);
163            }
164    
165            /**
166             * Adds a market taker to the client. This performs instrument registration
167             * with the market maker as well as subscribing to the required data from
168             * the feed.
169             * 
170             * @param taker
171             *            The market taker to be added.
172             * @return True if the taker was successfully added.
173             */
174            public <V extends Value<V>> boolean addTaker(final MarketTaker<V> taker) {
175                    return maker.register(taker);
176            }
177    
178            /**
179             * Removes a market taker from the client. If no other takers require its
180             * instruments, they are unsubscribed from the feed.
181             * 
182             * @param taker
183             *            THe market taker to be removed.
184             * @return True if the taker was successfully removed.
185             */
186            public <V extends Value<V>> boolean removeTaker(final MarketTaker<V> taker) {
187                    return maker.unregister(taker);
188            }
189    
190            /**
191             * Retrieves the instrument object denoted by symbol. The local instrument
192             * cache will be checked first. If the instrument is not stored locally, a
193             * remote call to the instrument service is made.
194             * 
195             * @return NULL_INSTRUMENT if the symbol is not resolved.
196             */
197            public MarketInstrument lookup(final String symbol) {
198                    return DDF_InstrumentProvider.find(symbol);
199            }
200    
201            /**
202             * Retrieves a list of instrument objects denoted by symbols provided. The
203             * local instrument cache will be checked first. If any instruments are not
204             * stored locally, a remote call to the instrument service is made.
205             * 
206             * @return An empty list if no symbols can be resolved.
207             */
208            public List<MarketInstrument> lookup(final List<String> symbolList) {
209                    final List<DDF_Instrument> list = DDF_InstrumentProvider
210                                    .find(symbolList);
211    
212                    return new ArrayList<MarketInstrument>(list);
213            }
214    
215            /**
216             * Makes a query to the market maker for a snapshot of a market field for a
217             * specific instrument. The returned values are frozen and disconnected from
218             * live market.
219             * 
220             * @return NULL_VALUE for all fields if market is not present.
221             */
222            public <S extends MarketInstrument, V extends Value<V>> V take(
223                            final S instrument, final MarketField<V> field) {
224                    return maker.take(instrument, field);
225            }
226    
227    }
