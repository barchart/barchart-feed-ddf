013    import org.slf4j.Logger;
014    import org.slf4j.LoggerFactory;
015    
016    import com.barchart.feed.base.bar.api.MarketBar;
017    import com.barchart.feed.base.bar.enums.MarketBarField;
018    import com.barchart.feed.base.instrument.values.MarketInstrument;
019    import com.barchart.feed.base.market.api.Market;

021    import com.barchart.feed.base.market.enums.MarketEvent;
022    import com.barchart.feed.base.market.enums.MarketField;
023    import com.barchart.feed.base.state.enums.MarketStateEntry;
024    import com.barchart.feed.client.api.FeedStateListener;
025    import com.barchart.feed.client.enums.FeedState;
026    import com.barchart.feed.client.provider.BarchartFeedClient;
027    
028    /**
029     * 
030     * Stress test to try and break the login/logout lifecycle
031     */
032    public class TestBarchartFeedClient {
033    
034            private static final Logger log = LoggerFactory
035                            .getLogger(TestBarchartFeedClient.class);
036    
037            /**
038             * @param args
039             */
040            public static void main(final String[] args) throws Exception {
041    
042                    final String username = System.getProperty("barchart.username");
043                    final String password = System.getProperty("barchart.password");
044    
045                    final BarchartFeedClient client = new BarchartFeedClient();
046    
047                    final MarketInstrument[] instruments = { client.lookup("GOOG")};
048                    final FeedStateListener feedListener = new FeedStateListener() {
049    
050                            @Override
051                            public void stateUpdate(final FeedState state) {
052    
053                                    if (state == FeedState.LOGGED_IN) {
054                                            client.addTaker(TakerFactory.makeFactory(instruments));
055                                    }
056    
057                            }
058    
059                    };
060    
061                    client.login(username, password);
062                    
063                    client.bindFeedStateListener(feedListener);
064    
065                    try {
066                            while (true) {
067                                    Thread.sleep(1000);
068                            }
069                    } catch (final Exception e) {
070                            // Interrupted
071                    }
072    
073                    client.shutdown();
074    
075            }
076    
077            private static class TakerFactory {
078    
079                    static MarketTaker<Market> makeFactory(
080                                    final MarketInstrument[] instruments) {
081                            
082                            return new MarketTaker<Market>() {
083    
084                                    @Override
085                                    public MarketField<Market> bindField() {
086                                            return MarketField.MARKET;
087                                    }
088    
089                                    @Override
090                                    public MarketEvent[] bindEvents() {
091    
092                                            //return MarketEvent.in(MarketEvent.values());
093                                            return new MarketEvent[] { MarketEvent.MARKET_UPDATED };
094    
095                                    }
096    
097                                    @Override
098                                    public MarketInstrument[] bindInstruments() {
099    
100                                            return instruments;
101    
102                                    }
103    
104                                    @Override
105                                    public void onMarketEvent(final MarketEvent event,
106                                                    final MarketInstrument instrument, final Market value) {
107    
108                                            final StringBuilder sb = new StringBuilder("Event: ")
109                                                            .append(event);
110    
111                                            final MarketBar barCurrent = value
112                                                            .get(MarketField.BAR_CURRENT);
113    
114                                            if (!barCurrent.isNull()) {
115                                                    sb.append("; price=")
116                                                                    .append(barCurrent.get(MarketBarField.CLOSE)
117                                                                                    .mantissa())
118                                                                    .append("; time=")
119                                                                    .append(barCurrent.get(MarketBarField.BAR_TIME)
120                                                                                    .asDateTime())
121                                                                    .append("; day=")
122                                                                    .append(barCurrent.get(
123                                                                                    MarketBarField.TRADE_DATE).asDateTime())
124                                                                    .append("; settled="
125                                                                                    + value.get(MarketField.STATE)
126                                                                                                    .contains(
127                                                                                                                    MarketStateEntry.IS_SETTLED));
128                                            }
129    
130                                            log.debug(sb.toString());
131    
132                                    }
133    
134                            };
135    
136                    }
137            }
