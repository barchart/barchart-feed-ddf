011    /**
012     * The Enum DDF_Option.
013     */
014    public enum DDF_Option {
015    
016            /*
017             * CALL/PUT IND 1 Byte, Alphabetic, indication of call or put for Options or
018             * Futures
019             */
020    
021            CALL('C'), //
022    
023            PUT('P'), // 
024    
025            UNKNOWN('?'), //
026    
027            /** The code. */
028     ;
029    
030            public final char code;
031    
032            private DDF_Option(final char code) {
033                    this.code = code;
034            }
035    
036            /**
037             * From code.
038             *
039             * @param code the code
040             * @return the dD f_ option
041             */
042            public final static DDF_Option fromCode(final char code) {
043                    switch (code) {
044                    case 'c':
045                    case 'C':
046                            return CALL;
047                    case 'p':
048                    case 'P':
049                            return PUT;
050                    default:
051                            return UNKNOWN;
052                    }
053            }
054    
055            /**
056             * Checks if is valid.
057             *
058             * @param code the code
059             * @return true, if is valid
060             */
061            public final static boolean isValid(final char code) {
062                    final DDF_Option indicator = fromCode(code);
063                    return indicator != UNKNOWN;
064            }
065    
066            /**
067             * From option year.
068             *
069             * @param indicator the indicator
070             * @return the dD f_ option
071             */
072            public final static DDF_Option fromOptionYear(final DDF_OptionYear indicator) {
073                    switch (indicator) {
074                    case CALL_0:
075                    case CALL_1:
076                    case CALL_2:
077                    case CALL_3:
078                    case CALL_4:
079                    case CALL_5:
080                    case CALL_6:
081                    case CALL_7:
082                    case CALL_8:
083                    case CALL_9:
084                            return CALL;
085                    case PUT_0:
086                    case PUT_1:
087                    case PUT_2:
088                    case PUT_3:
089                    case PUT_4:
090                    case PUT_5:
091                    case PUT_6:
092                    case PUT_7:
093                    case PUT_8:
094                    case PUT_9:
095                            return PUT;
096                    default:
097                            return UNKNOWN;
098                    }
099            }
