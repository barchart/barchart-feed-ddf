487 	/**
488 	 * Special casing for duplicate session codes that are handled different
489 	 * across exchanges
490 	 */
491 	private static DDF_Session resolve(final DDF_Session known,
492 			final byte exchOrd) {
493 
494 		if (known.code == _G_) {
495 			/* duplicate session code in future and stock (different meaning) */
496 			final DDF_Exchange exch = DDF_Exchange.fromOrd(exchOrd);
497 			if (exch.kind == DDF_ExchangeKind.FUTURE) {
498 				// electronic session for future
499 				return $_G_NET;
500 			} else {
501 				// default session for stock
502 				return $_G_COM;
503 			}
504 		}
505 
506 		return known;
507 
508 	}
