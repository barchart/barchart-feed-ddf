019            public static byte[] stripSpreadPreamble(final byte[] ba) {
020                    // _SCLN9,8_AJ10SP2CLV9,-232,-,,,GG_
021                    // 0123456789a123456789b123456789d123456789e
022    
023                    int pos = getIndexOf(ba, ',', 0);
024    
025                    final String symbol = parseStringValue(ba, 2, pos - 2);
026                    final char subrecord = (char) ba[pos + 1];
027    
028                    final int spStart = pos + 7;
029                    final String spreadType = parseStringValue(ba, pos + 7, 2);
030                    final int numberOfLegs = Character.getNumericValue(ba[pos + 9]);
031                    final String[] legs = new String[numberOfLegs];
032                    legs[0] = symbol;
033    
034                    pos += 10;
035                    for (int i = 1; i < numberOfLegs; i++) {
036                            final int pos2 = getIndexOf(ba, ',', pos);
037                            legs[i] = parseStringValue(ba, pos, pos2 - pos);
038                            pos = pos2 + 1;
039                    }
040    
041                    int start2 = pos;
042                    switch (subrecord) {
043                    case '1':
044                    case '2':
045                    case '3':
046                    case '4':
047                    case '5':
048                            // N.B. Keep the comma
049                            start2--;
050                            break;
051                    }
052                    final int length = spStart + (ba.length - start2);
053                    final byte[] ba2 = new byte[length];
054    
055                    System.arraycopy(ba, 0, ba2, 0, spStart);
056                    System.arraycopy(ba, start2, ba2, spStart, ba.length - start2);
057    
058                    final StringBuilder sb = new StringBuilder("_S_" + spreadType);
059                    for (final String s : legs) {
060                            sb.append("_" + s);
061                    }
062    
063                    final byte[] ba3 = new byte[ba2.length - symbol.length() + sb.length()];
064                    ba3[0] = 1;
065                    ba3[1] = 50;
066    
067                    for (int i = 0; i < sb.length(); i++) {
068                            ba3[2 + i] = (byte) sb.charAt(i);
069                    }
070    
071                    System.arraycopy(ba2, 2 + symbol.length(), ba3, 2 + sb.length(),
072                                    ba3.length - (2 + sb.length()));
073    
074                    return ba3;
075            }
076    
077            private static int
078                            getIndexOf(final byte[] b, final char c, final int offset) {
079    
080                    if (b.length < offset) {
081                            throw new IndexOutOfBoundsException(
082                                            "Offset greater than array length");
083                    }
084    
085                    for (int i = offset; i < b.length; i++) {
086                            if (b[i] == c) {
087                                    return i;
088                            }
089                    }
090    
091                    return -1;
092            }
093    
094            private static String parseStringValue(final byte[] b, final int start,
095                            final int length) {
096    
097                    if (b.length < start + length) {
098                            throw new IndexOutOfBoundsException(
099                                            "Requested string overflows array");
100                    }
101    
102                    final StringBuffer sb = new StringBuffer();
103    
104                    for (int i = start; i < start + length; i++) {
105                            sb.append((char) b[i]);
106                    }
107    
108                    return sb.toString();
109            }
