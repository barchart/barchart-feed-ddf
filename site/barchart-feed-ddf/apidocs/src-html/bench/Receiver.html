008    package bench;
009    
010    import com.barchart.feed.api.Agent;
011    import com.barchart.feed.api.Marketplace;
012    import com.barchart.feed.api.MarketObserver;
013    import com.barchart.feed.api.model.data.Book.Entry;
014    import com.barchart.feed.api.model.data.Book.Top;
015    import com.barchart.feed.api.model.data.Market;
016    import com.barchart.feed.client.provider.BarchartMarketplace;
017    
018    /**
019     * 
020     * you can invoke this utility from command line:
021     * 
022     * <pre>
023     * java - jar barchart-feed-ddf-client-X.Y.Z.jar bench.Receiver TCP 7500 IBM
024     * </pre>
025     * 
026     * arguments = { TCP / UDP , port, instrument1, instrument2, ...}
027     * 
028     */
029    public class Receiver {
030    
031            // TODO
032            
033            /**
034             * 
035             * @param args
036             * @throws Exception
037             */
038            public static void main(final String... args) throws Exception {
039    
040                    if (args == null || args.length < 3) {
041                            throw new RuntimeException("Bad arguments passed to main");
042                    }
043                    
044                    final Marketplace feed = new BarchartMarketplace(args[0], args[1]);
045    
046                    final MarketObserver<Market> observer = new MarketObserver<Market>() {
047    
048                            @Override
049                            public void onNext(Market value) {
050                                    
051                                    final StringBuilder sb = new StringBuilder(value.instrument().marketGUID())
052                                            .append(" ")
053                                            .append(" EventTime=")
054                                            .append(value.updated());
055    
056                                    final Top top = value.book().top();
057                                    
058                                    Entry entry = top.ask();
059                                    
060                                    if (!entry.isNull()) {
061                                            sb.append(" ASK TOP").append(" price=")
062                                                            .append(entry.price().asDouble())
063                                                            .append(" qty=").append(entry.size().asDouble());
064                                    }
065                                    
066                                    entry = top.bid();
067                                    
068                                    if (!entry.isNull()) {
069                                            sb.append(" BID TOP").append(" price=")
070                                                            .append(entry.price().asDouble())
071                                                            .append(" qty=").append(entry.size().asDouble());
072                                    }
073    
074                                    System.out.println(sb.toString());
075    
076                                    
077                            }
078                            
079                    };
080                    
081                    feed.startup();
082                    
083                    final Agent myAgent = feed.newAgent(Market.class, observer);
084                    
085                    for(int i = 2; i < args.length; i++) {
086                            myAgent.include(args[i]);
087                    }
088    
089                    System.in.read();
090                    feed.shutdown();
091                    System.exit(0);
092    
093            }
094    
095    }
