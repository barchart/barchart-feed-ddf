011    import com.barchart.feed.base.book.api.MarketBookEntry;

021    import com.barchart.util.values.util.ValueUtil;

023    /**
024     * 
025     * you can invoke this utility from command line:
026     * 
027     * <pre>
028     * java - jar barchart-feed-ddf-client-X.Y.Z.jar bench.Receiver TCP 7500 IBM
029     * </pre>
030     * 
031     * arguments = { TCP / UDP , port, instrument1, instrument2, ...}
032     * 
033     */
034    public class Receiver {
035    
036            /**
037             * 
038             * @param args
039             * @throws Exception
040             */
041            public static void main(final String... args) throws Exception {
042    
043                    final int INST_START = 2;
044    
045                    if (args == null) {
046                            throw new RuntimeException("No arguments passed to main");
047                    }
048    
049                    final BarchartFeedReceiver client = new BarchartFeedReceiver();
050    
051                    final MarketInstrument[] instruments = new MarketInstrument[args.length
052                                    - INST_START];
053    
054                    for (int i = INST_START; i < args.length; i++) {
055                            instruments[i - INST_START] = client.lookup(args[i]);
056                    }
057    
058                    if (args[0].equals("TCP")) {
059                            client.listenTCP(Integer.parseInt(args[1]), false);
060                    } else if (args[0].equals("UDP")) {
061                            client.listenUDP(Integer.parseInt(args[1]), false);
062                    } else {
063                            throw new RuntimeException("Bad protocol, expecting UDP or TCP");
064                    }
065    
066                    client.addAllMarketsTaker(TakerFactory.makeTaker(instruments));
067    
068                    System.in.read();
069                    client.shutdown();
070                    System.exit(0);
071    
072            }
073    
074            private static class TakerFactory {
075    
076                    static MarketTaker<Market> makeTaker(
077                                    final MarketInstrument[] instruments) {
078    
079                            return new MarketTaker<Market>() {
080    
081                                    @Override
082                                    public MarketField<Market> bindField() {
083                                            return MarketField.MARKET;
084                                    }
085    
086                                    @Override
087                                    public MarketEvent[] bindEvents() {
088    
089                                            return new MarketEvent[] { MarketEvent.MARKET_UPDATED };
090    
091                                    }
092    
093                                    @Override
094                                    public MarketInstrument[] bindInstruments() {
095    
096                                            return instruments;
097    
098                                    }
099    
100                                    @Override
101                                    public void onMarketEvent(final MarketEvent event,
102                                                    final MarketInstrument instrument, final Market value) {
103    
104                                            final StringBuilder sb = new StringBuilder(value.get(
105                                                            MarketField.INSTRUMENT).get(InstrumentField.ID))
106                                                            .append(" ")
107                                                            .append(event)
108                                                            .append(" EventTime=")
109                                                            .append(value.get(MarketField.MARKET_TIME)
110                                                                            .asDateTime().toString());
111    
112                                            final MarketBookTop top = value.get(MarketField.BOOK_TOP);
113    
114                                            MarketBookEntry entry = top.side(MarketBookSide.ASK);
115    
116                                            if (!entry.isNull()) {
117                                                    sb.append(" ASK TOP").append(" price=")
118                                                                    .append(ValueUtil.asDouble(entry.price()))
119                                                                    .append(" qty=").append(entry.size().asLong());
120                                            }
121    
122                                            entry = top.side(MarketBookSide.BID);
123    
124                                            if (!entry.isNull()) {
125                                                    sb.append(" BID TOP").append(" price=")
126                                                                    .append(ValueUtil.asDouble(entry.price()))
127                                                                    .append(" qty=").append(entry.size().asLong());
128                                            }
129    
130                                            final MarketBar barCurrent = value
131                                                            .get(MarketField.BAR_CURRENT);
132    
133                                            System.out.println(sb.toString());
134    
135                                    }
136    
137                            };
138    
139                    }
140            }
141
