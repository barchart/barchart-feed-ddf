350            /** The COM p_ code. */
351            public static Comparator<DDF_Session> COMP_CODE = new Comparator<DDF_Session>() {
352                    @Override
353                    public int compare(final DDF_Session o1, final DDF_Session o2) {
354                            if (o1 == null || o2 == null) {
355                                    return 0;
356                            }
357                            if (o1.code > o2.code) {
358                                    return +1;
359                            }
360                            if (o1.code < o2.code) {
361                                    return -1;
362                            }
363                            return 0;
364                    }
365            };

487            /**
488             * Special casing for duplicate session codes that are handled different
489             * across exchanges
490             */
491            private static DDF_Session resolve(final DDF_Session known,
492                            final byte exchOrd) {
493    
494                    if (known.code == _G_) {
495                            /* duplicate session code in future and stock (different meaning) */
496                            final DDF_Exchange exch = DDF_Exchange.fromOrd(exchOrd);
497                            if (exch.kind == DDF_ExchangeKind.FUTURE) {
498                                    // electronic session for future
499                                    return $_G_NET;
500                            } else {
501                                    // default session for stock
502                                    return $_G_COM;
503                            }
504                    }
505    
506                    return known;
507    
508            }

510            /**
511             * NOTE: ORD vs CODE.
512             * 
513             * @param exchOrd
514             *            the exch ord
515             * @param sessCode
516             *            the sess code
517             * @return the dD f_ session
518             */
519            public final static DDF_Session fromPair(final byte exchOrd,
520                            final byte sessCode) {
521    
522                    for (final DDF_Session known : ENUM_VALS) {
523                            if (known.code == sessCode) {
524                                    return resolve(known, exchOrd);
525                            }
526                    }
527    
528                    // log.debug("### UNKNOWN session code : {}", sessCode);
529    
530                    return UNKNOWN;
531    
532            }
