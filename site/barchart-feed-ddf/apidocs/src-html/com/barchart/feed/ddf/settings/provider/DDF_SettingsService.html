008    package com.barchart.feed.ddf.settings.provider;
009    
010    import org.slf4j.Logger;
011    import org.slf4j.LoggerFactory;
012    import org.w3c.dom.Element;
013    
014    import com.barchart.feed.ddf.settings.api.DDF_Settings;
015    import com.barchart.feed.ddf.util.HelperXML;
016    
017    /**
018     * Provides static factory methods for building DDF_Settings objects.
019     * <p>
020     * This will be the primary entry point for a user wishing to initiate a login
021     * to a DDF data feed.
022     * 
023     */
024    public final class DDF_SettingsService {
025    
026            private static final Logger log = LoggerFactory
027                            .getLogger(DDF_SettingsService.class);
028    
029            /** The Constant NULL_SETTINGS. */
030            public final static DDF_Settings NULL_SETTINGS = new SettingsDDF("");
031    
032            private DDF_SettingsService() {
033            }
034    
035            /**
036             * Basic static factory method for creating a DDF_Settings object.
037             * <p>
038             * Automatically pulls user settings from internal Barchart client info uri.
039             * 
040             * @param username
041             *            User's Barchart given user name.
042             * @param password
043             *            User's Barchart given password.
044             * @return DDF_Settings object instantiated with user's settings.
045             */
046            public static DDF_Settings newSettings(final String username,
047                            final String password) {
048    
049                    final String settingsURI = ConstSettingsDDF.urlCentral(username,
050                                    password);
051    
052                    return newSettings(settingsURI, username, password);
053    
054            }
055    
056            /**
057             * Static factory method for creating a DDF_Settings object with a user
058             * denoted uri from which to pull settings data.
059             * 
060             * @param xmlURI
061             *            URI of user settings
062             * @param username
063             *            User's Barchart given user name.
064             * @param password
065             *            User's Barchart given password.
066             * @return DDF_Settings object instantiated with user's settings.
067             */
068            static final DDF_Settings newSettings(final String xmlURI,
069                            final String username, final String password) {
070    
071                    final Element element;
072                    try {
073                            element = HelperXML.xmlDocumentDecode(xmlURI);
074                    } catch (final Exception e) {
075                            log.debug("", e);
076                            final String comment = e.getClass().getSimpleName() + " : "
077                                            + e.getMessage();
078                            return new SettingsDDF(comment);
079                    }
080    
081                    return new SettingsDDF(element, username, password);
082    
083            }
084    
085            /**
086             * New settings mock valid.
087             * 
088             * @param username
089             *            User's Barchart given user name.
090             * @param password
091             *            User's Barchart given password.
092             * @return DDF_Settings object instantiated with user's settings.
093             */
094            public static DDF_Settings newSettingsMockValid(final String username,
095                            final String password) {
096                    final String fileName = "ddf-settings-response-valid.xml";
097                    final String xmlURI = ConstSettingsDDF.class.getClassLoader()
098                                    .getResource(fileName).toExternalForm();
099                    return newSettings(xmlURI, username, password);
100            }
101    
102            /**
103             * New settings mock not valid.
104             * 
105             * @param username
106             *            User's Barchart given user name.
107             * @param password
108             *            User's Barchart given password.
109             * @return DDF_Settings object instantiated with user's settings.
110             */
111            public static DDF_Settings newSettingsMockNotValid(final String username,
112                            final String password) {
113                    final String fileName = "ddf-settings-response-not-valid.xml";
114                    final String xmlURI = ConstSettingsDDF.class.getClassLoader()
115                                    .getResource(fileName).toExternalForm();
116                    return newSettings(xmlURI, username, password);
117            }
118    
119    }
