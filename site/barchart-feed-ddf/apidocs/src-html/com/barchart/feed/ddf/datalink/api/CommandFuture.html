011    package com.barchart.feed.ddf.datalink.api;
012    
013    import java.util.concurrent.ExecutionException;
014    import java.util.concurrent.Future;
015    import java.util.concurrent.TimeUnit;
016    import java.util.concurrent.TimeoutException;
017    
018    import org.jboss.netty.channel.ChannelFuture;
019    
020    /**
021     * Wrapper for a Netty ChannelFuture.
022     * 
023     */
024    public class CommandFuture implements Future<Boolean> {
025    
026            private final ChannelFuture future;
027    
028            public CommandFuture(final ChannelFuture channelFuture) {
029    
030                    if (channelFuture == null) {
031                            throw new NullPointerException("ChannelFuture was null");
032                    }
033                    future = channelFuture;
034            }
035    
036            @Override
037            public boolean cancel(final boolean arg0) {
038                    return future.cancel();
039            }
040    
041            @Override
042            public Boolean get() throws InterruptedException, ExecutionException {
043                    future.await();
044                    return future.isSuccess();
045            }
046    
047            @Override
048            public Boolean get(final long timeout, final TimeUnit unit)
049                            throws InterruptedException, ExecutionException, TimeoutException {
050                    future.awaitUninterruptibly(timeout, unit);
051                    return future.isSuccess();
052            }
053    
054            @Override
055            public boolean isCancelled() {
056                    return future.isCancelled();
057            }
058    
059            @Override
060            public boolean isDone() {
061                    return future.isDone();
062            }
063    
064    }
