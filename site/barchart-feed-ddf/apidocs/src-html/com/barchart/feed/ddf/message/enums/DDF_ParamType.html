147            private static final Logger log = LoggerFactory
148                            .getLogger(DDF_ParamType.class);
149    
150            /**
151             * The Enum Kind.
152             */
153            public enum Kind {
154                    PRICE, SIZE, NONE
155            }
156    
157            /* (non-Javadoc)
158             * @see com.barchart.util.enums.EnumByteOrdinal#ord()
159             */
160            @Override
161            public final byte ord() {
162                    return ord;
163            }
164    
165            /* (non-Javadoc)
166             * @see com.barchart.util.enums.EnumCodeChar#code()
167             */
168            @Override
169            public final char code() {
170                    return code;
171            }
172    
173            /** The ord. */
174            public final byte ord;
175    
176            /** The element. */
177            public final DDF_ParamElement element;
178            
179            /** The modifier. */
180            public final DDF_ParamModifier modifier;
181    
182            /** combination of 2 element + modifier. */
183            public final char code;
184    
185            /** The kind. */
186            public final Kind kind;
187    
188            private DDF_ParamType(final DDF_ParamElement element,
189                            final DDF_ParamModifier modifier, final Kind kind) {
190    
191                    this.ord = (byte) ordinal();
192    
193                    this.element = element;
194                    this.modifier = modifier;
195    
196                    this.code = ByteConverters.charFromBytes(element.code, modifier.code);
197    
198                    this.kind = kind;
199    
200            }
201    
202            private final static DDF_ParamType[] ENUM_VALUES = values();
203    
204            /**
205             * Values unsafe.
206             *
207             * @return the dD f_ param type[]
208             */
209            @Deprecated
210            public final static DDF_ParamType[] valuesUnsafe() {
211                    return ENUM_VALUES;
212            }
213    
214            static {
215                    // validate use of byte ord
216                    MathExtra.castIntToByte(ENUM_VALUES.length);
217            }
218    
219            /**
220             * From ord.
221             *
222             * @param ord the ord
223             * @return the dD f_ param type
224             */
225            public final static DDF_ParamType fromOrd(final byte ord) {
226                    return ENUM_VALUES[ord];
227            }
228    
229            /**
230             * From pair.
231             *
232             * @param element the element
233             * @param modifier the modifier
234             * @return the dD f_ param type
235             */
236            public final static DDF_ParamType fromPair(final byte element,
237                            final byte modifier) {
238                    return fromCode(ByteConverters.charFromBytes(element, modifier));
239            }
240    
241            /**
242             * From code.
243             *
244             * @param code the code
245             * @return the dD f_ param type
246             */
247            public final static DDF_ParamType fromCode(final char code) {
248                    for (final DDF_ParamType known : ENUM_VALUES) {
249                            if (known.code == code) {
250                                    return known;
251                            }
252                    }
253                    log.debug("UNKNOWN PARAM CODE = " + code);
254                    return UNKNOWN;
255            }
256    
257            /**
258             * Checks if is known.
259             *
260             * @return true, if is known
261             */
262            public final boolean isKnown() {
263                    return this != UNKNOWN;
264            }
265    
266            /* (non-Javadoc)
267             * @see java.lang.Enum#toString()
268             */
269            @Override
270            public String toString() {
271                    return String.format("%c%c   %-20s", element.code, modifier.code,
272                                    name());
273            }
274    
275            /**
276             * The main method.
277             *
278             * @param args the arguments
279             */
280            public static void main(final String... args) {
281    
282                    log.debug("init");
283    
284                    for (final DDF_ParamType known : ENUM_VALUES) {
285                            System.out.println(known);
286                    }
287    
288                    log.debug("done");
289    
290            }
