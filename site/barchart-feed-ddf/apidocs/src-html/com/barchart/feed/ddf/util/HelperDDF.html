048            // from 10-1 (mantissa=101, fraction=1/4)
049            // into 10.25 (mantissa=1025, fraction=1/100)
050            /**
051             * From binary to decimal.
052             * 
053             * @param value
054             *            the value
055             * @param frac
056             *            the frac
057             * @return the long
058             */
059            public final static long fromBinaryToDecimal(final long value,
060                            final DDF_Fraction frac) {
061    
062                    if (value == HelperDDF.DDF_EMPTY || value == HelperDDF.DDF_CLEAR) {
063                            return value;
064                    }
065    
066                    if (frac.isBinary) {
067    
068                            //
069                            long whole = value;
070                            // shift right binary
071                            whole /= frac.spacesDenomPlus;
072                            // shift left decimal
073                            whole = MathExtra.longMult(whole, frac.decimalDenominator);
074    
075                            //
076                            long part = value;
077                            // partial shift left decimal
078                            part = MathExtra.longMult(part, frac.spacesDenomMinus);
079                            // remove whole part
080                            part -= whole;
081                            // remaining shift left decimal
082                            part = MathExtra.longMult(part, frac.spacesDenomPlus);
083                            // convert binary fraction to decimal fraction
084                            part /= frac.nativeDenominator;
085    
086                            // merge whole and part
087                            return MathExtra.longAdd(whole, part);
088    
089                    } else {
090    
091                            return value;
092    
093                    }
094    
095            }

097            /**
098             * From decimal to binary.
099             * 
100             * @param value
101             *            the value
102             * @param frac
103             *            the frac
104             * @return the long
105             */
106            public final static long fromDecimalToBinary(final long value,
107                            final DDF_Fraction frac) {
108    
109                    if (value == HelperDDF.DDF_EMPTY || value == HelperDDF.DDF_CLEAR) {
110                            return value;
111                    }
112    
113                    if (frac.isBinary) {
114    
115                            long whole = value;
116    
117                            whole /= frac.decimalDenominator;
118    
119                            whole = MathExtra.longMult(whole, frac.spacesDenomPlus);
120    
121                            long part = value;
122    
123                            part %= frac.decimalDenominator;
124                            part = MathExtra.longMult(part, frac.nativeDenominator);
125                            part /= frac.decimalDenominator;
126    
127                            return MathExtra.longAdd(whole, part);
128    
129                    } else {
130    
131                            return value;
132    
133                    }
134    
135            }

279            /**
280             * Long encode.
281             * 
282             * @param value
283             *            the value
284             * @param buffer
285             *            the buffer
286             * @param marker
287             *            the marker
288             */
289            public final static void longEncode(final long value,
290                            final ByteBuffer buffer, final byte marker) {
291                    if (value == DDF_EMPTY) {
292                            // empty
293                            buffer.put(marker);
294                            return;
295                    }
296                    if (value == DDF_CLEAR) {
297                            buffer.put(ASCII.DASH);
298                            buffer.put(marker);
299                            return;
300                    }
301                    // TODO optimize
302                    buffer.put(Long.toString(value).getBytes(ASCII.ASCII_CHARSET));
303                    if (marker == ASCII.NUL) {
304                            // write no marker
305                            return;
306                    }
307                    buffer.put(marker);
308            }

310            /**
311             * note: NUL marker does not consume; while non NUL does.
312             * 
313             * @param buffer
314             *            the buffer
315             * @param marker
316             *            the marker
317             * @return the long
318             */
319            public final static long longDecode(final ByteBuffer buffer,
320                            final byte marker) {
321    
322                    buffer.mark();
323    
324                    final boolean isDash = (buffer.get() == ASCII.DASH);
325    
326                    if (!isDash) {
327                            buffer.reset();
328                    }
329    
330                    long value = 0;
331                    int count = 0;
332    
333                    while (true) {
334    
335                            if (buffer.position() == buffer.limit()) {
336                                    break;
337                            }
338    
339                            final byte alpha = buffer.get();
340    
341                            if (ASCII.isDigit(alpha)) {
342                                    value *= 10;
343                                    value += alpha - ASCII._0_;
344                                    count++;
345                                    continue;
346                            }
347    
348                            // stop on any; do not consume
349                            if (marker == ASCII.NUL) {
350                                    buffer.position(buffer.position() - 1);
351                                    break;
352                            }
353    
354                            // stop on match; consume
355                            if (marker == alpha) {
356                                    break;
357                            } else {
358                                    throw new IllegalStateException();
359                            }
360    
361                    }
362    
363                    if (isDash) {
364                            if (count > 0) {
365                                    return -value;
366                            } else {
367                                    return DDF_CLEAR;
368                            }
369                    } else {
370                            if (count > 0) {
371                                    return value;
372                            } else {
373                                    return DDF_EMPTY;
374                            }
375                    }
376    
377            }

402             * Price decode.
403             * 
404             * @param string
405             *            the string
406             * @return the price value
407             */
408            public final static PriceValue priceDecode(final String string) {
409                    final String[] array = string.split(ASCII.REGEX_DOT);
410                    final String mantString = array[0].concat(array[1]);
411                    final long mantissa = Long.parseLong(mantString);
412                    final int exponent = -array[1].length();
413                    return ValueBuilder.newPrice(mantissa, exponent);
414            }

417            /**
418             * Price encode.
419             * 
420             * @param price
421             *            the price
422             * @return the string
423             */
424            public final static String priceEncode(final PriceValue price) {
425                    long whole = price.mantissa();
426                    int e1 = price.exponent();
427                    while (e1 > 0) {
428                            whole *= 10;
429                            e1--;
430                    }
431                    while (e1 < 0) {
432                            whole /= 10;
433                            e1++;
434                    }
435                    long part = 0;
436                    int e2 = price.exponent();
437                    if (e2 >= 0) {
438                            part = 0;
439                    } else {
440                            int denom = 1;
441                            while (e2 < 0) {
442                                    denom *= 10;
443                                    e2++;
444                            }
445                            part = price.mantissa() % denom;
446                    }
447                    return String.format("%d.%d", whole, part);
448            }

463             * 
464             * @param string

468            public static final long longDecode(final String string) {
469                    if (isEmptyDDF(string)) {
470                            return DDF_EMPTY;

495            /**
496             * from ddf "20100616124807" into millisUTC.
497             * 
498             * @param value
499             *            the value
500             * @param zone
501             *            the zone
502             * @return the long
503             */
504            public static final long timeDecode(/* local */long value,
505                            final DateTimeZone zone) {
506                    final int second = (int) (value % 100);
507                    value /= 100;
508                    final int minute = (int) (value % 100);
509                    value /= 100;
510                    final int hour = (int) (value % 100);
511                    value /= 100;
512                    final int day = (int) (value % 100);
513                    value /= 100;
514                    final int month = (int) (value % 100);
515                    value /= 100;
516                    final int year = (int) value;
517                    final DateTime dateTime = new DateTime(year, month, day, hour, minute,
518                                    second, 0, zone);
519                    return dateTime.getMillis();
520            }

522            /**
523             * from millisUTC into ddf "20100616124807".
524             * 
525             * @param millisUTC
526             *            the millis utc
527             * @param zone
528             *            the zone
529             * @return the long
530             */
531            public static final long timeEncode(final long millisUTC,
532                            final DateTimeZone zone) {
533                    final DateTime dateTime = new DateTime(millisUTC, zone);
534                    long value = 0;
535                    final int year = dateTime.getYearOfEra();
536                    value += year;
537                    value *= 100;
538                    final int month = dateTime.getMonthOfYear();
539                    value += month;
540                    value *= 100;
541                    final int day = dateTime.getDayOfMonth();
542                    value += day;
543                    value *= 100;
544                    final int hour = dateTime.getHourOfDay();
545                    value += hour;
546                    value *= 100;
547                    final int minute = dateTime.getMinuteOfHour();
548                    value += minute;
549                    value *= 100;
550                    final int second = dateTime.getSecondOfMinute();
551                    value += second;
552                    return value;
553            }
