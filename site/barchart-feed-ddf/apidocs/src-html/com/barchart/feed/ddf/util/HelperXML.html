111            /**
112             * Xml first child.
113             * 
114             * @param parent
115             *            the parent
116             * @param childName
117             *            the child name
118             * @param isThrow
119             *            the is throw
120             * @return the element
121             */
122            public static final Element xmlFirstChild(final Element parent,
123                            final String childName, final boolean isThrow) {
124                    final NodeList nodeList = parent.getChildNodes();
125                    if (nodeList == null) {
126                            if (isThrow) {
127                                    throw new IllegalArgumentException("nodeList == null");
128                            }
129                            return null;
130                    }
131                    final int size = nodeList.getLength();
132                    for (int k = 0; k < size; k++) {
133                            final Node child = nodeList.item(k);
134                            if (child.getNodeName().equalsIgnoreCase(childName)) {
135                                    return (Element) child;
136                            }
137                    }
138                    if (isThrow) {
139                            throw new IllegalArgumentException("child node not found");
140                    }
141                    return null;
142            }

189             * Xml document encode.
190             * 
191             * @param root
192             *            the root
193             * @param isThrow
194             *            the is throw
195             * @return the byte[]
196             */
197            public static final byte[] xmlDocumentEncode(final Element root,
198                            final boolean isThrow) {
199                    final Source source = new DOMSource(root);
200                    final StringWriter stream = new StringWriter();
201                    final Result result = new StreamResult(stream);
202                    try {
203                            XML_XFORMER.get().transform(source, result);
204                            final String string = stream.toString();
205                            if (ValueBuilder.isPureAscii(string)) {
206                                    return string.getBytes(ASCII.ASCII_CHARSET);
207                            }
208                    } catch (final Exception e) {
209                            // will return/throw below
210                    }
211                    if (isThrow) {
212                            throw new RuntimeException("can not encode");
213                    }
214                    return null;
215            }
