032    /**
033     * creates fake unresolved instruments.
034     */
035    @ThreadSafe
036    public class ServiceBasicDDF implements DDF_DefinitionService {
037    
038            protected final Logger log = LoggerFactory.getLogger(getClass());
039            
040            /**
041             * Instantiates a new service basic ddf.
042             */
043            public ServiceBasicDDF() {
044            }
045    
046            @SuppressWarnings("rawtypes")
047            @Override
048            public List<Instrument> lookup(final CharSequence symbol) {
049                    if(symbol == null || symbol.length() == 0) {
050                            return Collections.emptyList();
051                    }
052                    
053                    /** make an upper case id */
054                    final TextValue lookup = lookupFromSymbol(ValueBuilder.newText(symbol.toString()));
055    
056                    final Map<Tag, Object> map = new HashMap<Tag, Object>();
057                    map.put(InstrumentField.MARKET_GUID, lookup);
058                    final Instrument instrument = ObjectMapFactory.build(InstrumentDDF.class, map);
059    
060                    return Collections.singletonList(instrument);
061            }
062

066                    return null;
067            }

073            }
