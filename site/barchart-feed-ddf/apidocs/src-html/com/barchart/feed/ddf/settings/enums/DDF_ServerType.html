011    package com.barchart.feed.ddf.settings.enums;
012    
013    import com.barchart.feed.base.enums.EnumCodeString;
014    import com.barchart.util.common.math.MathExtra;
015    
016    /**
017     * Enum denoting the data source/server type for a connection.
018     * 
019     */
020    public enum DDF_ServerType implements EnumCodeString {
021    
022            /** market data service */
023            STREAM("stream"), //
024    
025            /** legacy historical system */
026            HISTORICAL("historical"), //
027    
028            /** current historical system */
029            HISTORICAL_V2("historicalv2"), //
030    
031            /** instrument lookup service */
032            EXTRAS("extras"), //
033    
034            /** market news service */
035            NEWS("news"), //
036    
037            //
038    
039            UNKNOWN(""), //
040    
041            ;
042    
043            private final String code;
044    
045            private DDF_ServerType(final String code) {
046                    this.code = code;
047            }
048    
049            private final static DDF_ServerType[] ENUM_VALUES = values();
050    
051            static {
052                    // validate use of byte ord
053                    MathExtra.castIntToByte(ENUM_VALUES.length);
054            }
055    
056            /**
057             * From ord.
058             * 
059             * @param ord
060             *            the ord
061             * @return the DDF_ServerType
062             */
063            public final static DDF_ServerType fromOrd(final byte ord) {
064                    return ENUM_VALUES[ord];
065            }
066    
067            /**
068             * From code.
069             * 
070             * @param code
071             *            the code
072             * @return the DDF_ServerType
073             */
074            public static final DDF_ServerType fromCode(final String code) {
075                    for (final DDF_ServerType known : ENUM_VALUES) {
076                            if (known.code.equalsIgnoreCase(code)) {
077                                    return known;
078                            }
079                    }
080                    return UNKNOWN;
081            }
082    
083            /*
084             * (non-Javadoc)
085             * 
086             * @see com.barchart.util.enums.EnumCodeString#code()
087             */
088            @Override
089            public final String code() {
090                    return code;
091            }
092    
093    }
