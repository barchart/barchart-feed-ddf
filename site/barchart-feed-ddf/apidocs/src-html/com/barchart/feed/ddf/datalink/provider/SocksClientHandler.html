111              @Override
112              public void messageReceived(ChannelHandlerContext ctx, MessageEvent e) throws Exception {
113    
114                     if(!(e.getMessage() instanceof ChannelBuffer)){
115                              return;
116                      }
117    
118                      ChannelBuffer msg = (ChannelBuffer) e.getMessage();
119    
120                      msg = msg.slice(msg.readerIndex(), msg.writerIndex());
121    
122                      //  SERVER AUTHENTICATION RESPONSE
123                      //  The server selects from one of the methods given in METHODS, and
124                      //  sends a METHOD selection message:
125                      //
126                      //     +----+--------+
127                      //     |VER | METHOD |
128                      //     +----+--------+
129                      //     | 1  |   1    |
130                      //     +----+--------+
131                      //
132                      //  If the selected METHOD is X'FF', none of the methods listed by the
133                      //  client are acceptable, and the client MUST close the connection.
134                      //
135                      //  The values currently defined for METHOD are:
136                      //   * X'00' NO AUTHENTICATION REQUIRED
137                      //   * X'01' GSSAPI
138                      //   * X'02' USERNAME/PASSWORD
139                      //   * X'03' to X'7F' IANA ASSIGNED
140                      //   * X'80' to X'FE' RESERVED FOR PRIVATE METHODS
141                      //   * X'FF' NO ACCEPTABLE METHODS
142    
143                      //  receive the server response 
144    
145                      if(lastStep==null){
146    
147                              byte acceptedAuthMethod = msg.getByte(1);
148    
149                              // if the server does not accept any of our supported authentication methods then throw an error
150                              if (acceptedAuthMethod == SOCKS5_AUTH_METHOD_REPLY_NO_ACCEPTABLE_METHODS){
151                                     e.getChannel().close();
152                                     throw new SocksProxyException("The proxy destination does not accept the supported proxy client authentication methods.");
153                              }
154    
155                              if(acceptedAuthMethod == SOCKS5_AUTH_METHOD_NO_AUTHENTICATION_REQUIRED){
156                                     
157                                      log.debug("Using no authentication for SOCKS5.");
158    
159                                      //send connect command
160                                      
161                                      SendCommand(e.getChannel(), SOCKS5_CMD_CONNECT);
162                                      
163                                      lastStep = Step.CONNECT;
164                                      
165                                      return;
166                              }
167    
168                              if (acceptedAuthMethod == SOCKS5_AUTH_METHOD_USERNAME_PASSWORD && _proxyAuthMethod == SocksAuthentication.None){
169                                      e.getChannel().close();
170                                      throw new SocksProxyException("The proxy destination requires a username and password for authentication, none are set");
171                              }
172    
173                              if (acceptedAuthMethod == SOCKS5_AUTH_METHOD_USERNAME_PASSWORD) {
174    
175                                      log.debug("Using username and password for SOCKS5 authentication");
176    
177                                      _proxyUserName = proxySettings.getProxyUsername();
178                                      _proxyPassword = proxySettings.getProxyPassword();
179                                      
180                                      // USERNAME / PASSWORD SERVER REQUEST
181                                      // Once the SOCKS V5 server has started, and the client has selected the
182                                      // Username/Password Authentication protocol, the Username/Password
183                                      // subnegotiation begins.  This begins with the client producing a
184                                      // Username/Password request:
185                                      //
186                                      //       +----+------+----------+------+----------+
187                                      //       |VER | ULEN |  UNAME   | PLEN |  PASSWD  |
188                                      //       +----+------+----------+------+----------+
189                                      //       | 1  |  1   | 1 to 255 |  1   | 1 to 255 |
190                                      //       +----+------+----------+------+----------+
191    
192                                      // create a data structure (binary array) containing credentials
193                                      // to send to the proxy server which consists of clear username and password data
194                                      byte[] credentials = new byte[_proxyUserName.length() + _proxyPassword.length() + 3];
195    
196                                      credentials[0] = SOCKS5_VERSION_NUMBER;
197                                      credentials[1] = (byte)_proxyUserName.length(); 
198    
199                                      System.arraycopy(_proxyUserName.getBytes(), 0, credentials,2, _proxyUserName.length());
200    
201                                      credentials[_proxyUserName.length() + 2] = (byte)_proxyPassword.length();
202    
203                                      System.arraycopy(_proxyPassword.getBytes(), 0, credentials, _proxyUserName.length() + 3, _proxyPassword.length()); 
204    
205                                      // USERNAME / PASSWORD SERVER RESPONSE
206                                      // The server verifies the supplied UNAME and PASSWD, and sends the
207                                      // following response:
208                                      //
209                                      //   +----+--------+
210                                      //   |VER | STATUS |
211                                      //   +----+--------+
212                                      //   | 1  |   1    |
213                                      //   +----+--------+
214                                      //
215                                      // A STATUS field of X'00' indicates success. If the server returns a
216                                      // `failure' (STATUS value other than X'00') status, it MUST close the
217                                      // connection.
218    
219                                      // transmit credentials to the proxy server
220                                      
221                                      e.getChannel().write(ChannelBuffers.wrappedBuffer(credentials));
222    
223                                      lastStep = Step.AUTH;
224                                      
225                                      return;
226    
227                              }
228    
229                              // read the response from the proxy server
230    
231                      }else if(lastStep==Step.AUTH){
232    
233                              byte authResponse =  msg.getByte(1);
234                              
235                              log.debug("Proxy authentication resp {} " , authResponse);
236    
237                              // check to see if the proxy server accepted the credentials
238                              if (authResponse != 0) {
239                                      
240                                      e.getChannel().close();
241                                      
242                                      log.debug("Proxy authentication failure: {}", authResponse);
243                                      
244                                      throw new SocksProxyException("Proxy authentication failure - The proxy server has " +
245                                                    "reported that the userid and/or password is not valid.");
246                                      
247                              }else{
248    
249                                      log.debug("Proxy authentication success: {}", authResponse);
250                                      
251                                      // send a connect command to the proxy server for destination host and port
252                                      
253                                      SendCommand(e.getChannel(), SOCKS5_CMD_CONNECT);
254                                      
255                                      lastStep = Step.CONNECT;
256                                      
257                                      return;
258                              }
259                              
260                      }
261    
262                      // read connect command response
263    
264                      //  PROXY SERVER RESPONSE
265                      //  +----+-----+-------+------+----------+----------+
266                      //  |VER | REP |  RSV  | ATYP | BND.ADDR | BND.PORT |
267                      //  +----+-----+-------+------+----------+----------+
268                      //  | 1  |  1  | X'00' |  1   | Variable |    2     |
269                      //  +----+-----+-------+------+----------+----------+
270                      //
271                      // * VER protocol version: X'05'
272                      // * REP Reply field:
273                      //   * X'00' succeeded
274                      //   * X'01' general SOCKS server failure
275                      //   * X'02' connection not allowed by ruleset
276                      //   * X'03' Network unreachable
277                      //   * X'04' Host unreachable
278                      //   * X'05' Connection refused
279                      //   * X'06' TTL expired
280                      //   * X'07' Command not supported
281                      //   * X'08' Address itemType not supported
282                      //   * X'09' to X'FF' unassigned
283                      //* RSV RESERVED
284                      //* ATYP address itemType of following address
285    
286                      // return this channel for DDF3
287    
288                      if(lastStep==Step.CONNECT){
289    
290                              byte replyCode = msg.getByte(1);
291    
292                              if (replyCode == SOCKS5_CMD_REPLY_SUCCEEDED){
293                                      
294                                      log.debug("Connect Command SUCCESS :  {}" , replyCode);
295    
296                                      connectCommandResult = true;
297                                      
298                              }else{
299                                      HandleProxyCommandError(msg.array(), 
300                                                      proxySettings.getFeedServer().getPrimary(), 7500);
301                                      
302                                      connectCommandResult = false;
303                              }
304                              
305                              lastStep = Step.BIND;
306                              
307                              return;
308                      }
309    
310    
311                      // if our connection has been proxied, allow the DDF handlers
312                      // to process messages from JERQ
313    
314                      if(lastStep == Step.BIND && proxied == false){
315                              feedClient.setProxiedChannel(ctx, e, true);
316                              proxied = true;
317                      }
318                      
319                      if(lastStep == Step.BIND && proxied){
320                              ctx.sendUpstream(e);
321                      }
322    
323              }

325              private void SendCommand(Channel channel, byte command) throws Exception{
326       
327                      if(proxySettings.getFeedServer()==null){
328                              throw new SocksProxyException("No destination server set");
329                      }
330                      
331                      String destinationHost ="174.129.40.242";
332                      Integer destinationPort = 7500;
333                      
334              byte addressType = GetDestAddressType(destinationHost);
335              byte[] destAddr = GetDestAddressBytes(addressType, destinationHost);
336              byte[] destPort = GetDestPortBytes(destinationPort);
337    
338              //  The connection request is made up of 6 bytes plus the
339              //  length of the variable address byte array
340              //
341              //  +----+-----+-------+------+----------+----------+
342              //  |VER | CMD |  RSV  | ATYP | DST.ADDR | DST.PORT |
343              //  +----+-----+-------+------+----------+----------+
344              //  | 1  |  1  | X'00' |  1   | Variable |    2     |
345              //  +----+-----+-------+------+----------+----------+
346              //
347              // * VER protocol version: X'05'
348              // * CMD
349              //   * CONNECT X'01'
350              //   * BIND X'02'
351              //   * UDP ASSOCIATE X'03'
352              // * RSV RESERVED
353              // * ATYP address itemType of following address
354              //   * IP V4 address: X'01'
355              //   * DOMAINNAME: X'03'
356              //   * IP V6 address: X'04'
357              // * DST.ADDR desired destination address
358              // * DST.PORT desired destination port in network octet order            
359    
360              byte[] request = new byte[4 + destAddr.length + 2];
361              request[0] = SOCKS5_VERSION_NUMBER;
362              request[1] = command;
363              request[2] = SOCKS5_RESERVED;
364              request[3] = addressType;
365              System.arraycopy(destAddr, 0, request, 4, destAddr.length);
366              System.arraycopy(destPort, 0, request, 4 + destAddr.length, destPort.length);
367              
368              // send connect request.
369              channel.write(ChannelBuffers.wrappedBuffer(request));
370          
371              sentConnectCommand = true;
372          
373          }

375              private byte GetDestAddressType(String host) {
376    
377             boolean isIp = isAnIP(host);
378    
379             if (!isIp) 
380             {
381                     System.out.println("Address is domain");
382                     return SOCKS5_ADDRTYPE_DOMAIN_NAME;
383             }
384             
385             InetAddress inet = null;
386             
387             try {
388                            inet = InetAddress.getByName(host);
389             } catch (UnknownHostException e) {
390                            return -1;
391             }
392              
393            if(inet instanceof Inet4Address){
394                    return SOCKS5_ADDRTYPE_IPV4;
395            }else if(inet instanceof Inet6Address){
396                    return SOCKS5_ADDRTYPE_IPV6;
397            }
398            
399            log.error("The host addess " + host + " is not a supported address type.  " +
400                            "The supported types are domainm, IPV4 and InterNetworkV6(TODO).");
401            
402            return -1;
403          }

405              private byte[] GetDestAddressBytes(byte addressType, String host){
406                      
407              if (addressType == SOCKS5_ADDRTYPE_IPV4){
408                      
409                      InetAddress inet = null;
410                      try {
411                                    inet = InetAddress.getByName(host);
412                            } catch (UnknownHostException e) {
413                                    e.printStackTrace();
414                                    return null;
415                            }
416                            
417                    byte [] binaryIP = inet.getAddress();
418                  
419                    return binaryIP;
420                    
421              }else if(addressType == SOCKS5_ADDRTYPE_IPV6){
422                      
423                      // TODO
424                      
425                      return null;
426                      
427              }else if(addressType == SOCKS5_ADDRTYPE_DOMAIN_NAME){
428                      
429                  byte[] bytes = new byte[host.getBytes().length + 1]; 
430                  
431                      //  create a byte array to hold the host name bytes plus one byte to store the length
432                  //  if the address field contains a fully-qualified domain name.  The first
433                  //  octet of the address field contains the number of octets of name that
434                  //  follow, there is no terminating NUL octet.
435                  
436                  // byte[] host_length_byte = ByteBuffer.allocate(1).putInt(host.length()).array();
437                  // byte[] host_byte = ByteBuffer.allocate(1).p(host.length()).array();
438                  
439                  bytes[0] = (byte) host.getBytes().length;
440                  System.arraycopy( host.getBytes(), 0, bytes, 1, host.getBytes().length);
441    
442                  return bytes;
443                  
444              }
445            
446              return null; 
447

491                    private void NegotiateServerAuthMethod(Channel channel){
492                            
493    
494                    // SERVER AUTHENTICATION REQUEST
495                    // The client connects to the server, and sends a version
496                    // identifier/method selection message:
497                    //
498                    //      +----+----------+----------+
499                    //      |VER | NMETHODS | METHODS  |
500                    //      +----+----------+----------+
501                    //      | 1  |    1     | 1 to 255 |
502                    //      +----+----------+----------+
503                            
504                            byte[] authRequest = new byte[4];
505                            
506                            authRequest[0] = SOCKS5_VERSION_NUMBER;
507                    authRequest[1] = SOCKS5_AUTH_NUMBER_OF_AUTH_METHODS_SUPPORTED;
508                    authRequest[2] = SOCKS5_AUTH_METHOD_NO_AUTHENTICATION_REQUIRED; 
509                    authRequest[3] = SOCKS5_AUTH_METHOD_USERNAME_PASSWORD; 
510    
511                    // send the request to the server specifying authentication 
512                    // types supported by the client.
513                   
514                    channel.write( ChannelBuffers.wrappedBuffer(authRequest));
515                               
516                    }

518          private byte[] GetDestPortBytes(int value){
519    
520              byte[] bytes = new byte[2];
521    
522              bytes[0] = (byte) (value / 256);
523              bytes[1] = (byte) (value % 256);
524              
525              return bytes;
526          }

528              public static boolean isAnIP (String sip){
529                      
530                    String [] parts = sip.split ("\\.");
531                    
532                    for (String s : parts) {
533                        int i =0;
534                       
535                        try {
536                                            i= Integer.parseInt (s);
537                                    } catch (NumberFormatException e) {
538                                            return false;
539                                    }
540                        
541                        if (i < 0 || i > 255){
542                            return false;
543                        }
544                    }
545                    
546                    return true;
547              }

549              private void HandleProxyCommandError(byte[] response, String destinationHost, int destinationPort) throws SocksProxyException{
550                      String proxyErrorText;
551                      
552                      byte replyCode = response[1];
553                      byte addrType = response[3];
554                      
555                      String addr = "";
556                      Integer port = 0;
557    
558                      if(addrType == SOCKS5_ADDRTYPE_DOMAIN_NAME){
559                              int addrLen = response[4];
560                              byte[] addrBytes = new byte[addrLen];
561                              for (int i = 0; i < addrLen; i++)
562                                      addrBytes[i] = response[i + 5];
563                              addr = new String(addrBytes);
564                              byte[] portBytesDomain = new byte[2];
565                              portBytesDomain[0] = response[6 + addrLen];
566                              portBytesDomain[1] = response[5 + addrLen];
567                              ByteBuffer bb = ByteBuffer.wrap(portBytesDomain);
568                              port = bb.getInt(); 
569    
570                      }else if (addrType == SOCKS5_ADDRTYPE_IPV4){
571    
572                              byte[] ipv4Bytes = new byte[4];
573                              for (int i = 0; i < 4; i++)
574                                      ipv4Bytes[i] = response[i + 4];
575    
576                              byte[] portBytesIpv4 = new byte[2];
577                              portBytesIpv4[0] = response[9];
578                              portBytesIpv4[1] = response[8];
579    
580                              InetAddress ipv4 = null;
581                              try {
582                                      ipv4 = InetAddress.getByAddress(ipv4Bytes);
583                              } catch (UnknownHostException e1) {
584                                      e1.printStackTrace();
585                              }
586                              
587                              port = (((0xFF & portBytesIpv4[1]) << 8) + (0xFF & portBytesIpv4[0]));
588                              addr = ipv4.toString();
589    
590                      }else if (addrType== SOCKS5_ADDRTYPE_IPV6){
591                              byte[] ipv6Bytes = new byte[16];
592                              for (int i = 0; i < 16; i++)
593                                      ipv6Bytes[i] = response[i + 4];
594    
595                              //IPAddress ipv6 = new IPAddress(ipv6Bytes);
596                              InetAddress ipv6 = null;
597    
598                              try {
599                                      ipv6 = InetAddress.getByAddress(ipv6Bytes);
600                              } catch (UnknownHostException e) {
601                                      e.printStackTrace();
602                              }
603    
604                              byte[] portBytesIpv6 = new byte[2];
605                              portBytesIpv6[0] = response[21];
606                              portBytesIpv6[1] = response[20];
607    
608                              port = (((0xFF & portBytesIpv6[1]) << 8) + (0xFF & portBytesIpv6[0]));
609                              addr = ipv6.toString();
610    
611                      }
612    
613    
614                      if(replyCode == SOCKS5_CMD_REPLY_GENERAL_SOCKS_SERVER_FAILURE){
615                              proxyErrorText = "a general socks destination failure occurred";
616    
617                      }else if(replyCode == SOCKS5_CMD_REPLY_CONNECTION_NOT_ALLOWED_BY_RULESET){
618    
619                              proxyErrorText = "the connection is not allowed by proxy destination rule set";
620    
621                      }else if(replyCode == SOCKS5_CMD_REPLY_NETWORK_UNREACHABLE){
622                              proxyErrorText = "the network was unreachable";
623    
624                      }else if(replyCode == SOCKS5_CMD_REPLY_HOST_UNREACHABLE){
625                              proxyErrorText = "the host was unreachable";
626    
627                      }else if(replyCode == SOCKS5_CMD_REPLY_CONNECTION_REFUSED){
628                              proxyErrorText = "the connection was refused by the remote network";
629    
630                      }else if(replyCode ==SOCKS5_CMD_REPLY_TTL_EXPIRED){
631    
632                              proxyErrorText = "the time to live (TTL) has expired";
633    
634                      }else if(replyCode == SOCKS5_CMD_REPLY_COMMAND_NOT_SUPPORTED){
635    
636                              proxyErrorText = "the command issued by the proxy client is not supported by the proxy destination";
637    
638                      }else if(replyCode == SOCKS5_CMD_REPLY_ADDRESS_TYPE_NOT_SUPPORTED){
639    
640                              proxyErrorText = "the address type specified is not supported";
641    
642                      }else{
643                              proxyErrorText = String.format("that an unknown reply with the code value '%s' was received " +
644                                            "by the destination", Byte.toString(replyCode));
645    
646                      }
647    
648                      @SuppressWarnings("unused")
649                      String exceptionMsg = String.format("%s destination host %s port number %s.  The destination reported " +
650                                    "the host as %s port %s type of %s.", proxyErrorText, destinationHost, destinationPort, addr, Integer.toString(port), addrType);
651    
652                      feedClient.postEvent(DDF_FeedEvent.LINK_CONNECT_PROXY_TIMEOUT);
653                      feedClient.setProxiedChannel(null, null, false);
654                      
655                      log.error("Socks5 Error : {}", proxyErrorText);
656                      
657                      // don't blow up, allow user forced reconnect
658                      
659                      //throw new SocksProxyException(exceptionMsg);
660    
661              }
