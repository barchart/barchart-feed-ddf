008    package com.barchart.feed.ddf.message.api;
009    
010    import com.barchart.feed.base.market.api.MarketMessage;
011    import com.barchart.feed.base.values.api.TextValue;
012    import com.barchart.feed.base.values.api.TimeValue;
013    import com.barchart.feed.ddf.message.enums.DDF_Session;
014    import com.barchart.feed.ddf.message.enums.DDF_TradeDay;
015    import com.barchart.feed.ddf.symbol.api.DDF_Symbol;
016    import com.barchart.feed.ddf.symbol.enums.DDF_Exchange;
017    import com.barchart.feed.ddf.symbol.enums.DDF_SpreadType;
018    import com.barchart.feed.ddf.util.enums.DDF_Fraction;
019    import com.barchart.util.common.anno.NotMutable;
020    
021    /**
022     * Base type for ddf feed market data messages.
023     */
024    @NotMutable
025    public interface DDF_MarketBase extends DDF_BaseMessage, MarketMessage {
026    
027            /* GENERIC */
028    
029            /** market instrument resolved from {@link #getId()} */
030            //InstrumentEntity instrument();
031    
032            /** time from message time stamp if present or from message arrival time; */
033            TimeValue getTime();
034    
035            /* PROPRIETARY */
036    
037            /** ddf feed symbol parser; resolved from {@link #getId()} */
038            DDF_Symbol getSymbol();
039    
040            /** raw ddf feed symbol, such as MSFT or RJZ1 */
041            TextValue getId();
042    
043            /** ddf exchange classifier */
044            DDF_Exchange getExchange();
045    
046            /** ddf "base code" represented as generic "price fraction" */
047            DDF_Fraction getFraction();
048    
049            /** ddf spread type, if any */
050            DDF_SpreadType getSpreadType();
051    
052            /** ddf trading date of message */
053            DDF_TradeDay getTradeDay();
054    
055            /** ddf trading session classifier; */
056            DDF_Session getSession();
057    
058            /** nominal message delay time indicator, minutes; when delay is present */
059            int getDelay();
060    
061    }
