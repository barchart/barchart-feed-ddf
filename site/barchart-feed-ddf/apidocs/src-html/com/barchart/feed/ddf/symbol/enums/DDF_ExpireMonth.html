008    package com.barchart.feed.ddf.symbol.enums;
009    
010    import org.joda.time.DateTime;
011    import org.joda.time.DateTimeZone;
012    import org.slf4j.Logger;
013    import org.slf4j.LoggerFactory;
014    
015    // TODO: Auto-generated Javadoc
016    /**
017     * The Enum DDF_ExpireMonth.
018     */
019    public enum DDF_ExpireMonth {
020    
021            /*
022             * EXPIRATION MONTH CODE 1 Byte, Alphabetic. It indicates the expiration
023             * month for the security future.
024             */
025    
026            JAN('F', 1), //
027            JAN_Long('A', 1), //
028            FEB('G', 2), //
029            FEB_Long('B', 2), //
030            MAR('H', 3), //
031            MAR_Long('C', 3), //
032            APR('J', 4), //
033            APR_Long('D', 4), //
034            MAY('K', 5), //
035            MAY_Long('E', 5), //
036            JUN('M', 6), //
037            JUN_Long('I', 6), //
038            JUL('N', 7), //
039            JUL_Long('L', 7), //
040            AUG('Q', 8), //
041            AUG_Long('O', 8), //
042            SEP('U', 9), //
043            SEP_Long('P', 9), //
044            OCT('V', 10), //
045            OCT_Long('R', 10), //
046            NOV('X', 11), //
047            NOV_Long('S', 11), //
048            DEC('Z', 12), //
049            DEC_Long('T', 12), //
050    
051            CASH('Y', 0), //
052    
053            UNKNOWN('?', -1), //
054    
055            ;
056    
057            private static final Logger log = LoggerFactory
058                            .getLogger(DDF_ExpireMonth.class);
059    
060            /** The code. */
061            public final char code;
062    
063            /** The value. */
064            public final int value;
065    
066            private DDF_ExpireMonth(char code, int value) {
067                    this.code = code;
068                    this.value = value;
069            }
070    
071            /**
072             * From code.
073             *
074             * @param code the code
075             * @return the dD f_ expire month
076             */
077            public static final DDF_ExpireMonth fromCode(final char code) {
078                    switch (code) {
079                    case 'F':
080                            return JAN;
081                    case 'G':
082                            return FEB;
083                    case 'H':
084                            return MAR;
085                    case 'J':
086                            return APR;
087                    case 'K':
088                            return MAY;
089                    case 'M':
090                            return JUN;
091                    case 'N':
092                            return JUL;
093                    case 'Q':
094                            return AUG;
095                    case 'U':
096                            return SEP;
097                    case 'V':
098                            return OCT;
099                    case 'X':
100                            return NOV;
101                    case 'Z':
102                            return DEC;
103                            //
104                    case 'Y':
105                            return CASH;
106                            // TODO 'long' months
107                    default:
108                            log.error(String.format(
109                                            "unknown expiration month code: %1$c (ascii %1$d)", code));
110                            return UNKNOWN;
111                    }
112            }
113    
114            /**
115             * From millis utc.
116             *
117             * @param millisUTC the millis utc
118             * @return the dD f_ expire month
119             */
120            public static final DDF_ExpireMonth fromMillisUTC(final long millisUTC) {
121                    final DateTime dateTime = new DateTime(millisUTC, DateTimeZone.UTC);
122                    return fromDateTime(dateTime);
123            }
124    
125            /**
126             * From date time.
127             *
128             * @param dateTime the date time
129             * @return the dD f_ expire month
130             */
131            public static final DDF_ExpireMonth fromDateTime(final DateTime dateTime) {
132                    if (dateTime == null) {
133                            return UNKNOWN;
134                    }
135                    // DateTime 'default' calendar is the ISO8601
136                    // JAN==1 ... DEC==12
137                    final int month = dateTime.getMonthOfYear();
138                    switch (month) {
139                    case 1:
140                            return JAN;
141                    case 2:
142                            return FEB;
143                    case 3:
144                            return MAR;
145                    case 4:
146                            return APR;
147                    case 5:
148                            return MAY;
149                    case 6:
150                            return JUN;
151                    case 7:
152                            return JUL;
153                    case 8:
154                            return AUG;
155                    case 9:
156                            return SEP;
157                    case 10:
158                            return OCT;
159                    case 11:
160                            return NOV;
161                    case 12:
162                            return DEC;
163                    default:
164                            return UNKNOWN;
165                    }
166            }
167    
168            /**
169             * Checks if is valid.
170             *
171             * @param code the code
172             * @return true, if is valid
173             */
174            public static final boolean isValid(final char code) {
175                    final DDF_ExpireMonth month = fromCode(code);
176                    return month != UNKNOWN;
177            }
178    
179    }
