001    package com.barchart.feed.ddf.instrument.provider;
002    
003    import java.util.Collection;
004    import java.util.Map;
005    
006    import com.barchart.feed.api.model.meta.Instrument;
007    import com.barchart.feed.inst.InstrumentService;
008    
009    public class InstrumentProviderWrapper implements InstrumentService<String> {
010    
011            @Override
012            public Instrument lookup(String symbol) {
013                    return DDF_InstrumentProvider.fromSymbol(symbol.toString());
014            }
015    
016    
017            @Override
018            public Map<String, Instrument> lookup(Collection<String> symbols) {
019                    return DDF_InstrumentProvider.fromSymbols(symbols);
020            }
021    
022    }
