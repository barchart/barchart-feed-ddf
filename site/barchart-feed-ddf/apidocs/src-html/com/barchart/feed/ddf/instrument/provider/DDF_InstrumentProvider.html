130            /**
131             * This takes an instrument stub from a feed message, and does several things.
132             * 
133             * 1. Checks symbol against map.  If the stub represents an instrument already
134             * in the system, then it returns the canonical reference to that instrument.
135             * 
136             * 2. If the symbol is unknown, it will make a new InstrumentState from the info
137             * in the stub and begin an async lookup of info.
138             * 
139             * @param inst
140             * @return
141             */
142            public static Instrument fromMessage(final Instrument inst) {
143                    
144                    if(inst == null || inst.isNull()) {
145                            return Instrument.NULL;
146                    }
147                    
148                    /* NOTE id() in ddf is just the realtime symbol, not an actual GUID */
149                    final String symbol = Symbology.formatSymbol(inst.id().toString());
150                    
151                    if(symbolMap.containsKey(symbol)) {
152                            return symbolMap.get(symbol).get(0);
153                    }
154                    
155                    if(db.containsKey(symbol)) {
156                            
157                            final InstrumentState instState = InstrumentStateFactory.
158                                            newInstrument(symbol);
159                            instState.process(db.get(symbol));
160                            final List<InstrumentState> list = new ArrayList<InstrumentState>();
161                            list.add(instState);
162                            symbolMap.put(symbol, list);
163                            return instState;
164                    }
165                    
166                    /* New symbol, create stub */
167                    final InstrumentState instState = InstrumentStateFactory.
168                                    newInstrumentFromStub(inst);
169                    
170                    final List<InstrumentState> list = new ArrayList<InstrumentState>();
171                    list.add(instState);
172                    symbolMap.put(symbol, list);
173                    log.debug("Put {} stub into map", symbol);
174                    
175                    /* Asnyc lookup */
176                    try {
177                            remoteQueue.put(symbol);
178                    } catch (final Exception e) {
179                            failedRemoteQueue.add(symbol);
180                    }
181                    
182                    return instState;
183                    
184            }

322            private static Observer<InstrumentDefinitionResult> observer = 
323                            new Observer<InstrumentDefinitionResult>() {
324    
325                    @Override
326                    public void onNext(final InstrumentDefinitionResult result) {
327                            
328                            final String symbol = result.expression();
329                            
330                            /* If exception, add to failed */
331                            if(result.exception() != null) {
332                                    failedRemoteQueue.add(symbol);
333                            }
334                            
335                            final InstrumentDefinition def = result.result();
336                            
337                            if(def == InstrumentDefinition.getDefaultInstance()) {
338                                    log.trace("Instrument result was empty for {}", symbol);
339                                    return;  // Ignore
340                            }
341                            
342                            final InstrumentState iState = symbolMap.get(symbol).get(0);
343                            
344                            if(iState == null || iState.isNull()) {
345                                    final List<InstrumentState> list = new ArrayList<InstrumentState>();
346                                    list.add(InstrumentFactory.instrumentState(result.result()));
347                                    symbolMap.put(symbol, list);
348                            } else {
349                                    iState.process(result.result());
350                            }
351                            
352                    }
353    
354                    @Override
355                    public void onError(Throwable error) {
356                            // TODO Auto-generated method stub
357                            
358                    }
359    
360                    @Override
361                    public void onCompleted() {
362                            // TODO Auto-generated method stub
363                            
364                    }
365                    
366            };

369    
370                    private final String symbol;
371                    private final InstrumentDefinition def;
372                    private final Throwable t;
373                    
374                    InstDefResult(final String symbol, final InstrumentDefinition def) {
375                            this.symbol = symbol;
376                            this.def = def;
377                            t = null;
378                    }
379                    
380                    InstDefResult(final String symbol, final Throwable t) {
381                            this.symbol = symbol;
382                            def = InstrumentDefinition.getDefaultInstance();
383                            this.t = t;
384                    }
385                    
386                    @Override
387                    public InstrumentDefinition result() {
388                            return def;
389                    }
390    
391                    @Override
392                    public String expression() {
393                            return symbol;
394                    }
395    
396                    @Override
397                    public Throwable exception() {
398                            return t;
399                    }

409            static Callable<InstrumentDefinition> remoteSingle(final String lookup) {
410                    
411                    return new Callable<InstrumentDefinition>() {
412            
413                            @Override
414                            public InstrumentDefinition call() throws Exception {
415                                    
416                                    try {
417                                            
418                                            log.debug("Starting remote lookup for {}", lookup);
419                                            
420                                            final String symbolURI = urlInstrumentLookup(lookup);
421                                            final Element root = HelperXML.xmlDocumentDecode(symbolURI);
422                                            final Element tag = xmlFirstChild(root, "instrument", XML_STOP);
423                                            final InstrumentDefinition instDOM = InstrumentXML.decodeXML(tag);
424                                            
425                                            if(instDOM == null || instDOM == InstrumentDefinition.getDefaultInstance()) {
426                                                    log.trace("Empty instrument def returned from remote lookup: {}", lookup);
427                                                    failedRemoteQueue.add(lookup);
428                                                    return InstrumentDefinition.getDefaultInstance();
429                                            }
430                                            
431                                            return instDOM;
432                                            
433                                    } catch (final Throwable t) {
434                                            failedRemoteQueue.add(lookup);
435                                            log.trace("Remote instrument lookup callable exception: {}", t);
436                                            return InstrumentDefinition.getDefaultInstance();
437                                    }
438                                    
439                            }
440                    
441                    };
442                    
443            }

445            static Callable<Map<String, InstrumentDefinition>> remoteBatch(final String symbols) {
446                    
447                    return new Callable<Map<String, InstrumentDefinition>>() {
448    
449                            @Override
450                            public Map<String, InstrumentDefinition> call() throws Exception {
451                                    
452                                    try {
453                                            
454                                            final Map<String, InstrumentDefinition> defs = 
455                                                            new HashMap<String, InstrumentDefinition>();
456                                            
457                                            log.debug("remote batch on {}", urlInstrumentLookup(symbols));
458                                            
459                                            final URL url = new URL(urlInstrumentLookup(symbols));
460                                            
461                                            final HttpURLConnection connection = (HttpURLConnection) url
462                                                            .openConnection();
463                                            
464                                            connection.setRequestProperty("Accept-Encoding", "gzip");
465                                            
466                                            connection.connect();
467                                            
468                                            InputStream input = connection.getInputStream();
469    
470                                            if (connection.getContentEncoding().equals("gzip")) {
471                                                    input = new GZIPInputStream(input);
472                                            }
473                                            
474                                            final BufferedInputStream stream =
475                                                            new BufferedInputStream(input);
476    
477                                            final SAXParserFactory factory =
478                                                            SAXParserFactory.newInstance();
479                                            final SAXParser parser = factory.newSAXParser();
480                                            
481                                            final DefaultHandler handler = new DefaultHandler() {
482                                                    
483                                                    InstrumentDefinition def;
484                                                    
485                                                    @Override
486                                                    public void startElement(final String uri,
487                                                                    final String localName, final String qName,
488                                                                    final Attributes ats) throws SAXException {
489                                                            
490                                                            if (qName != null && qName.equals("instrument")) {
491    
492                                                                    try {
493                                                                            def = InstrumentXML.decodeSAX(ats);
494                                                                            if (def != InstrumentDefinition.getDefaultInstance()) {
495                                                                                    defs.put(def.getSymbol(), def);
496                                                                            }
497                                                                    } catch (final SymbolNotFoundException se) {
498                                                                            observer.onNext(new InstDefResult(se.getMessage(), se));
499                                                                    } catch (final Exception e) {
500                                                                            log.trace("Exception in parsing batch lookup {}", e);
501                                                                    }
502                                                                    
503                                                            }
504                                                            
505                                                    }
506                                                    
507                                            };
508                                            
509                                            parser.parse(stream, handler);
510                                            
511                                            return defs;
512                                            
513                                    } catch (final Throwable t) {
514                                            failedRemoteQueue.addAll(Arrays.asList(symbols.split(",")));
515                                            return null;
516                                    }
517                                    
518                            }
519                            
520                    };
521                    
522            }

524            static class RemoteRunner implements Runnable {
525                    
526                    private List<Future<Map<String, InstrumentDefinition>>> futures = 
527                                    new ArrayList<Future<Map<String, InstrumentDefinition>>>();
528                    
529                    private final List<Callable<Map<String, InstrumentDefinition>>> callables = 
530                                    new ArrayList<Callable<Map<String, InstrumentDefinition>>>();
531                    
532                    @Override
533                    public void run() {
534                            
535                            try {
536                                    
537                                    while(!Thread.interrupted()) {
538                                            
539                                            Thread.sleep(REMOTE_LOOKUP_INTERVAL);
540                                            
541                                            while(!remoteQueue.isEmpty()) {
542                                                    callables.add(remoteBatch(buildQuerey()));
543                                            }
544                                            
545                                            futures = executor.invokeAll(callables, DEFAULT_TIMEOUT, MILLIS);
546                                            
547                                            for(final Future<Map<String, InstrumentDefinition>> f : futures) {
548                                                    
549                                                    for(final Entry<String, InstrumentDefinition> e : f.get().entrySet()) {
550                                                            
551                                                            final InstrumentDefinition def = e.getValue();
552                                                            
553                                                            if(def == null || def == InstrumentDefinition.getDefaultInstance()) {
554                                                                    observer.onNext(new InstDefResult(e.getKey(),
555                                                                                    new Throwable("Could not find " + e.getKey())));
556                                                            } else {
557                                                                    observer.onNext(new InstDefResult(e.getKey(), def));
558                                                            }
559                                                            
560                                                    }
561                                                    
562                                            }
563                                            
564                                            futures.clear();
565                                            callables.clear();
566                                            
567                                    }
568                                    
569                            } catch (final Throwable t) {
570                                    log.error("Exception in Remote Runner Thread", t);
571                            }
572                            
573                    }
574                    
575            }
576            
577            private static String buildQuerey() throws Exception {
578                    
579                    final StringBuilder sb = new StringBuilder();
580                    
581                    int len = 0;
582                    int symCount = 0;
583                    
584                    while(len < MAX_URL_LEN && symCount < 400 && !remoteQueue.isEmpty()) {
585                            
586                            final String s = remoteQueue.take();
587                            
588                            log.debug("Pulled {} from remote queue", s);
589                            
590                            symCount++;
591                            len += s.length() + 1;
592                            sb.append(s).append(",");
593                            
594                    }
595                    
596                    /* Remove trailing comma */
597                    sb.deleteCharAt(sb.length() - 1);
598                    
599                    log.debug("Sending {} to remote lookup", sb.toString());
600                    
601                    return sb.toString();
602                    
603            }
