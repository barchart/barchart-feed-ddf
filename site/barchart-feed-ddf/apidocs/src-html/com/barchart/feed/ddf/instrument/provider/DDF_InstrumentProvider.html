167            /**
168             * cache via instrument service;.
169             * 
170             * @param symbols
171             * @return a map of resolved instruments
172             */
173            public static Map<CharSequence, List<Instrument>> find(
174                            final Collection<? extends CharSequence> symbols) {
175                    
176                    final Map<CharSequence, List<Instrument>> insts = 
177                                    new HashMap<CharSequence, List<Instrument>>();
178                    
179                    final Collection<CharSequence> formatted = new ArrayList<CharSequence>();
180                    
181                    for(CharSequence c : symbols) {
182                            formatted.add(formatSymbol(c));
183                    }
184                    
185                    final Map<CharSequence, List<Instrument>> charInsts = instance().lookup(formatted);
186                    
187                    for(final Entry<CharSequence, List<Instrument>> e : charInsts.entrySet()) {
188                            insts.put(e.getKey().toString(), e.getValue());
189                    }
190                    
191                    return insts;
192            }

214            public static List<Instrument> fetch(final List<String> symbolList) {
215    
216                    if(symbolList == null || symbolList.size() == 0) {
217                            return NULL_LIST;
218                    }
219                    
220                    try {
221                            return remoteLookup(symbolList);
222                    } catch (final Exception e) {
223                            log.error("", e);
224                            return NULL_LIST;
225                    }
226    
227            }

242            static String concatenate(final List<String> symbolList) {
243    
244                    final StringBuilder text = new StringBuilder(1024);
245    
246                    int count = 0;
247    
248                    for (final String symbol : symbolList) {
249                            text.append(symbol);
250                            count++;
251                            if (count != symbolList.size()) {
252                                    text.append(",");
253                            }
254                    }
255    
256                    return text.toString();
257    
258            }

260            private static List<Instrument> remoteLookup(final List<String> symbolList)
261                            throws Exception {
262    
263                    final List<Instrument> list =
264                                    new ArrayList<Instrument>(symbolList.size());
265    
266                    final String symbolString = concatenate(symbolList);
267    
268                    final String symbolURI = urlInstrumentLookup(symbolString);
269    
270                    log.debug("BATCH symbolURI");
271                    // log.debug("BATCH symbolURI : {}", symbolURI);
272    
273                    final URL url = new URL(symbolURI);
274    
275                    final InputStream input = url.openStream();
276    
277                    final BufferedInputStream stream = new BufferedInputStream(input);
278    
279                    try {
280    
281                            final SAXParserFactory factory = SAXParserFactory.newInstance();
282    
283                            final SAXParser parser = factory.newSAXParser();
284    
285                            final DefaultHandler handler = new DefaultHandler() {
286    
287                                    int count;
288    
289                                    @Override
290                                    public void startElement(final String uri,
291                                                    final String localName, final String qName,
292                                                    final Attributes attributes) throws SAXException {
293    
294                                            if (XmlTagExtras.TAG.equals(qName)) {
295    
296                                                    try {
297    
298                                                            final InstrumentDDF instrument = InstrumentXML.decodeSAX(attributes);
299                                                            final InstrumentDDF ddfInst = ObjectMapFactory.build(InstrumentDDF.class, instrument);
300                                                            list.add(ddfInst);
301    
302                                                    } catch (final SymbolNotFoundException e) {
303    
304                                                            log.warn("symbol not found : {}", e.getMessage());
305    
306                                                    } catch (final Exception e) {
307    
308                                                            log.error("decode failure", e);
309                                                            HelperXML.log(attributes);
310    
311                                                    }
312    
313                                                    count++;
314    
315                                                    if (count % 1000 == 0) {
316                                                            log.debug("fetch count : {}", count);
317                                                    }
318    
319                                            }
320    
321                                    }
322                            };
323    
324                            parser.parse(stream, handler);
325    
326                    } catch (final SAXParseException e) {
327    
328                            log.warn("parse failed : {} ", symbolURI);
329    
330                    } finally {
331    
332                            input.close();
333    
334                    }
335    
336                    return list;
337    
338            }

340            public static CharSequence formatSymbol(CharSequence symbol) {
341    
342                    if(symbol == null) {
343                            return "";
344                    }
345                    
346                    if(symbol.length() < 3) {
347                            return symbol;
348                    }
349                    
350                    if(symbol.charAt(0) == '_') {
351                            return symbol;
352                    }
353                    
354                    /* e.g. GOOG */
355                    if(!Character.isDigit(symbol.charAt(symbol.length() - 1))) {
356                            return symbol;
357                    }
358                    
359                    /* e.g. ESH3 */
360                    if(!Character.isDigit(symbol.charAt(symbol.length() - 2))) {
361                            
362                            final StringBuilder sb = new StringBuilder(symbol);
363                            int last = Character.getNumericValue(symbol.charAt(symbol.length() - 1));
364                            if(YEAR % 2010 < last) {
365                                    symbol = sb.insert(symbol.length() - 1, T_Z_O);
366                            } else if(YEAR % 2010 > last) {
367                                    symbol = sb.insert(symbol.length() - 1, T_Z_T);
368                            } else {
369                                    if(symbol.charAt(symbol.length() - 2) >= MONTH) {
370                                            symbol = sb.insert(symbol.length() - 1, T_Z_O);
371                                    } else {
372                                            symbol = sb.insert(symbol.length() - 1, T_Z_T);
373                                    }
374                            }
375                            
376                            return symbol;
377                    }
378                    
379                    /* e.g. ESH13 */
380                    if(!Character.isDigit(symbol.charAt(symbol.length() - 3))) {
381                            
382                            return new StringBuilder(symbol).insert(symbol.length()-2, T_Z);
383                            
384                    }
385                    
386                    return symbol;
387            }

389            public static CharSequence formatHistoricalSymbol(CharSequence symbol) {
390                    
391                    if(symbol == null) {
392                            return "";
393                    }
394                    
395                    if(symbol.length() < 3) {
396                            return symbol;
397                    }
398                    
399                    /* e.g. GOOG */
400                    if(!Character.isDigit(symbol.charAt(symbol.length() - 1))) {
401                            return symbol;
402                    }
403                    
404                    /* e.g. ESH3 */
405                    if(!Character.isDigit(symbol.charAt(symbol.length() - 2))) {
406                            return new StringBuilder(symbol).insert(symbol.length() - 1, O);
407                    }
408                    
409                    return symbol;
410            }
