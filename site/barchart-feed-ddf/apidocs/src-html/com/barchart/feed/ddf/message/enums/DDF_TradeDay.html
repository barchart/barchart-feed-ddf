236            private static final DateTimeZone ZONE_CST = DateTimeZone

239            private static final DateTimeZone ZONE_UTC = DateTimeZone.UTC;
240    
241            /** longest market holiday duration */
242            private static final int HOLIDAY_THESHOLD = 5;
243    
244            /**
245             * recover full trade date from DDF day code and todays date
246             * 
247             * expressed in UTC zone
248             * 
249             * year, month, day : should be treated as local market trade date.
250             *
251             * @param tradeDay the trade day
252             * @param todayDate the today date
253             * @return the time value
254             */
255            public static TimeValue tradeDateFrom(final DDF_TradeDay tradeDay,
256                            final DateTime todayDate) {
257    
258                    // trading day of month reported by the feed
259                    final int tradingDayNum = tradeDay.day;
260    
261                    // current day of month
262                    final int currentDayNum = todayDate.getDayOfMonth();
263    
264                    // positive for same month if trading date is in the future
265                    // unless day enum is not a day in the month ???
266                    final int difference = tradingDayNum - currentDayNum;
267    
268                    final boolean isSmall = Math.abs(difference) <= HOLIDAY_THESHOLD;
269                    final boolean isLarge = !isSmall;
270    
271                    //
272    
273                    final boolean isSameMonthSameDay = (difference == 0);
274    
275                    final boolean isSameMonthPastDay = difference < 0 & isSmall;
276                    final boolean isSameMonthNextDay = difference > 0 & isSmall;
277    
278                    final boolean isPastMonthPastDay = difference > 0 & isLarge;
279                    final boolean isNextMonthNextDay = difference < 0 & isLarge;
280    
281                    //
282    
283                    DateTime generated;
284    
285                    try {
286                            if (isSameMonthSameDay) {
287                                    generated = todayDate;
288                            } else if (isSameMonthPastDay) {
289                                    generated = todayDate.withDayOfMonth(tradingDayNum);
290                            } else if (isSameMonthNextDay) {
291                                    generated = todayDate.withDayOfMonth(tradingDayNum);
292                            } else if (isPastMonthPastDay) {
293                                    generated = todayDate.minusMonths(1).withDayOfMonth(tradingDayNum);
294                            } else if (isNextMonthNextDay) {
295                                    generated = todayDate.plusMonths(1).withDayOfMonth(tradingDayNum);
296                            } else {
297                                    logger.error("should not happen");
298                                    generated = todayDate;
299                            }
300                    } catch (final Exception e) {
301                            generated = todayDate;
302                    }
303    
304                    final DateTime result = new DateTime(//
305                                    generated.getYear(), //
306                                    generated.getMonthOfYear(), //
307                                    generated.getDayOfMonth(), //
308                                    0, 0, 0, 0, ZONE_UTC);
309    
310                    final long millisUTC = result.getMillis();
311    
312                    return ValueBuilder.newTime(millisUTC);
313    
314            }
