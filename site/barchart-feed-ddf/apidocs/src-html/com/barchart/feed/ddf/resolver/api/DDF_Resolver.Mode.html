008    package com.barchart.feed.ddf.resolver.api;
009    
010    import java.util.List;
011    import java.util.concurrent.Future;
012    
013    import com.barchart.feed.ddf.instrument.api.DDF_Instrument;
014    
015    /**
016     * The Interface DDF_Resolver.
017     */
018    public interface DDF_Resolver {
019    
020            enum Mode {
021    
022                    /** start index update only if index is old */
023                    DEFAULT, //
024    
025                    /** remove and create the index anew */
026                    REBUILD, //
027    
028                    /** start index update now, even if not due */
029                    REINDEX, //
030    
031            }
032    
033            /**
034             * start index;
035             * 
036             * TODO @return readiness future.
037             * 
038             * @param mode
039             *            the mode
040             * @return the future
041             */
042            public Future<?> open(Mode mode);
043    
044            /**
045             * TODO
046             * 
047             * stop index; release resources
048             */
049            void close();
050    
051            /**
052             * LUCENE advanced search syntax;
053             * 
054             * http://lucene.apache.org/java/3_4_0/queryparsersyntax.html
055             */
056            List<DDF_Instrument> searchLucene(String phrase) throws Exception;
057    
058            /**
059             * BARCHART simplified search syntax;
060             * 
061             * 0) use "space" as term separator;
062             * 
063             * 1) if "*" or "?" are present, treat term as WILDCARD;
064             * 
065             * 2) else use "term" as prefix; i.e. "term*"
066             * 
067             * 3) combine all terms with "AND";
068             * 
069             */
070            List<DDF_Instrument> searchSimple(String phrase) throws Exception;
071    
072    }
