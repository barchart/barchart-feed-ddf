231            /** byte sized enum ordinal */

240            // /////////////////////
241    
242            private DDF_ExchangeChannel(final String code, final DDF_Exchange exchange) {
243                    this.ord = (byte) ordinal();
244                    this.code = code;
245                    this.exchange = exchange;
246            }
247    
248            private final static DDF_ExchangeChannel[] ENUM_VALUES = values();
249    
250            /**
251             * Values unsafe.
252             *
253             * @return the dD f_ exchange channel[]
254             */
255            @Deprecated
256            public final static DDF_ExchangeChannel[] valuesUnsafe() {
257                    return ENUM_VALUES;
258            }
259    
260            static {
261                    // validate use of byte ord
262                    MathExtra.castIntToByte(ENUM_VALUES.length);
263            }
264    
265            /**
266             * From code.
267             *
268             * @param code the code
269             * @return the dD f_ exchange channel
270             */
271            public final static DDF_ExchangeChannel fromCode(final String code) {
272                    for (final DDF_ExchangeChannel known : ENUM_VALUES) {
273                            if (known.code.equalsIgnoreCase(code)) {
274                                    return known;
275                            }
276                    }
277                    return UNKNOWN;
278            }
279    
280            /**
281             * From ord.
282             *
283             * @param ord the ord
284             * @return the dD f_ exchange channel
285             */
286            public final static DDF_ExchangeChannel fromOrd(final byte ord) {
287                    return ENUM_VALUES[ord];
288            }
289    
290            /**
291             * Checks if is known.
292             *
293             * @return true, if is known
294             */
295            public final boolean isKnown() {
296                    return this != UNKNOWN;
297            }
