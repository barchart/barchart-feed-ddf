020    /**
021     * Flags for ddf JERQ "go" command to activate specific information channels in
022     * the feed.
023     */
024    public enum DDF_FeedInterest {

026            UNKNOWN('?'), //
027    
028            /** A one-time snapshot of a symbol's full book */
029            BOOK_SNAPSHOT('b'), //
030            /** A stream of the symbol's top of book */
031            BOOK_UPDATE('B'), // only book top
032    
033            //
034            /** A one-time cumulative volume snapshot */
035            CUVOL_SNAPSHOT('c'), //
036    
037            //
038            /** A one-time snapshot quote for a symbol */
039            QUOTE_SNAPSHOT('s'), //
040            /** A stream of quote updates for a symbol */
041            QUOTE_UPDATE('S'), //
042    
043            ;
044    
045            private static final DDF_FeedInterest[] ENUM_VALS = values();
046    
047            public static final Set<DDF_FeedInterest> setValues() {
048                    final Set<DDF_FeedInterest> vals = new HashSet<DDF_FeedInterest>();
049                    for (final DDF_FeedInterest i : values()) {
050                            vals.add(i);
051                    }
052                    vals.remove(DDF_FeedInterest.UNKNOWN);
053                    return vals;
054            }
055    
056            public static final int size() {
057                    return ENUM_VALS.length;
058            }
059    
060            /** The code. */
061            public final char code;
062    
063            private DDF_FeedInterest(final char code) {
064                    this.code = code;
065            }
066    
067            public static final DDF_FeedInterest of(final char code) {
068                    for (final DDF_FeedInterest known : values()) {
069                            if (known.code == code) {
070                                    return known;
071                            }
072                    }
073                    return UNKNOWN;
074            }

145            /**
146             * Parses a set of MarketEvents to a Set of FeedInterest enums.
147             * 
148             * @param eventSet
149             *            The set of events to parse.
150             * @return a Set of FeedInterest enums.
151             */
152            public static final Set<DDF_FeedInterest> fromEvents(
153                            final Set<MarketEvent> eventSet) {
154    
155                    final Set<DDF_FeedInterest> result =
156                                    EnumSet.noneOf(DDF_FeedInterest.class);
157    
158                    if (eventSet == null || eventSet.isEmpty()) {
159                            return result;
160                    }
161    
162                    for (final MarketEvent event : eventSet) {
163                            switch (event) {
164                            
165                            case MARKET_UPDATED:
166                                    result.addAll(DDF_FeedInterest.setValues());
167                                    break;
168                                    
169                            //MARKET_STATUS_* not supported by ddf
170                                    
171                            case NEW_TRADE:
172                                    result.add(QUOTE_UPDATE);
173                                    continue;
174                                    
175                            case NEW_BAR_CURRENT:
176                            case NEW_BAR_PREVIOUS:
177                            case NEW_OPEN:
178                            case NEW_HIGH:
179                            case NEW_LOW:
180                            case NEW_CLOSE:
181                            case NEW_SETTLE:
182                            case NEW_VOLUME:
183                            case NEW_INTEREST:
184                                    result.add(QUOTE_SNAPSHOT);
185                                    result.add(QUOTE_UPDATE);
186                                    continue;
187                                    
188                            case NEW_BOOK_ERROR:
189                                    // debug use only
190                                    continue;
191    
192                            case NEW_BOOK_SNAPSHOT:
193                                    result.add(BOOK_SNAPSHOT);
194                                    continue;
195    
196                            case NEW_BOOK_UPDATE:
197                            case NEW_BOOK_TOP:
198                                    result.add(BOOK_UPDATE);
199                                    result.add(BOOK_SNAPSHOT);
200                                    continue;
201    
202                            //NEW_CUVOL_UPDATE not supported by ddf
203                                    
204                            case NEW_CUVOL_SNAPSHOT:
205                                    result.add(CUVOL_SNAPSHOT);
206                                    continue;
207    
208                            default:
209                                    result.add(QUOTE_UPDATE);
210                                    continue;
211                            }
212                    }
213    
214                    return result;
215    
216            }
217    
218            public static String from(final Collection<DDF_FeedInterest> interests) {
219                    final StringBuilder sb = new StringBuilder();
220                    for (final DDF_FeedInterest i : interests) {
221                            sb.append(i.code);
222                    }
223                    return sb.toString();
224            }

226            /**
227             * Command.
228             * 
229             * @param instrumentDDF
230             *            the instrument ddf
231             * @param eventSet
232             *            the event set
233             * @return the char sequence
234             */
235            public static CharSequence command(final DDF_Instrument instrumentDDF,
236                            final Set<MarketEvent> eventSet) {
237    
238                    final CharSequence symbol =
239                                    instrumentDDF.get(DDF_InstrumentField.DDF_SYMBOL_REALTIME);
240    
241                    final CharSequence interest = DDF_FeedInterest.from(eventSet);
242    
243                    final CharSequence command = FeedDDF.tcpGo(symbol, interest);
244    
245                    return command;
246    
247            }
