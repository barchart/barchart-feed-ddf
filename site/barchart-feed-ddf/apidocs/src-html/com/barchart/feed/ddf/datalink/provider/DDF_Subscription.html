011    package com.barchart.feed.ddf.datalink.provider;
012    
013    import java.util.Set;
014    
015    import com.barchart.feed.api.model.meta.Instrument;
016    import com.barchart.feed.base.market.enums.MarketEvent;
017    import com.barchart.feed.base.sub.Subscription;
018    import com.barchart.feed.base.sub.SubscriptionType;
019    
020    /**
021     * Represents a subscription to a single instrument for JERQ.
022     */
023    public class DDF_Subscription implements Subscription {
024    
025            private final String interest;
026            private final Set<SubscriptionType> interests;
027    
028            /**
029             * @param instrument
030             *            The instrument to subscribe or unsubscribe to
031             * @param interests
032             *            A set of DDF_FeedInterests specifying what data for JERQ to
033             *            send a client. An empty set is interpreted as a request to
034             *            unsubscribe.
035             */
036            public DDF_Subscription(final String instrument,
037                            final Set<SubscriptionType> interests) {
038                    this.interest = instrument;
039                    this.interests = interests;
040            }
041            
042            public DDF_Subscription(final Subscription sub) {
043                    interest = sub.interest();
044                    interests = sub.types();
045            }
046    
047            /**
048             * A feed-base friendly constructor for convenience.
049             * 
050             * @param instrument
051             * @param events
052             */
053            public DDF_Subscription(final Instrument instrument,
054                            final Set<MarketEvent> events) {
055                    this.interest =
056                                    instrument.symbol();
057                    this.interests = DDF_FeedInterest.fromEvents(events);
058            }
059    
060            @Override
061            public Set<SubscriptionType> types() {
062                    return interests;
063            }
064            
065            @Override
066            public String interest() {
067                    return interest;
068            }
069            
070            @Override
071            public String encode() {
072                    return interest + "=" + DDF_FeedInterest.from(interests);
073            }
074    
075            @Override
076            public void addTypes(final Set<SubscriptionType> insts) {
077                    interests.addAll(insts);
078            }
079            
080            @Override
081            public void removeTypes(final Set<SubscriptionType> insts) {
082                    interests.removeAll(insts);
083            }
084            
085            /**
086             * Returns the JERQ command to request this subscription.
087             */
088            @Override
089            public String toString() {
090                    return interest + " " + DDF_FeedInterest.from(interests);
091            }
092    
093            @Override
094            public boolean isNull() {
095                    return false;
096            }
097    
098    }
