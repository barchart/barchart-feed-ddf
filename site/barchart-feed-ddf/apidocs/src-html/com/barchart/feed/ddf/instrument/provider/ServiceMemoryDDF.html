034    /**
035     * keeps in memory cache.
036     */
037    @ThreadSafe
038    public class ServiceMemoryDDF implements DDF_DefinitionService {
039    
040            static final Logger log = LoggerFactory.getLogger(ServiceMemoryDDF.class);
041    
042            private final ConcurrentMap<InstrumentGUID, Instrument> guidMap = 
043                            new ConcurrentHashMap<InstrumentGUID, Instrument>();
044            
045            private final LocalCacheSymbologyContextDDF cache = 
046                            new LocalCacheSymbologyContextDDF();
047            private final SymbologyContext<CharSequence> remote = 
048                            new RemoteSymbologyContextDDF(guidMap);
049                            
050            private final ExecutorService executor;
051            
052            /**
053             * Instantiates a new service memory ddf.  Uses system default cached
054             * thread pool executor.
055             */
056            public ServiceMemoryDDF() {
057                    executor = Executors.newCachedThreadPool();
058            }
059            
060            /**
061             * Instantiates a new service memory ddf using specified executor service.
062             * @param executor
063             */
064            public ServiceMemoryDDF(final ExecutorService executor) {
065                    this.executor = executor;
066            }
067    
068            @Override
069            public List<Instrument> lookup(final CharSequence symbol) {
070                    
071                    try {
072                            return retrieve(symbol);
073                    } catch (final Throwable t) {
074                            log.error("Lookup failed", t);
075                            return Collections.emptyList();
076                    }
077                    
078            }
079    
080            @Override
081            public Map<CharSequence, List<Instrument>> lookup(
082                            Collection<? extends CharSequence> symbols) {
083                    
084                    try {
085                            return retrieveMap(symbols);
086                    } catch (final Throwable t) {
087                            log.error("Lookup failed", t);
088                            return Collections.emptyMap();
089                    }
090                    
091            }
092    
093            @Override
094            public InstrumentFuture lookupAsync(CharSequence symbol) {
095                    
096                    InstrumentFuture future = new InstrumentFuture();
097                    
098                    executor.submit(new LookupCallable(symbol, future));
099                    
100                    return future;
101                    
102            }
103    
104            @Override
105            public InstrumentFutureMap<CharSequence> lookupAsync(
106                            Collection<? extends CharSequence> symbols) {
107                    
108                    InstrumentFutureMap<CharSequence> future = 
109                                    new InstrumentFutureMap<CharSequence>();
110                    
111                    executor.submit(new LookupMapCallable(symbols, future));
112                    
113                    return future;
114                    
115            }
116            
117            private final class LookupCallable implements Callable<List<Instrument>> {
118    
119                    private final CharSequence symbol;
120                    private final InstrumentFuture future;
121                    
122                    LookupCallable(final CharSequence symbol, final InstrumentFuture future) {
123                            this.symbol = symbol;
124                            this.future = future;
125                    }
126                    
127                    @Override
128                    public List<Instrument> call() throws Exception {
129                            
130                            List<Instrument> inst;
131                            
132                            try {
133                            
134                                    inst = retrieve(symbol);
135                                    future.succeed(inst);
136                                    
137                            } catch (final Throwable t) {
138                                    future.fail(t);
139                                    return Collections.emptyList();
140                            }
141                            
142                            return inst;
143                            
144                    }
145                    
146            }
147            
148            private List<Instrument> retrieve(final CharSequence symbol) {
149                    
150                    if(symbol == null || symbol.length() == 0) {
151                            return Collections.emptyList();
152                    }
153                    
154                    InstrumentGUID guid = cache.lookup(symbol.toString().toUpperCase()); 
155                                    
156                    if(guid.isNull()) {
157                            guid = remote.lookup(symbol.toString().toUpperCase());
158                    }
159                    
160                    if(guid.equals(InstrumentGUID.NULL)) {
161                            return Collections.emptyList();
162                    }
163                    
164                    cache.storeGUID(symbol, guid);
165                    
166                    final InstrumentDDF instrument = (InstrumentDDF) guidMap.get(guid);
167    
168                    if (instrument == null) {
169                            return Collections.emptyList();
170                    }
171    
172                    Instrument inst = ObjectMapFactory.build(InstrumentDDF.class, instrument);
173                    
174                    return Collections.singletonList(inst);
175                    
176            }
177            
178            private final class LookupMapCallable implements 
179                            Callable<Map<CharSequence, List<Instrument>>> {
180                    
181                    private final Collection<? extends CharSequence> symbols;
182                    private final InstrumentFutureMap<CharSequence> future;
183                    
184                    LookupMapCallable(final Collection<? extends CharSequence> symbols,
185                                    final InstrumentFutureMap<CharSequence> future) {
186                            this.symbols = symbols;
187                            this.future = future;
188                    }
189    
190                    @Override
191                    public Map<CharSequence, List<Instrument>> call() throws Exception {
192                            
193                            Map<CharSequence, List<Instrument>> map;
194                            
195                            try {
196                                    
197                                    map = retrieveMap(symbols);
198                                    future.succeed(map);
199                                    
200                            } catch (final Throwable t) {
201                                    future.fail(t);
202                                    return Collections.emptyMap();
203                            }
204                            
205                            return map;
206                    }
207                    
208            }
209            
210            private Map<CharSequence, List<Instrument>> retrieveMap(
211                            final Collection<? extends CharSequence> symbols) {
212                    
213                    if (symbols == null || symbols.size() == 0) {
214                            log.warn("Lookup called with empty collection");
215                            return Collections.emptyMap(); 
216                    }
217                    
218                    final Map<CharSequence, List<Instrument>> result = 
219                                    new HashMap<CharSequence, List<Instrument>>();
220    
221                    for (final CharSequence symbol : symbols) {
222                            try {
223                                    result.put(symbol.toString(), retrieve(symbol));
224                            } catch (final Exception e) {
225                                    result.put(symbol.toString(), new ArrayList<Instrument>(0));
226                            }
227                    }
228                    
229                    return result;
230                    
231            }
232            
233    }
