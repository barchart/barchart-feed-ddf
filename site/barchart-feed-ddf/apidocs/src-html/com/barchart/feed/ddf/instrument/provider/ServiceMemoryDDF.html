011    import static com.barchart.feed.ddf.instrument.enums.DDF_InstrumentField.DDF_SYMBOL_REALTIME;
012    import static com.barchart.feed.ddf.instrument.enums.DDF_InstrumentField.DDF_SYMBOL_UNIVERSAL;
013    import static com.barchart.feed.ddf.symbol.provider.DDF_Symbology.lookupFromSymbol;
014    
015    import java.util.ArrayList;
016    import java.util.List;
017    import java.util.concurrent.ConcurrentHashMap;
018    import java.util.concurrent.ConcurrentMap;
019    
020    import org.slf4j.Logger;
021    import org.slf4j.LoggerFactory;

029    /**
030     * keeps in memory cache.
031     */
032    @ThreadSafe
033    public class ServiceMemoryDDF extends ServiceBasicDDF {
034    
035            static final Logger log = LoggerFactory.getLogger(ServiceMemoryDDF.class);
036    
037            private final ConcurrentMap<TextValue, DDF_Instrument> instrumentMap = //
038                            new ConcurrentHashMap<TextValue, DDF_Instrument>();
039    
040            private final ConcurrentMap<TextValue, DDF_Instrument> ddfInstrumentMap = //
041                            new ConcurrentHashMap<TextValue, DDF_Instrument>();
042    
043            /**
044             * Instantiates a new service memory ddf.
045             */
046            public ServiceMemoryDDF() {
047            }
048    
049            /*
050             * (non-Javadoc)
051             * 
052             * @see com.barchart.feed.ddf.instrument.provider.ServiceBasicDDF#clear()
053             */
054            @Override
055            public final void clear() {
056                    instrumentMap.clear();
057                    ddfInstrumentMap.clear();
058            }
059    
060            /** make an upper case id */
061            private DDF_Instrument load(final TextValue symbol) {
062    
063                    final TextValue lookup = lookupFromSymbol(symbol);
064    
065                    return instrumentMap.get(lookup);
066    
067            }
068    
069            /** this will make 3 entries for futures and 1 entry for equities */
070            private void store(final DDF_Instrument instrument) {
071    
072                    /**
073                     * making assumption that first lookup of ESM0 will set symbol GUID per
074                     * DDF convention; that is ESM2020 in year 2011;
075                     * "if symbol expired, move forward"
076                     * 
077                     * this logic can overwrite previously defined symbol; say we are in
078                     * year 2011; say ESM0 was already defined for ESM2020 resolution; now
079                     * request comes for ESM2010; now ESM0 will resolve to ESM2010, and not
080                     * ESM2020
081                     * 
082                     * @author g-litchfield Removed mapping by DDF_SYMBOL_REALTIME because
083                     *         this is not always unique and was causing caching problems,
084                     *         specifically in KCK2 vs KCK02
085                     * 
086                     */
087    
088                    final TextValue symbolDDF =
089                                    instrument.get(DDF_SYMBOL_REALTIME).toUpperCase();
090                    final TextValue symbolHIST =
091                                    instrument.get(DDF_SYMBOL_HISTORICAL).toUpperCase();
092                    final TextValue symbolGUID =
093                                    instrument.get(DDF_SYMBOL_UNIVERSAL).toUpperCase();
094    
095                    ddfInstrumentMap.put(symbolDDF, instrument);
096    
097                    // hack for bats
098    
099                    if (symbolDDF.toString().contains(".BZ")) {
100                            final TextValue lookup =
101                                            ValueBuilder.newText(symbolDDF.toString()
102                                                            .replace(".BZ", ""));
103    
104                            instrumentMap.put(lookup, instrument);
105                    }
106    
107                    instrumentMap.put(symbolHIST, instrument);
108                    instrumentMap.put(symbolGUID, instrument);
109    
110                    log.debug("defined instrument={}", symbolGUID);
111    
112            }
113    
114            /*
115             * (non-Javadoc)
116             * 
117             * @see
118             * com.barchart.feed.ddf.instrument.provider.ServiceBasicDDF#lookupDDF(com
119             * .barchart.util.values.api.TextValue)
120             */
121            @Override
122            public DDF_Instrument lookupDDF(final TextValue symbol) {
123    
124                    if (CodecHelper.isEmpty(symbol)) {
125                            return DDF_InstrumentProvider.NULL_INSTRUMENT;
126                    }
127    
128                    final TextValue lookup = symbol.toUpperCase();
129    
130                    DDF_Instrument instrument = ddfInstrumentMap.get(lookup);
131    
132                    if (instrument == null) {
133                            
134                            try {
135    
136                                    instrument = DDF_InstrumentProvider.remoteLookup(symbol);
137    
138                                    store(instrument);
139    
140                            } catch (final Exception e) {
141                                    log.warn("lookupDDF : instrument lookup failed; symbol : {}",
142                                                    symbol);
143                                    return DDF_InstrumentProvider.NULL_INSTRUMENT;
144                            }
145    
146                    }
147    
148                    return instrument;
149    
150            }
151    
152            /*
153             * (non-Javadoc)
154             * 
155             * @see
156             * com.barchart.feed.ddf.instrument.provider.ServiceBasicDDF#lookup(com.
157             * barchart.util.values.api.TextValue)
158             */
159            @Override
160            public final DDF_Instrument lookup(final TextValue symbol) {
161    
162                    if (CodecHelper.isEmpty(symbol)) {
163                            return DDF_InstrumentProvider.NULL_INSTRUMENT;
164                    }
165    
166                    DDF_Instrument instrument = load(symbol);
167    
168                    if (instrument == null) {
169    
170                            try {
171    
172                                    instrument = DDF_InstrumentProvider.remoteLookup(symbol);
173    
174                                    store(instrument);
175    
176                            } catch (final Exception e) {
177                                    log.warn("instrument lookup failed; symbol : {}", symbol);
178                                    return DDF_InstrumentProvider.NULL_INSTRUMENT;
179                            }
180    
181                    }
182    
183                    return instrument;
184    
185            }
186    
187            /*
188             * (non-Javadoc)
189             * 
190             * @see
191             * com.barchart.feed.ddf.instrument.provider.ServiceBasicDDF#lookup(java
192             * .util.List)
193             */
194            @Override
195            public List<DDF_Instrument> lookup(final List<String> symbolList) {
196    
197                    if (CodecHelper.isEmpty(symbolList)) {
198                            return DDF_InstrumentProvider.NULL_LIST;
199                    }
200    
201                    final int size = symbolList.size();
202    
203                    final List<String> fetchList = new ArrayList<String>(size);
204    
205                    final List<DDF_Instrument> oldList =
206                                    new ArrayList<DDF_Instrument>(size);
207    
208                    for (final String symbol : symbolList) {
209    
210                            final TextValue id = ValueBuilder.newText(symbol);
211    
212                            final DDF_Instrument instrument = load(id);
213    
214                            if (instrument == null) {
215                                    fetchList.add(symbol);
216                            } else {
217                                    oldList.add(instrument);
218                            }
219    
220                    }
221    
222                    final List<DDF_Instrument> newList =
223                                    DDF_InstrumentProvider.fetch(fetchList);
224    
225                    for (final DDF_Instrument instrument : newList) {
226                            store(instrument);
227                    }
228    
229                    oldList.addAll(newList);
230    
231                    return oldList;
232    
233            }
234
