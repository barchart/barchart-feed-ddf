152            private final static DDF_Exchange[] ENUM_VALUES = values();
153    
154            static {
155                    // validate use of byte ord
156                    MathExtra.castIntToByte(ENUM_VALUES.length);
157            }

159            /**
160             * From code.
161             *
162             * @param code the code
163             * @return the dD f_ exchange
164             */
165            public final static DDF_Exchange fromCode(final byte code) {
166                    for (final DDF_Exchange known : ENUM_VALUES) {
167                            if (known.code == code) {
168                                    return known;
169                            }
170                    }
171                    return UNKNOWN;
172            }
