214            private DDF_Exchange(final byte code, final DDF_ExchangeKind kind) {
215    
216                    this.ord = (byte) ordinal();
217                    this.code = code;
218                    this.kind = kind;
219    
220                    this.description = description();
221            }
222    
223            /** TODO use real human readable description as ENUM property */
224            private String description() {
225                    return name().replaceAll("_", " ");
226            }
227    
228            private final static DDF_Exchange[] ENUM_VALUES = values();
229    
230            static {
231                    // validate use of byte ord
232                    MathExtra.castIntToByte(ENUM_VALUES.length);
233            }
234    
235            /**
236             * From code.
237             *
238             * @param code the code
239             * @return the dD f_ exchange
240             */
241            public final static DDF_Exchange fromCode(final byte code) {
242                    for (final DDF_Exchange known : ENUM_VALUES) {
243                            if (known.code == code) {
244                                    return known;
245                            }
246                    }
247                    return UNKNOWN;
248            }
249    
250            /**
251             * From ord.
252             *
253             * @param ord the ord
254             * @return the dD f_ exchange
255             */
256            public final static DDF_Exchange fromOrd(final byte ord) {
257                    return ENUM_VALUES[ord];
258            }
