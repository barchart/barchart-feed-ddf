056    /** a.k.a base code */
057    public enum DDF_Fraction implements EnumCodeByte, EnumByteOrdinal {
058    
059            // binary range; exponent is power of 2;
060            Q2(_0_, -10, BIN_N01), // 1/2 XXX: unit code invalid
061            Q4(_1_, -11, BIN_N02), // 1/4 XXX: unit code invalid
062            Q8(_2_, -1, BIN_N03), // 1/8
063            Q16(_3_, -2, BIN_N04), // 1/16
064            Q32(_4_, -3, BIN_N05), // 1/32
065            Q64(_5_, -4, BIN_N06), // 1/64
066            Q128(_6_, -5, BIN_N07), // 1/128
067            Q256(_7_, -6, BIN_N08), // 1/256
068    
069            // decimal range; exponent is power of 10;
070            Z0(_8_, 0, DEC_Z00), // 1
071            N1(_9_, 1, DEC_N01), // 10
072            N2(_A_, 2, DEC_N02), // 100
073            N3(_B_, 3, DEC_N03), // 1K
074            N4(_C_, 4, DEC_N04), // 10K
075            N5(_D_, 5, DEC_N05), // 100K
076            N6(_E_, 6, DEC_N06), // 1M
077            N7(_F_, 7, DEC_N07), // 10M
078            N8(_G_, 8, DEC_N08), // 100M
079            N9(_H_, 9, DEC_N09), // 1B
080    
081            ZZ(STAR, 0, DEC_Z00), // 1
082    
083            UNKNOWN(QUEST, 0, DEC_Z00), //
084    
085            /*
086             * (non-Javadoc)
087             * 
088             * @see com.barchart.util.enums.EnumByteOrdinal#ord()
089             */
090            ;
091    
092            @Override
093            public final byte ord() {
094                    return ord;
095            }
096    
097            /*
098             * (non-Javadoc)
099             * 
100             * @see com.barchart.util.enums.EnumCodeByte#code()
101             */
102            @Override
103            public final byte code() {
104                    return baseCode;
105            }
106    
107            /** form #1 of ddf price exponent encoding. */
108            public final byte baseCode;
109    
110            /** form #2 of ddf price exponent encoding. */
111            public final int unitCode;
112    
113            /** byte sized ordinal of this enum. */
114            public final byte ord;
115    
116            /** base-10 exponent and denominator, regardless of enum base. */
117            public final int decimalExponent;
118    
119            /** The decimal denominator. */
120            public final long decimalDenominator;
121    
122            /** base-2 or base-10 exponent and denominator, depending on enum base. */
123            public final int nativeExponent;
124    
125            /** The native denominator. */
126            public final long nativeDenominator;
127    
128            /** number of digits needed to display fraction in native form. */
129            public final int spaces;
130    
131            /** "positive" difference between native and decimal spaces. */
132            public final long spacesDenomPlus;
133    
134            /** "negative" difference between native and decimal spaces. */
135            public final long spacesDenomMinus;
136    
137            /** base generic fraction api enum used by ddf fraction enum. */
138            public final Fraction fraction;
139    
140            /** convenience flag to differentiate base-2 vs base-10 fractions. */
141            public final boolean isBinary;
142    
143            private DDF_Fraction(final byte baseCode, final int unitCode,
144                            final Fraction fraction) {
145    
146                    this.baseCode = baseCode;
147                    this.unitCode = unitCode;
148    
149                    this.ord = (byte) ordinal();
150    
151                    this.fraction = fraction;
152                    this.decimalExponent = fraction.decimalExponent;
153                    this.decimalDenominator = fraction.decimalDenominator;
154    
155                    this.isBinary = fraction.base.isBinary();
156    
157                    this.nativeExponent = fraction.exponent.value;
158                    this.nativeDenominator = fraction.denominator;
159    
160                    this.spaces = fraction.places;
161    
162                    this.spacesDenomPlus = (long) pow(10, spaces);
163                    this.spacesDenomMinus = (long) pow(10, -decimalExponent - spaces);
164    
165            }
166    
167            private final static DDF_Fraction[] ENUM_VALUES = values();
168    
169            /**
170             * Values unsafe.
171             * 
172             * @return the dD f_ fraction[]
173             */
174            @Deprecated
175            public final static DDF_Fraction[] valuesUnsafe() {
176                    return ENUM_VALUES;
177            }
178    
179            static {
180                    // validate use of byte ord
181                    MathExtra.castIntToByte(ENUM_VALUES.length);
182            }
183

187             * From base code.
188             * 
189             * @param baseCode
190             *            the base code
191             * @return the dD f_ fraction
192             */
193            public final static DDF_Fraction fromBaseCode(final byte baseCode) {
194                    for (final DDF_Fraction known : ENUM_VALUES) {
195                            if (known.baseCode == baseCode) {
196                                    return known;
197                            }
198                    }
199                    return UNKNOWN;
200            }
201

204            /**
205             * From unit code.
206             * 
207             * @param unitCode
208             *            the unit code
209             * @return the dD f_ fraction
210             */
211            public final static DDF_Fraction fromUnitCode(final int unitCode) {
212                    for (final DDF_Fraction known : ENUM_VALUES) {
213                            if (known.unitCode == unitCode) {
214                                    return known;
215                            }
216                    }
217                    return UNKNOWN;
218            }
219    
220            // TODO optimize; replace with 1 tableswitch block
221            /**
222             * From fraction.
223             * 
224             * @param fraction
225             *            the fraction
226             * @return the dD f_ fraction
227             */
228            public final static DDF_Fraction fromFraction(final Fraction fraction) {
229                    for (final DDF_Fraction known : ENUM_VALUES) {
230                            if (known.fraction == fraction) {
231                                    return known;
232                            }
233                    }
234                    return UNKNOWN;
235            }
236    
237            /**
238             * From ord.
239             * 
240             * @param ord
241             *            the ord
242             * @return the dD f_ fraction
243             */
244            public final static DDF_Fraction fromOrd(final byte ord) {
245                    return ENUM_VALUES[ord];
246            }
247    
248            /**
249             * Checks if is known.
250             * 
251             * @return true, if is known
252             */
253            public final boolean isKnown() {
254                    return this != UNKNOWN;
255            }
