008    package com.barchart.feed.ddf.datalink.provider;
009    
010    
011    import java.util.concurrent.Executor;
012    
013    import org.jboss.netty.channel.ChannelPipeline;
014    import org.jboss.netty.channel.ChannelPipelineFactory;
015    import org.jboss.netty.channel.Channels;
016    
017    import com.barchart.feed.ddf.datalink.api.DDF_SocksProxy;
018    
019    public class PipelineFactorySocks implements ChannelPipelineFactory {
020    
021            @SuppressWarnings("unused")
022            private Executor exec;
023            private FeedClientDDF feedClient;
024            private DDF_SocksProxy proxySettings;
025    
026            public PipelineFactorySocks(Executor pipelineExecutor, FeedClientDDF feedClient, DDF_SocksProxy proxy) {
027                     
028                     this.exec = pipelineExecutor;
029                     this.feedClient = feedClient;
030                     this.proxySettings = proxy;
031                     
032            }
033    
034            @Override
035            public ChannelPipeline getPipeline() throws Exception {
036    
037                    final ChannelPipeline pipeline = Channels.pipeline();
038    
039                    pipeline.addLast("socks5handler", new SocksClientHandler(feedClient, proxySettings));
040                    
041                    // remove after debug
042                    //pipeline.addLast("ddf logger", new LoggerDDF());
043                    
044                    return pipeline;
045            }
046    
047    }
