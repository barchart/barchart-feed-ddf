001    package com.barchart.feed.ddf.message.provider;
002    
003    import static com.barchart.util.ascii.ASCII.COMMA;
004    
005    import java.nio.ByteBuffer;
006    
007    import com.barchart.feed.ddf.message.api.DDF_MarketParameter;
008    import com.barchart.feed.ddf.message.api.DDF_MessageVisitor;
009    import com.barchart.feed.ddf.message.enums.DDF_MessageType;
010    import com.barchart.feed.ddf.message.enums.DDF_ParamType;
011    import com.barchart.feed.ddf.util.HelperDDF;
012    import com.barchart.util.values.api.PriceValue;
013    import com.barchart.util.values.api.SizeValue;
014    
015    public class DF_25_Param extends BaseMarket implements DDF_MarketParameter {
016    
017            @Override
018            public <Result, Param> Result accept(
019                            DDF_MessageVisitor<Result, Param> visitor, Param param) {
020                    return visitor.visit(this, param);
021            }
022            
023            DF_25_Param() {
024                    super(DDF_MessageType.DDF_25);
025            }
026            
027            DF_25_Param(final DDF_MessageType messageType) {
028                    super(messageType);
029            }
030            
031            // //////////////////////////////////////
032    
033            private byte ordParam = DDF_ParamType.UNKNOWN.ord;
034    
035            /**
036             * can represent:
037             * 
038             * 1) price (with exponent from fraction);
039             * 
040             * 2) size (regardless of the exponent in fraction):
041             */
042            protected long value = HelperDDF.DDF_EMPTY;
043    
044            // //////////////////////////////////////
045    
046            
047            @Override
048            public DDF_ParamType getParamType() {
049                    return DDF_ParamType.fromOrd(ordParam);
050            }
051            
052            /**
053             * Sets the param type.
054             *
055             * @param type the new param type
056             */
057            public final void setParamType(final DDF_ParamType type) {
058                    ordParam = type.ord;
059            }
060    
061            @Override
062            public PriceValue getAsPrice() {
063                    return HelperDDF.newPriceDDF(value, getFraction());
064            }
065    
066            @Override
067            public SizeValue getAsSize() {
068                    return HelperDDF.newSizeDDF(value);
069            }
070    
071            @Override
072            protected final void encodeBody(final ByteBuffer buffer) {
073    
074                    final DDF_ParamType param = getParamType();
075    
076                    switch (param.kind) {
077                    default:
078                    case SIZE:
079                            HelperDDF.longEncode(value, buffer, COMMA); // <value>,
080                            break;
081                    case PRICE:
082                            HelperDDF.decimalEncode(value, getFraction(), buffer, COMMA); // <value>,
083                            break;
084                    }
085    
086                    buffer.putChar(param.code); // <element> <modifier>
087    
088            }
089            
090            @Override
091            protected final void decodeBody(final ByteBuffer buffer) {
092    
093                    final long value = HelperDDF.longDecode(buffer, COMMA); // <value>,
094    
095                    final char code = buffer.getChar(); // <element> <modifier>
096    
097                    final DDF_ParamType param = DDF_ParamType.fromCode(code);
098    
099                    ordParam = param.ord;
100    
101                    switch (param.kind) {
102                    default:
103                    case SIZE:
104                            this.value = value;
105                            break;
106                    case PRICE:
107                            this.value = HelperDDF.fromBinaryToDecimal(value, getFraction());
108                            break;
109                    }
110    
111            }
112    
113            @Override
114            protected void appedFields(final StringBuilder text) {
115    
116                    super.appedFields(text);
117    
118                    text.append("param type  : ");
119                    text.append(getParamType());
120                    text.append("\n");
121    
122                    text.append("param value : ");
123                    text.append(value);
124                    text.append("\n");
125    
126            }
127            
128    }
