010    /**
011     * The listener interface for receiving DDF_Result events.
012     * The class that is interested in processing a DDF_Result
013     * event implements this interface, and the object created
014     * with that class is registered with a component using the
015     * component's <code>addDDF_ResultListener<code> method. When
016     * the DDF_Result event occurs, that object's appropriate
017     * method is invoked.
018     *
019     * @see DDF_ResultEvent
020     */
021    public interface DDF_ResultListener {
022    
023            int PROGRESS_SIZE = 1000;
024    
025            void onProgressEvent(int progressCount);
026    
027    }
