046    public class HistoricReplayListenerClientDDF extends SimpleChannelHandler implements
047                    DDF_FeedClientBase {
048    
049            /** use slf4j for internal NETTY LoggingHandler facade */
050            static {
051                    final InternalLoggerFactory defaultFactory = new Slf4JLoggerFactory();
052                    InternalLoggerFactory.setDefaultFactory(defaultFactory);
053            }
054    
055            private static final Logger log = LoggerFactory
056                            .getLogger(HistoricReplayListenerClientDDF.class);
057    
058            private final ConnectionlessBootstrap boot;
059    
060            private DatagramChannel channel;
061    
062            private volatile DDF_MessageListener msgListener = null;
063    
064            private final Executor runner;
065    
066            private final int socketAddress;
067    
068            private final Map<String, Sub> subscriptions = 
069                            new ConcurrentHashMap<String, Sub>();
070            
071            HistoricReplayListenerClientDDF(final int socketAddress, final Executor executor) {
072    
073                    this.socketAddress = socketAddress;
074                    runner = executor;
075    
076                    final DatagramChannelFactory channelFactory = new NioDatagramChannelFactory(
077                                    runner);
078    
079                    boot = new ConnectionlessBootstrap(channelFactory);
080    
081                    final ChannelPipelineFactory pipelineFactory = new PipelineFactoryDDF(
082                                    this);
083    
084                    boot.setPipelineFactory(pipelineFactory);
085    
086                    boot.setOption("broadcast", "false");
087                    boot.setOption("receiveBufferSizePredictorFactory", //
088                                    new FixedReceiveBufferSizePredictorFactory(2 * 1024));
089    
090            }
091    
092            private final BlockingQueue<DDF_BaseMessage> messageQueue = new LinkedBlockingQueue<DDF_BaseMessage>();
093    
094            private final RunnerDDF messageTask = new RunnerDDF() {
095    
096                    @Override
097                    protected void runCore() {
098                            while (true) {
099                                    try {
100                                            final DDF_BaseMessage message = messageQueue.take();
101    
102                                            if (msgListener != null && filter(message)) {
103                                                    msgListener.handleMessage(message);
104                                            }
105    
106                                    } catch (final InterruptedException e) {
107                                            log.trace("terminated");
108                                            return;
109                                    } catch (final Throwable e) {
110                                            log.error("message delivery failed", e);
111                                    }
112                            }
113                    }
114    
115            };
116            
117            private boolean filter(final DDF_BaseMessage message) {
118                    
119                    /* Filter by market message */
120                    if(!message.getMessageType().isMarketMessage) {
121                            return false;
122                    }
123                    
124                    final DDF_MarketBase marketMsg = (DDF_MarketBase) message;
125                    
126                    /* Filter by instrument */
127                    if(subscriptions.containsKey(marketMsg.getInstrument().symbol())) {
128                            
129                            return true;
130                    }
131                    
132                    return false;
133            }
134    
135            @Override
136            public void startup() {
137    
138                    runner.execute(messageTask);
139    
140                    boot.bind(new InetSocketAddress(socketAddress));
141    
142            }
143    
144            @Override
145            public void shutdown() {
146    
147                    //In FeedClientDDF, subscriptions are cleared on logout, do we want to do this here?
148                    
149                    messageTask.interrupt();
150    
151                    messageQueue.clear();
152    
153                    if (channel != null) {
154                            channel.close();
155                            channel = null;
156                    }
157    
158            }
159    
160            @Override
161            public void exceptionCaught(final ChannelHandlerContext ctx,
162                            final ExceptionEvent e) throws Exception {
163                    log.warn("SimpleChannelHandler caught exception");
164    
165            }
166    
167            private void postMessage(final DDF_BaseMessage message) {
168                    try {
169                            messageQueue.put(message);
170                    } catch (final InterruptedException e) {
171                            log.trace("terminated");
172                    }
173            }
174    
175            private void doMarket(final DDF_BaseMessage message) {
176                    postMessage(message);
177            }
178    
179            private void doTimestamp(final DDF_BaseMessage message) {
180                    postMessage(message);
181            }
182    
183            @Override
184            public void messageReceived(final ChannelHandlerContext context,
185                            final MessageEvent eventIn) throws Exception {
186    
187                    final Object messageIn = eventIn.getMessage();
188    
189                    if (!(messageIn instanceof DDF_BaseMessage)) {
190                            context.sendUpstream(eventIn);
191                            return;
192                    }
193    
194                    final DDF_BaseMessage message = (DDF_BaseMessage) messageIn;
195                    final DDF_MessageType type = message.getMessageType();
196    
197                    if (type.isMarketMessage) {
198                            doMarket(message);
199                            return;
200                    }
201    
202                    if (type.isControlTimestamp) {
203                            doTimestamp(message);
204                            return;
205                    }
206    
207                    log.debug("unknown message : {}", message);
208    
209            }
210

212            public void bindMessageListener(final DDF_MessageListener msgListener) {
213                    this.msgListener = msgListener;
214            }

219            }

223                    // TODO Auto-generated method stub
224                    
225            }

227            @Override
228            public Future<Boolean> subscribe(Set<Sub> subs) {
229                    
230                    if (subs == null) {
231                            log.error("Null subscribes request recieved");
232                            return new FailedFuture();
233                    }
234                    
235                    for (final Sub sub : subs) {
236    
237                            if (sub != null) {
238                                    
239                                    final String inst = sub.encode();
240                                    
241                                    /* If we're subscribed already, add new interests, otherwise add */
242                                    if(subscriptions.containsKey(inst)) {
243                                            subscriptions.get(inst).addTypes(sub.types());
244                                    } else {
245                                            subscriptions.put(inst, sub);
246                                    }
247                                    
248                            }
249                    }
250                    
251                    return new DummyFuture();
252            }
253    
254            @Override
255            public Future<Boolean> subscribe(Sub sub) {
256                    
257                    if (sub == null) {
258                            log.error("Null subscribe request recieved");
259                            return new FailedFuture();
260                    }
261                    
262                    final String inst = sub.encode();
263                    if(subscriptions.containsKey(inst)) {
264                            subscriptions.get(inst).addTypes(sub.types());
265                    } else {
266                            subscriptions.put(inst, sub);
267                    }
268                    
269                    return new DummyFuture();
270            }
271            
272            // Unsubscribe is somewhat ambiguous, there is shared behavior between
273            // the registration and unregistration of instruments in the 
274            // market maker and the feed.  
275            @Override
276            public Future<Boolean> unsubscribe(Set<Sub> subs) {
277                    
278                    if (subs == null) {
279                            log.error("Null subscribes request recieved");
280                            return new FailedFuture();
281                    }
282                    
283                    for (final Sub sub : subs) {
284    
285                            if (sub != null) {
286                                    subscriptions.remove(sub.encode());
287                            }
288                    }
289                    
290                    return new DummyFuture();
291            }
292            
293            @Override
294            public Future<Boolean> unsubscribe(Sub sub) {
295                    
296                    if (sub == null) {
297                            log.error("Null subscribe request recieved");
298                            return new FailedFuture();
299                    }
300                    
301                    subscriptions.remove(sub.encode());
302                    
303                    return new DummyFuture();
304            }
