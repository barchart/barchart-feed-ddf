051    /**
052     * A stateless, connectionless UDP listener with startup and shutdown methods.
053     */
054    public class UDPListenerClientDDF extends SimpleChannelHandler implements
055                    DDF_FeedClientBase {
056    
057            /** use slf4j for internal NETTY LoggingHandler facade */
058            static {
059                    final InternalLoggerFactory defaultFactory = new Slf4JLoggerFactory();
060                    InternalLoggerFactory.setDefaultFactory(defaultFactory);
061            }
062    
063            private static final Logger log = LoggerFactory
064                            .getLogger(UDPListenerClientDDF.class);
065    
066            private final ConnectionlessBootstrap boot;
067    
068            private DatagramChannel channel;
069    
070            private volatile DDF_MessageListener msgListener = null;
071    
072            private final Executor runner;
073    
074            private final int socketAddress;
075    
076            private final Map<String, Subscription> subscriptions = 
077                            new ConcurrentHashMap<String, Subscription>();
078            
079            UDPListenerClientDDF(final int socketAddress, final Executor executor) {
080    
081                    this.socketAddress = socketAddress;
082                    runner = executor;
083    
084                    final DatagramChannelFactory channelFactory = new NioDatagramChannelFactory(
085                                    runner);
086    
087                    boot = new ConnectionlessBootstrap(channelFactory);
088    
089                    final ChannelPipelineFactory pipelineFactory = new PipelineFactoryDDF(
090                                    this);
091    
092                    boot.setPipelineFactory(pipelineFactory);
093    
094                    boot.setOption("broadcast", "false");
095                    boot.setOption("receiveBufferSizePredictorFactory", //
096                                    new FixedReceiveBufferSizePredictorFactory(2 * 1024));
097    
098            }
099    
100            private final BlockingQueue<DDF_BaseMessage> messageQueue = new LinkedBlockingQueue<DDF_BaseMessage>();
101    
102            private final RunnerDDF messageTask = new RunnerDDF() {
103    
104                    @Override
105                    protected void runCore() {
106                            while (true) {
107                                    try {
108                                            final DDF_BaseMessage message = messageQueue.take();
109    
110                                            if (msgListener != null && filter(message)) {
111                                                    msgListener.handleMessage(message);
112                                            }
113    
114                                    } catch (final InterruptedException e) {
115                                            log.trace("terminated");
116                                            return;
117                                    } catch (final Throwable e) {
118                                            log.error("message delivery failed", e);
119                                    }
120                            }
121                    }
122    
123            };
124            
125            private boolean filter(final DDF_BaseMessage message) {
126                    
127                    /* Filter by market message */
128                    if(!message.getMessageType().isMarketMessage) {
129                            return false;
130                    }
131                    
132                    final DDF_MarketBase marketMsg = (DDF_MarketBase) message;
133                    
134                    /* Filter by instrument */
135                    if(subscriptions.containsKey(marketMsg.getInstrument().get(
136                                    DDF_InstrumentField.DDF_SYMBOL_REALTIME).toString())) {
137                            
138                            return true;
139                    }
140                    
141                    return false;
142            }
143    
144            @Override
145            public void startup() {
146    
147                    runner.execute(messageTask);
148    
149                    boot.bind(new InetSocketAddress(socketAddress));
150    
151            }
152    
153            @Override
154            public void shutdown() {
155    
156                    //In FeedClientDDF, subscriptions are cleared on logout, do we want to do this here?
157                    
158                    messageTask.interrupt();
159    
160                    messageQueue.clear();
161    
162                    if (channel != null) {
163                            channel.close();
164                            channel = null;
165                    }
166    
167            }
168    
169            @Override
170            public void exceptionCaught(final ChannelHandlerContext ctx,
171                            final ExceptionEvent e) throws Exception {
172                    log.warn("SimpleChannelHandler caught exception");
173    
174            }
175    
176            private void postMessage(final DDF_BaseMessage message) {
177                    try {
178                            messageQueue.put(message);
179                    } catch (final InterruptedException e) {
180                            log.trace("terminated");
181                    }
182            }
183    
184            private void doMarket(final DDF_BaseMessage message) {
185                    postMessage(message);
186            }
187    
188            private void doTimestamp(final DDF_BaseMessage message) {
189                    postMessage(message);
190            }
191    
192            @Override
193            public void messageReceived(final ChannelHandlerContext context,
194                            final MessageEvent eventIn) throws Exception {
195    
196                    final Object messageIn = eventIn.getMessage();
197    
198                    if (!(messageIn instanceof DDF_BaseMessage)) {
199                            context.sendUpstream(eventIn);
200                            return;
201                    }
202    
203                    final DDF_BaseMessage message = (DDF_BaseMessage) messageIn;
204                    final DDF_MessageType type = message.getMessageType();
205    
206                    if (type.isMarketMessage) {
207                            doMarket(message);
208                            return;
209                    }
210    
211                    if (type.isControlTimestamp) {
212                            doTimestamp(message);
213                            return;
214                    }
215    
216                    log.debug("unknown message : {}", message);
217    
218            }
219    
220            @Override
221            public void bindMessageListener(final DDF_MessageListener msgListener) {
222                    this.msgListener = msgListener;
223            }
224    
225            @Override
226            public void bindStateListener(final FeedStateListener stateListener) {
227                    // TODO Implement connection notifications for TCP listeners
228            }
229    
230            @Override
231            public void startUpProxy() {
232                    // TODO Auto-generated method stub
233                    
234            }
235            
236            @Override
237            public Future<Boolean> subscribe(Set<Subscription> subs) {
238                    
239                    if (subs == null) {
240                            log.error("Null subscribes request recieved");
241                            return new FailedFuture();
242                    }
243                    
244                    for (final Subscription sub : subs) {
245    
246                            if (sub != null) {
247                                    
248                                    final String inst = sub.getInstrument();
249                                    
250                                    /* If we're subscribed already, add new interests, otherwise add */
251                                    if(subscriptions.containsKey(inst)) {
252                                            subscriptions.get(inst).addInterests(sub.getInterests());
253                                    } else {
254                                            subscriptions.put(inst, sub);
255                                    }
256                                    
257                            }
258                    }
259                    
260                    return new DummyFuture();
261            }
262    
263            @Override
264            public Future<Boolean> subscribe(Subscription sub) {
265                    
266                    if (sub == null) {
267                            log.error("Null subscribe request recieved");
268                            return new FailedFuture();
269                    }
270                    
271                    final String inst = sub.getInstrument();
272                    if(subscriptions.containsKey(inst)) {
273                            subscriptions.get(inst).addInterests(sub.getInterests());
274                    } else {
275                            subscriptions.put(inst, sub);
276                    }
277                    
278                    return new DummyFuture();
279            }
280            
281            // Unsubscribe is somewhat ambiguous, there is shared behavior between
282            // the registration and unregistration of instruments in the 
283            // market maker and the feed.  
284            @Override
285            public Future<Boolean> unsubscribe(Set<Subscription> subs) {
286                    
287                    if (subs == null) {
288                            log.error("Null subscribes request recieved");
289                            return new FailedFuture();
290                    }
291                    
292                    for (final Subscription sub : subs) {
293    
294                            if (sub != null) {
295                                    subscriptions.remove(sub.getInstrument());
296                            }
297                    }
298                    
299                    return new DummyFuture();
300            }
301            
302            @Override
303            public Future<Boolean> unsubscribe(Subscription sub) {
304                    
305                    if (sub == null) {
306                            log.error("Null subscribe request recieved");
307                            return new FailedFuture();
308                    }
309                    
310                    subscriptions.remove(sub.getInstrument());
311                    
312                    return new DummyFuture();
313            }
314

316            public void setPolicy(DDF_FeedEvent event, EventPolicy policy) {
317                    // Does nothing for now, some functionality will be added
318                    // for tcp listeners
319            }
