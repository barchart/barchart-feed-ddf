011    package com.barchart.feed.ddf.datalink.api;
012    
013    import java.util.Set;
014    import java.util.concurrent.Future;
015    
016    import com.barchart.feed.api.connection.Connection;
017    import com.barchart.feed.base.sub.Subscription;
018    import com.barchart.feed.base.sub.SubscriptionHandler;
019    import com.barchart.feed.ddf.datalink.enums.DDF_FeedEvent;
020    import com.barchart.util.anno.UsedOnce;
021    
022    /**
023     * Base interface for DDF feed clients. Classes directly implementing this
024     * interface should be connectionless listeners. Feeds requiring 2 way
025     * communication and a managed connection should implement DDF_FeedClient.
026     */
027    public interface DDF_FeedClientBase extends SubscriptionHandler {
028    
029            /**
030             * Binds the feed client to a port or other data source and begins
031             * listening.
032             */
033            void startup();
034            
035            void startUpProxy();
036    
037            /**
038             * Stops listening to the data source.
039             */
040            void shutdown();
041    
042            /**
043             * Attaches a message listener to the client for data consumption.
044             */
045            @UsedOnce
046            void bindMessageListener(DDF_MessageListener msgListener);
047    
048            /**
049             * Attach single feed state listener to the client.
050             */
051            @UsedOnce
052            void bindStateListener(Connection.Monitor stateListener);
053            
054            /**
055             * Handles multiple subscription requests.
056             * <p>
057             * If the client already has a subscription for any instrument of the set
058             * then this will overwrite it.
059             * <p>
060             * If called while the client is offline, registers the subscriptions and
061             * returns a future which immediately succeeds.
062             * 
063             * @param subscription
064             *            The set of subscriptions to subscribe.
065             * @return A Future which returns true if successful.
066             */
067            @Override
068            Future<Boolean> subscribe(Set<Subscription> subscriptions);
069    
070            /**
071             * Handles a subscription request.
072             * <p>
073             * If the client already has a subscription for the instrument then this
074             * will overwrite it.
075             * <p>
076             * If called while the client is offline, registers the subscription and
077             * returns a future which immediately succeeds.
078             * 
079             * @param subscription
080             *            The subscription to subscribe.
081             * @return A Future which returns true if successful.
082             */
083            @Override
084            Future<Boolean> subscribe(Subscription subscription);
085    
086            /**
087             * Handles multiple unsubscription requests.
088             * <p>
089             * If called while the client is offline, unregisters the subscriptions and
090             * returns a future which immediately succeeds.
091             * 
092             * @param subscription
093             *            The set of subscriptions to unsubscribe.
094             * @return A Future which returns true if successful.
095             */
096            @Override
097            Future<Boolean> unsubscribe(Set<Subscription> subscriptions);
098    
099            /**
100             * Handles an unsubscription request.
101             * <p>
102             * If called while the client is offline, unregisters the subscription and
103             * returns a future which immediately succeeds.
104             * 
105             * @param subscription
106             *            The subscription to unsubscribe.
107             * @return A Future which returns true if successful.
108             */
109            @Override
110            Future<Boolean> unsubscribe(Subscription subscription);
111    
112            /**
113             * Sets the event policy for a specific feed event. Default policies are set
114             * by the constructor to handle disconnects and login success.
115             * 
116             * @param event
117             *            The feed event on which to enact the policy.
118             * @param policy
119             *            The even policy to register.
120             */
121            void setPolicy(DDF_FeedEvent event, EventPolicy policy);
122    
123    }
