046            private static final Logger log = LoggerFactory.getLogger(VarBookDDF.class);

048            protected volatile MarketBookEntry lastEntry = MarketConst.NULL_BOOK_ENTRY;
049            
050            private Set<Component> changeSet = EnumSet.noneOf(Component.class);
051    
052            @SuppressWarnings("serial")
053            private static class EntryMap extends TreeMap<PriceValue, MarketBookEntry> {
054                    public EntryMap(final Comparator<PriceValue> comp) {
055                            super(comp);
056                    }
057            }
058    
059            private long millisUTC;
060            
061            private static final Comparator<PriceValue> CMP_ASC = new Comparator<PriceValue>() {
062                    @Override
063                    public int compare(final PriceValue o1, final PriceValue o2) {
064                            return o1.compareTo(o2);
065                    }
066            };
067    
068            private static final Comparator<PriceValue> CMP_DES = new Comparator<PriceValue>() {
069                    @Override
070                    public int compare(final PriceValue o1, final PriceValue o2) {
071                            return o2.compareTo(o1);
072                    }
073            };
074    
075            private final EntryMap bids = new EntryMap(CMP_ASC);
076            private final EntryMap asks = new EntryMap(CMP_DES);
077    
078            private MarketBookEntry topBid;
079            private MarketBookEntry topAsk;
080    
081            private final Instrument instrument;
082

084                            final SizeValue size, final PriceValue step) {
085                    // XXX
086                    this.instrument = instrument;
087            }

089            // #####################################
090    
091            @Override
092            public final UniBookResult setEntry(final MarketDoBookEntry entry) {
093    
094                    if(entry != null) {
095                            lastEntry = entry.freeze();
096                    }
097                    
098                    changeSet.clear();
099                    
100                    final Book.Side side = entry.side();
101    
102                    final int place = entry.place();
103    
104                    // NOTE: This only updates top.  
105    
106                    //final EntryMap map = map(side);
107    
108                    switch (side) {
109                    case BID:
110                            if (place == ENTRY_TOP) {
111                                    topBid = entry;
112                                    changeSet.add(Component.TOP_BID);
113                            }
114                            break;
115                    case ASK:
116                            if (place == ENTRY_TOP) {
117                                    topAsk = entry;
118                                    changeSet.add(Component.TOP_ASK);
119                            }
120                            break;
121                    default:
122                            return UniBookResult.ERROR;
123                    }
124    
125                    if (place == ENTRY_TOP) {
126                            return UniBookResult.TOP;
127                    } else {
128                            return UniBookResult.NORMAL;
129                    }
130    
131            }
132    
133            private EntryMap map(final Book.Side side) {
134                    switch (side) {
135                    default:
136                    case BID:
137                            return bids;
138                    case ASK:
139                            return asks;
140                    }
141            }
142    
143            @Override
144            public final MarketBookEntry[] entries(final Book.Side side) {
145    
146                    final EntryMap map = map(side);
147    
148                    final Collection<MarketBookEntry> values = map.values();
149    
150                    final int size = values.size();
151                    
152                    if(size == 0) {
153                            if(side == Side.BID) {
154                                    return new MarketBookEntry[]{topBid};
155                            } else {
156                                    return new MarketBookEntry[]{topAsk};
157                            }
158                    }
159    
160                    final MarketBookEntry[] array = new MarketBookEntry[size];
161    
162                    int index = 0;
163    
164                    for (final MarketBookEntry entry : values) {
165                            array[index] = entry;
166                            index++;
167                    }
168    
169                    return array;
170    
171            }
172    
173            @Override
174            public final DefBook freeze() {
175                    return new DefBook(instrument, time(), 
176                                    entries(Book.Side.BID),
177                                    entries(Book.Side.ASK),
178                                    lastEntry, EnumSet.copyOf(changeSet));
179            }
180            
181            @Override
182            public TimeValue time() {
183                    return ValueBuilder.newTime(millisUTC);
184            }
185    
186            @Override
187            public final void setTime(final TimeValue time) {
188                    millisUTC = time.asMillisUTC();
189            }
190    
191            @Override
192            public final boolean isFrozen() {
193                    return false;
194            }

196            @Override
197            public final MarketBookEntry last() {
198    
199    //              log.debug("last called in VarBookDDF");
200    //              final DefBookEntry entry = (DefBookEntry) lastEntry.freeze();
201    //
202    //              return entry == null ? NULL_BOOK_ENTRY : entry;
203                    throw new UnsupportedOperationException("UNUSED");
204            }
205    
206            @Override
207            public final MarketBookEntry top(final Book.Side side) {
208    
209                    final MarketBookEntry entry;
210                    switch (side) {
211                    case BID:
212                            entry = topBid;
213                            break;
214                    case ASK:
215                            entry = topAsk;
216                            break;
217                    default:
218                            entry = null;
219                            break;
220                    }
221    
222                    // System.err.println("### TOP : " + side + " ### " + entry);
223    
224                    return entry == null ? NULL_BOOK_ENTRY : entry;
225    
226            }

275            @Override
276            public UniBookResult setSnapshot(final MarketDoBookEntry[] entries) {
277    
278                    clear();
279                    
280                    changeSet.add(Component.NORMAL_BID);
281                    changeSet.add(Component.NORMAL_ASK);
282                    changeSet.add(Component.TOP_BID);
283                    changeSet.add(Component.TOP_ASK);
284                    changeSet.add(Component.ANY_BID);
285                    changeSet.add(Component.ANY_ASK);
286    
287                    for (final MarketDoBookEntry entry : entries) {
288    
289                            final EntryMap map = map(entry.side());
290    
291                            map.put(entry.priceValue(), entry);
292    
293                    }
294    
295                    /*
296                     * 
297                     */
298                    
299                    return UniBookResult.NORMAL;
300    
301            }
