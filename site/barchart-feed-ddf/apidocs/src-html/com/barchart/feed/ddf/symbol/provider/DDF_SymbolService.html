053            /**
054             * Find.
055             *
056             * @param text the text
057             * @return the dD f_ symbol
058             */
059            public static final DDF_Symbol find(final TextValue text) {
060                    final TextValue guid = DDF_Symbology.lookupFromSymbol(text);
061                    DDF_Symbol symbol = symbolMap.get(guid);
062    
063                    //log.error("find {}", text);
064    
065                    if (symbol == null) {
066                            symbol = decode(guid.toString());
067                            // TODO handle null symbol
068                            symbolMap.putIfAbsent(guid, symbol);
069                    }
070                    return symbol;
071            }
072    
073            /**
074             * Clear cache.
075             */
076            public static final void clearCache() {
077                    symbolMap.clear();
078            }

082            /**
083             * Decode.
084             *
085             * @param symbolName the symbol name
086             * @return the dD f_ symbol
087             */
088            public static final DDF_Symbol decode(final String symbolName) {
089    
090                    if (symbolName == null) {
091                            log.error("symbolName == null");
092                            return null;
093                    }
094    
095                    if (isOptionExtended(symbolName)) {
096                            final OptionExtended symbol = new OptionExtended();
097                            symbol.name = symbolName;
098                            final int finish = symbolName.length();
099                            final int middle = symbolName.indexOf(OPTION_SEPARATOR);
100                            symbol.group = symbolName.substring(0, middle - 2);
101                            symbol.month = DDF_ExpireMonth.fromCode(symbolName
102                                            .charAt(middle - 2));
103                            symbol.year = DDF_ExpireYear
104                                            .fromCode(symbolName.charAt(middle - 1));
105                            symbol.strikePrice = symbolName.substring(middle + 1, finish - 1);
106                            symbol.optionIndicator = DDF_Option.fromCode(symbolName
107                                            .charAt(finish - 1));
108                            return symbol;
109                    }
110    
111                    if (isOptionBasic(symbolName)) {
112                            final OptionBasic symbol = new OptionBasic();
113                            symbol.name = symbolName;
114                            final int finish = symbolName.length();
115                            int middle = 0;
116                            for (int k = finish - 2; k > 0; k--) {
117                                    if (ASCII.isDigit(symbolName.charAt(k))) {
118                                            continue;
119                                    } else {
120                                            middle = k;
121                                            break;
122                                    }
123                            }
124                            symbol.group = symbolName.substring(0, middle);
125                            symbol.month = DDF_ExpireMonth.fromCode(symbolName.charAt(middle));
126                            final DDF_OptionYear indicator = DDF_OptionYear.fromCode(symbolName
127                                            .charAt(finish - 1));
128                            symbol.year = DDF_ExpireYear.fromOptionYear(indicator);
129                            symbol.strikePrice = symbolName.substring(middle + 1, finish - 1);
130                            symbol.optionIndicator = DDF_Option.fromOptionYear(indicator);
131                            
132                            return symbol;
133                    }
134    
135                    if (isIndex(symbolName)) {
136                            final BaseIndex symbol = new BaseIndex();
137                            symbol.name = symbolName;
138                            symbol.group = symbolName.substring(1);
139                            return symbol;
140                    }
141    
142                    if (isEquityCAN(symbolName)) {
143                            final BaseEquity symbol = new BaseEquity();
144                            symbol.name = symbolName;
145                            symbol.group = getGroupEquityCAN(symbolName);
146                            symbol.equityType = DDF_Equity.CAN;
147                            return symbol;
148                    }
149    
150                    if (isEquityLSE(symbolName)) {
151                            final BaseEquity symbol = new BaseEquity();
152                            symbol.name = symbolName;
153                            symbol.group = getGroupEquityLSE(symbolName);
154                            symbol.equityType = DDF_Equity.LSE;
155                            return symbol;
156                    }
157    
158                    if (isEquityNSE(symbolName)) {
159                            final BaseEquity symbol = new BaseEquity();
160                            symbol.name = symbolName;
161                            symbol.group = getGroupEquityNSE(symbolName);
162                            symbol.equityType = DDF_Equity.NSE;
163                            return symbol;
164                    }
165    
166                    if (isFuture(symbolName)) {
167                            final BaseFuture symbol = new BaseFuture();
168                            symbol.name = symbolName;
169                            symbol.group = getGroupFuture(symbolName);
170                            symbol.year = getFutureExpireYear(symbolName);
171                            symbol.month = getFutureExpireMonth(symbolName);
172                            return symbol;
173                    }
174    
175                    if (isEquityOther(symbolName)) {
176                            final BaseEquity symbol = new BaseEquity();
177                            symbol.name = symbolName;
178                            symbol.group = symbolName;
179                            return symbol;
180                    }
181    
182                    // TODO finish remaining types
183    
184                    log.error("can not parse; symbolName={}", symbolName);
185    
186                    return NULL_SYMBOL;
187    
188            }
