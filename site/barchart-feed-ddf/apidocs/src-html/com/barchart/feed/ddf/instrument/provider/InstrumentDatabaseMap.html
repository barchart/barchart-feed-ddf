001    package com.barchart.feed.ddf.instrument.provider;
002    
003    import java.io.File;
004    import java.io.IOException;
005    import java.io.InputStream;
006    import java.util.Set;
007    import java.util.zip.ZipEntry;
008    import java.util.zip.ZipException;
009    import java.util.zip.ZipFile;
010    
011    import org.openfeed.proto.inst.InstrumentDefinition;
012    import org.slf4j.Logger;
013    import org.slf4j.LoggerFactory;
014    
015    import com.barchart.feed.inst.provider.InstrumentMap;
016    import com.google.protobuf.InvalidProtocolBufferException;
017    import com.sleepycat.bind.EntityBinding;
018    import com.sleepycat.bind.EntryBinding;
019    import com.sleepycat.collections.StoredMap;
020    import com.sleepycat.je.Database;
021    import com.sleepycat.je.DatabaseConfig;
022    import com.sleepycat.je.DatabaseEntry;
023    import com.sleepycat.je.Durability;
024    import com.sleepycat.je.Environment;
025    import com.sleepycat.je.EnvironmentConfig;
026    import com.sleepycat.je.EnvironmentLockedException;
027    
028    public final class InstrumentDatabaseMap implements InstrumentMap {
029            
030            private static final Logger log = LoggerFactory
031                            .getLogger(InstrumentDatabaseMap.class);
032    
033            private final StoredMap<String, InstrumentDefinition> map;
034            
035            private final Durability durability = new Durability(
036                            Durability.SyncPolicy.WRITE_NO_SYNC,
037                            Durability.SyncPolicy.WRITE_NO_SYNC, 
038                            Durability.ReplicaAckPolicy.NONE);
039            
040            private final EnvironmentConfig envConfig = 
041                            (EnvironmentConfig) new EnvironmentConfig()
042                    .setAllowCreate(true)
043                    .setTransactional(true)
044                    .setDurability(durability);
045            
046            private final DatabaseConfig dbConfig = new DatabaseConfig()
047                    .setAllowCreate(true)
048                    .setTransactional(true)
049                    .setSortedDuplicates(false);
050            
051            private final Database dbase;
052            
053            /**
054             * 
055             * @param dbase
056             */
057            public InstrumentDatabaseMap(final Database dbase) {
058                    
059                    this.dbase = dbase;
060                    
061                    map = new StoredMap<String, InstrumentDefinition>(dbase, 
062                                    new SymbolBinding(), new InstDefBinding(), true);
063            }
064            
065            /**
066             * 
067             * @param dbFolder
068             * @throws EnvironmentLockedException if provided File is locked by another application
069             */
070            public InstrumentDatabaseMap(final File dbFolder) throws EnvironmentLockedException {
071                    
072                    final Environment env = new Environment(dbFolder, envConfig);
073                    dbase = env.openDatabase(null, "InstrumentDef", dbConfig);
074                    
075                    map = new StoredMap<String, InstrumentDefinition>(dbase, new SymbolBinding(), 
076                                    new InstDefBinding(), true);
077                    
078            }
079            
080            /**
081             * 
082             * @param dbFolder
083             * @param inStream
084             */
085            public InstrumentDatabaseMap(final File dbFolder, final InputStream inStream) {
086                    
087                    final Environment env = new Environment(dbFolder, envConfig);
088                    dbase = env.openDatabase(null, "InstrumentDef", dbConfig);
089                    map =  new StoredMap<String, InstrumentDefinition>(dbase, new SymbolBinding(), 
090                                    new InstDefBinding(), true);
091                    
092                    populateDB(inStream);
093                    
094            }
095            
096            /**
097             * 
098             * @param dbFolder
099             * @param instDefZip
100             * @throws ZipException
101             * @throws IOException
102             */
103            public InstrumentDatabaseMap(final File dbFolder, final File instDefZip) 
104                            throws ZipException, IOException {
105                    
106                    final Environment env = new Environment(dbFolder, envConfig);
107                    dbase = env.openDatabase(null, "InstrumentDef", dbConfig);
108                    map = new StoredMap<String, InstrumentDefinition>(dbase, new SymbolBinding(), 
109                                    new InstDefBinding(), true);
110                    
111                    InputStream inStream = null;
112                    
113                    try {
114                    
115                            final ZipFile zFile = new ZipFile(instDefZip);
116                            final ZipEntry entry = zFile.entries().nextElement();
117                            inStream = zFile.getInputStream(entry);
118                            
119                            populateDB(inStream);
120                    
121                    } finally {
122                            if(inStream != null) {
123                                    inStream.close();
124                            }
125                    }
126                    
127            }
128            
129            private void populateDB(final InputStream inStream) {
130                    
131                    long counter = 0;
132                    while(true) {
133                            InstrumentDefinition def;
134                            
135                            try {
136                                    def = InstrumentDefinition.
137                                                    parseDelimitedFrom(inStream);
138                            } catch (Exception e) {
139                                    e.printStackTrace();
140                                    break;
141                            }
142                            
143                            if(def!= null) {
144                                    
145                                    if(def.hasSymbol()) {
146                                            map.put(def.getSymbol(), def);
147                                    }
148                                    
149                                    if(counter % 10000 == 0) {
150                                            System.out.println("Build count " + counter);
151                                    }
152                                    
153                                    counter++;
154                                    
155                            } else {
156                                    break;
157                            }
158                    }
159                    
160            }
161            
162            /**
163             * 
164             * @param key
165             * @return
166             */
167            @Override
168            public boolean containsKey(final String key) {
169                    return map.containsKey(key);
170            }
171            
172            /**
173             * 
174             * @param key
175             * @return
176             */
177            @Override
178            public InstrumentDefinition get(final String key) {
179                    return map.get(key);
180            }
181            
182            /**
183             * 
184             * @param key
185             * @param value
186             */
187            @Override
188            public void put(final String key, final InstrumentDefinition value) {
189                    map.put(key, value);
190            }
191            
192            /**
193             * 
194             * @return
195             */
196            @Override
197            public int size() {
198                    return map.size();
199            }
200            
201            /**
202             * 
203             */
204            @Override
205            public void clear() {
206                    map.clear();
207            }
208            
209            /**
210             * 
211             */
212            @Override
213            public void close() {
214                    log.info("DB Closed");
215                    dbase.close();
216            }
217            
218            @Override
219            public Set<String> keySet() {
220                    return map.keySet();
221            }
222    
223            
224            private class SymbolBinding implements EntryBinding<String> {
225    
226                    @Override
227                    public String entryToObject(DatabaseEntry entry) {
228                            return new String(entry.getData());
229                    }
230    
231                    @Override
232                    public void objectToEntry(String object, DatabaseEntry entry) {
233                            entry.setData(object.getBytes());
234                    }
235                    
236            }
237            
238            private class InstDefBinding implements EntityBinding<InstrumentDefinition> {
239    
240                    @Override
241                    public InstrumentDefinition entryToObject(DatabaseEntry key,
242                                    DatabaseEntry data) {
243                            try {
244                                    return InstrumentDefinition.parseFrom(data.getData());
245                            } catch (InvalidProtocolBufferException e) {
246                                    return InstrumentDefinition.getDefaultInstance();
247                            }
248                    }
249    
250                    @Override
251                    public void objectToKey(InstrumentDefinition object, DatabaseEntry key) {
252                            key.setData(object.getSymbol().getBytes());
253                    }
254    
255                    @Override
256                    public void objectToData(InstrumentDefinition object, DatabaseEntry data) {
257                            data.setData(object.toByteArray());
258                    }
259                    
260            }
261    
262    }
