060            public static InstrumentDefinition decodeXML(final Element tag) throws Exception {
061                    
062                    InstrumentDefinition.Builder builder = InstrumentDefinition.newBuilder();
063                    
064                    // lookup status
065                    final String statusCode = xmlStringDecode(tag, STATUS, XML_STOP);
066                    final StatusXML status = StatusXML.fromCode(statusCode);
067                    if (!status.isFound()) {
068                            final String lookup = xmlStringDecode(tag, LOOKUP, XML_STOP);
069                            throw new SymbolNotFoundException(lookup);
070                    }
071                    
072                    /* market identifier; must be globally unique; */
073                    try {
074                            builder.setMarketId(Long.parseLong(xmlStringDecode(tag, ID, XML_STOP)));
075                    } catch(Exception e) {
076                            return InstrumentDefinition.getDefaultInstance();
077                    }
078                    /* type of security, Forex, Equity, etc. */
079                    builder.setInstrumentType(InstrumentType.NO_INSTRUMENT);
080                    
081                    /* liquidy type, default / implied / combined */
082                    builder.setBookLiquidity(BookLiquidity.NO_BOOK_LIQUIDITY);
083                    
084                    /* structure of book */
085                    builder.setBookStructure(BookStructure.NO_BOOK_STRUCTURE);
086                    
087                    /* vendor */
088                    builder.setVendorId("Barchart");
089                    
090                    /* market symbol; can be non unique; */
091                    builder.setSymbol(xmlStringDecode(tag, SYMBOL_REALTIME, XML_STOP));
092                    
093                    /* market free style description; can be used in full text search */
094                    builder.setDescription(String.valueOf(xmlStringDecode(tag, SYMBOL_COMMENT,
095                                    XML_PASS)));
096                    
097                    /* stock vs future vs etc. */
098                    builder.setCfiCode(xmlStringDecode(tag, SYMBOL_CODE_CFI, XML_PASS));
099                    
100                    /* price currency */
101                    builder.setCurrencyCode("USD");
102                    
103                    /* market originating exchange identifier */
104                    final DDF_Exchange exchange = DDF_Exchange.fromCode( 
105                                    xmlByteDecode(tag, EXCHANGE_DDF, XML_PASS));
106                    String eCode = new String(new byte[] {exchange.code});
107                    if(eCode == null || eCode.isEmpty()) {
108                            eCode = Exchanges.NULL_CODE;
109                    }
110                    builder.setExchangeCode(eCode);
111                    
112                    final DDF_Fraction frac = DDF_Fraction.fromBaseCode(
113                                    xmlByteDecode(tag, BASE_CODE_DDF, XML_STOP));
114                    
115                    /* price step / increment size / tick size */
116                    try {
117                            final long priceStepMantissa = xmlDecimalDecode(frac, tag,
118                                    PRICE_TICK_INCREMENT, XML_STOP);
119                            builder.setMinimumPriceIncrement(buildDecimal(priceStepMantissa, frac.decimalExponent));
120                    } catch (Exception e) {
121                            builder.setMinimumPriceIncrement(buildDecimal(0,0));
122                    }
123                    
124                    /* value of a future contract / stock share */
125                    final String pricePointString = xmlStringDecode(tag, PRICE_POINT_VALUE, XML_PASS);
126                    if(pricePointString == null || pricePointString.isEmpty()) {
127                            builder.setContractPointValue(buildDecimal(0,0));
128                    } else {
129                            PriceValue pricePoint = ValueBuilder.newPrice(Double
130                                            .valueOf(pricePointString));
131                            builder.setContractPointValue(buildDecimal(
132                                            pricePoint.mantissa(), pricePoint.exponent()));
133                    }
134                    
135                    /* display fraction base : decimal(10) vs binary(2), etc. */
136                    builder.setDisplayBase((int)frac.fraction.base());
137                    builder.setDisplayExponent(frac.fraction.exponent());
138                    
139                    /* Calendar */
140                    final Time expire = xmlTimeDecode(tag, SYMBOL_EXPIRE, XML_PASS);
141                    final Calendar.Builder calBuilder = Calendar.newBuilder();
142                    final Interval.Builder intBuilder = Interval.newBuilder();
143                    
144                    intBuilder.setTimeStart(0);
145                    if(expire == null) {
146                            intBuilder.setTimeFinish(0);
147                    } else {
148                            intBuilder.setTimeFinish(expire.millisecond());
149                    }
150                    
151                    calBuilder.setLifeTime(intBuilder.build());
152                    builder.setCalendar(calBuilder.build());
153    
154                    //
155                    final DDF_TimeZone zone = DDF_TimeZone.fromCode(xmlStringDecode(
156                                    tag, TIME_ZONE_DDF, XML_STOP));
157                    
158                    /* time zone name as text */
159                    builder.setTimeZoneName(zone.name());
160                    
161                    return builder.build();
162                    
163            }

165            public static InstrumentDefinition decodeSAX(final Attributes ats) throws Exception {
166    
167                    InstrumentDefinition.Builder builder = InstrumentDefinition.newBuilder();
168                    
169                    try {
170                            
171                            // lookup status
172                            final String statusCode = xmlStringDecode(ats, STATUS, XML_STOP);
173                            final StatusXML status = StatusXML.fromCode(statusCode);
174                            if (!status.isFound()) {
175                                    final String lookup = xmlStringDecode(ats, LOOKUP, XML_STOP);
176                                    throw new SymbolNotFoundException(lookup);
177                            }
178                            
179                            /* market identifier; must be globally unique; */
180                            builder.setMarketId(Long.parseLong(xmlStringDecode(ats, ID, XML_STOP)));
181                            
182                            /* type of security, Forex, Equity, etc. */
183                            builder.setInstrumentType(InstrumentType.NO_INSTRUMENT);
184                            
185                            /* liquidy type, default / implied / combined */
186                            builder.setBookLiquidity(BookLiquidity.NO_BOOK_LIQUIDITY);
187                            
188                            /* structure of book */
189                            builder.setBookStructure(BookStructure.NO_BOOK_STRUCTURE);
190                            
191                            /* book depth */
192                            builder.setBookDepth(0);
193                            
194                            /* vendor */
195                            builder.setVendorId("Barchart");
196                            
197                            /* market symbol; can be non unique; */
198                            builder.setSymbol(xmlStringDecode(ats, SYMBOL_REALTIME, XML_STOP));
199                            
200                            /* Barchart symbol, only used for options */
201                            final String bar = xmlStringDecode(ats, SYMBOL_DDF_REAL, XML_PASS);
202                            
203                            if(bar != null) {
204                                    Symbol.Builder b = Symbol.newBuilder();
205                                    b.setVendor(VendorID.BARCHART.toString());
206                                    b.setSymbol(bar);
207                                    final Symbol symbol = b.build();
208                                    builder.addSymbols(symbol);
209                            }
210                            
211                            // NOTE: Hard coded for CQG
212                            final String cqg = xmlStringDecode(ats, ALT_SYMBOL, XML_PASS);
213                            
214                            if(cqg != null) {
215                                    Symbol.Builder b = Symbol.newBuilder();
216                                    b.setVendor(DDF_RxInstrumentProvider.CQG_VENDOR_ID.toString());
217                                    b.setSymbol(cqg);
218                                    final Symbol symbol = b.build();
219                                    builder.addSymbols(symbol);
220                            }
221                            
222                            /* market free style description; can be used in full text search */
223                            builder.setDescription(String.valueOf(xmlStringDecode(ats, SYMBOL_COMMENT,
224                                            XML_PASS)));
225                            
226                            /* stock vs future vs etc. */
227                            builder.setCfiCode(xmlStringDecode(ats, SYMBOL_CODE_CFI, XML_PASS));
228                            
229                            /* price currency */
230                            builder.setCurrencyCode("USD");
231                            
232                            /* market originating exchange identifier */
233                            final DDF_Exchange exchange = DDF_Exchange.fromCode( 
234                                            xmlByteDecode(ats, EXCHANGE_DDF, XML_PASS));
235                            String eCode = new String(new byte[] {exchange.code});
236                            if(eCode == null || eCode.isEmpty()) {
237                                    eCode = Exchanges.NULL_CODE;
238                            }
239                            builder.setExchangeCode(eCode);
240                            
241                            final DDF_Fraction frac = DDF_Fraction.fromBaseCode(
242                                            xmlByteDecode(ats, BASE_CODE_DDF, XML_STOP));
243                            
244                            /* price step / increment size / tick size */
245                            try {
246                                    final long priceStepMantissa = xmlDecimalDecode(frac, ats,
247                                            PRICE_TICK_INCREMENT, XML_STOP);
248                                    builder.setMinimumPriceIncrement(buildDecimal(priceStepMantissa, frac.decimalExponent));
249                            } catch (Exception e) {
250                                    builder.setMinimumPriceIncrement(buildDecimal(0,0));
251                            }
252                            
253                            /* value of a future contract / stock share */
254                            final String pricePointString = xmlStringDecode(ats, PRICE_POINT_VALUE, XML_PASS);
255                            if(pricePointString == null || pricePointString.isEmpty()) {
256                                    builder.setContractPointValue(buildDecimal(0,0));
257                            } else {
258                                    PriceValue pricePoint = ValueBuilder.newPrice(Double
259                                                    .valueOf(pricePointString));
260                                    builder.setContractPointValue(buildDecimal(
261                                                    pricePoint.mantissa(), pricePoint.exponent()));
262                            }
263                            
264                            /* display fraction base : decimal(10) vs binary(2), etc. */
265                            builder.setDisplayBase((int)frac.fraction.base());
266                            builder.setDisplayExponent(frac.fraction.exponent());
267                            
268                            /* Calendar */
269                            final Time expire = xmlTimeDecode(ats, SYMBOL_EXPIRE, XML_PASS);
270                            final Calendar.Builder calBuilder = Calendar.newBuilder();
271                            final Interval.Builder intBuilder = Interval.newBuilder();
272                            
273                            intBuilder.setTimeStart(0);
274                            if(expire == null) {
275                                    intBuilder.setTimeFinish(0);
276                            } else {
277                                    intBuilder.setTimeFinish(expire.millisecond());
278                            }
279                            
280                            calBuilder.setLifeTime(intBuilder.build());
281                            builder.setCalendar(calBuilder.build());
282            
283                            //
284                            final DDF_TimeZone zone = DDF_TimeZone.fromCode(xmlStringDecode(
285                                            ats, TIME_ZONE_DDF, XML_STOP));
286                            
287            //              /* timezone represented as offset in minutes from utc */
288            //              builder.setTimeZoneOffset(zone.getUTCOffset());
289                            
290                            /* time zone name as text */
291                            builder.setTimeZoneName(zone.name());
292                            
293                            return builder.build();
294                    
295                    } catch (final Exception e) {
296                            final String lookup = xmlStringDecode(ats, LOOKUP, XML_STOP);
297                            throw new SymbolNotFoundException(lookup);
298                    }
299                    
300            }

302            public static Decimal buildDecimal(final long mantissa, final int exponent) {
303                    
304                    final Decimal.Builder builder = Decimal.newBuilder();
305                    builder.setMantissa(mantissa);
306                    builder.setExponent(exponent);
307                    return builder.build();
308                    
309            }
