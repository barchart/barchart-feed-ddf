057            public static InstrumentDefinition decodeXML(final Element tag) throws Exception {
058                    
059                    InstrumentDefinition.Builder builder = InstrumentDefinition.newBuilder();
060                    
061                    // lookup status
062                    final String statusCode = xmlStringDecode(tag, STATUS, XML_STOP);
063                    final StatusXML status = StatusXML.fromCode(statusCode);
064                    if (!status.isFound()) {
065                            final String lookup = xmlStringDecode(tag, LOOKUP, XML_STOP);
066                            throw new SymbolNotFoundException(lookup);
067                    }
068                    
069                    /* market identifier; must be globally unique; */
070                    try {
071                            builder.setMarketId(Long.parseLong(xmlStringDecode(tag, ID, XML_STOP)));
072                    } catch(Exception e) {
073                            return InstrumentDefinition.getDefaultInstance();
074                    }
075                    /* type of security, Forex, Equity, etc. */
076                    builder.setInstrumentType(InstrumentType.NO_INSTRUMENT);
077                    
078                    /* liquidy type, default / implied / combined */
079                    builder.setBookLiquidity(BookLiquidity.NO_BOOK_LIQUIDITY);
080                    
081                    /* structure of book */
082                    builder.setBookStructure(BookStructure.NO_BOOK_STRUCTURE);
083                    
084                    /* vendor */
085                    builder.setVendorId("Barchart");
086                    
087                    /* market symbol; can be non unique; */
088                    builder.setSymbol(xmlStringDecode(tag, SYMBOL_REALTIME, XML_STOP));
089                    
090                    /* market free style description; can be used in full text search */
091                    builder.setDescription(String.valueOf(xmlStringDecode(tag, SYMBOL_COMMENT,
092                                    XML_PASS)));
093                    
094                    /* stock vs future vs etc. */
095                    builder.setCfiCode(xmlStringDecode(tag, SYMBOL_CODE_CFI, XML_PASS));
096                    
097                    /* price currency */
098                    builder.setCurrencyCode("USD");
099                    
100                    /* market originating exchange identifier */
101                    final DDF_Exchange exchange = DDF_Exchange.fromCode( 
102                                    xmlByteDecode(tag, EXCHANGE_DDF, XML_PASS));
103                    String eCode = new String(new byte[] {exchange.code});
104                    if(eCode == null || eCode.isEmpty()) {
105                            eCode = Exchanges.NULL_CODE;
106                    }
107                    builder.setExchangeCode(eCode);
108                    
109                    final DDF_Fraction frac = DDF_Fraction.fromBaseCode(
110                                    xmlByteDecode(tag, BASE_CODE_DDF, XML_STOP));
111                    
112                    /* price step / increment size / tick size */
113                    try {
114                            final long priceStepMantissa = xmlDecimalDecode(frac, tag,
115                                    PRICE_TICK_INCREMENT, XML_STOP);
116                            builder.setMinimumPriceIncrement(buildDecimal(priceStepMantissa, frac.decimalExponent));
117                    } catch (Exception e) {
118                            builder.setMinimumPriceIncrement(buildDecimal(0,0));
119                    }
120                    
121                    /* value of a future contract / stock share */
122                    final String pricePointString = xmlStringDecode(tag, PRICE_POINT_VALUE, XML_PASS);
123                    if(pricePointString == null || pricePointString.isEmpty()) {
124                            builder.setContractPointValue(buildDecimal(0,0));
125                    } else {
126                            PriceValue pricePoint = ValueBuilder.newPrice(Double
127                                            .valueOf(pricePointString));
128                            builder.setContractPointValue(buildDecimal(
129                                            pricePoint.mantissa(), pricePoint.exponent()));
130                    }
131                    
132                    /* display fraction base : decimal(10) vs binary(2), etc. */
133                    builder.setDisplayBase((int)frac.fraction.base());
134                    builder.setDisplayExponent(frac.fraction.exponent());
135                    
136                    /* Calendar */
137                    final Time expire = xmlTimeDecode(tag, SYMBOL_EXPIRE, XML_PASS);
138                    final Calendar.Builder calBuilder = Calendar.newBuilder();
139                    final Interval.Builder intBuilder = Interval.newBuilder();
140                    
141                    intBuilder.setTimeStart(0);
142                    if(expire == null) {
143                            intBuilder.setTimeFinish(0);
144                    } else {
145                            intBuilder.setTimeFinish(expire.millisecond());
146                    }
147                    
148                    calBuilder.setLifeTime(intBuilder.build());
149                    builder.setCalendar(calBuilder.build());
150    
151                    //
152                    final DDF_TimeZone zone = DDF_TimeZone.fromCode(xmlStringDecode(
153                                    tag, TIME_ZONE_DDF, XML_STOP));
154                    
155                    /* time zone name as text */
156                    builder.setTimeZoneName(zone.name());
157                    
158                    return builder.build();
159                    
160            }

277            public static Decimal buildDecimal(final long mantissa, final int exponent) {
278                    
279                    final Decimal.Builder builder = Decimal.newBuilder();
280                    builder.setMantissa(mantissa);
281                    builder.setExponent(exponent);
282                    return builder.build();
283
