090            public static InstrumentDDF decodeXML(final Element tag) throws Exception {
091                    
092                    // lookup status
093    
094                    final String statusCode = xmlStringDecode(tag, STATUS, XML_STOP);
095                    final StatusXML status = StatusXML.fromCode(statusCode);
096    
097                    if (!status.isFound()) {
098                            final String lookup = xmlStringDecode(tag, LOOKUP, XML_STOP);
099                            throw new SymbolNotFoundException(lookup);
100                    }
101    
102                    // decode DOM
103                    TextValue guid;
104                    try {
105                            guid = ValueBuilder.newText(xmlStringDecode(tag, GUID, XML_STOP));
106                    } catch (Exception e) {
107                            // TODO
108                            //return (InstrumentDDF) Instrument.NULL;
109                            return null;
110                    }
111                    
112                    final TextValue symbolReal = ValueBuilder.newText(xmlStringDecode(
113                                    tag, SYMBOL_REALTIME, XML_STOP));
114                    final byte exchCode = xmlByteDecode(tag, EXCHANGE_DDF, XML_PASS); 
115                    final byte baseCode = xmlByteDecode(tag, BASE_CODE_DDF, XML_STOP);
116                    final String codeCFI = xmlStringDecode(tag, SYMBOL_CODE_CFI, XML_PASS);
117                    final String zoneCode = xmlStringDecode(tag, TIME_ZONE_DDF, XML_STOP);
118                    final String symbolComment = xmlStringDecode(tag, SYMBOL_COMMENT,
119                                    XML_PASS);
120                    
121                    final Time expire = xmlTimeDecode(tag, SYMBOL_EXPIRE, XML_PASS);
122    
123                    // month code for exp of futures contract
124                    final DDF_TimeZone zone = DDF_TimeZone.fromCode(zoneCode);
125                    final DDF_Exchange exchange = DDF_Exchange.fromCode(exchCode);
126                    final DDF_Fraction frac = DDF_Fraction.fromBaseCode(baseCode);
127                    final long priceStepMantissa = xmlDecimalDecode(frac, tag,
128                                    PRICE_TICK_INCREMENT, XML_STOP);
129                    final String pricePointString = xmlStringDecode(tag, PRICE_POINT_VALUE,
130                                    XML_PASS);
131    
132                    PriceValue pricePoint = ValueBuilder.newPrice(0);
133                    if (pricePointString != null) {
134                            try {
135                                    pricePoint = ValueBuilder.newPrice(Double
136                                                    .valueOf(pricePointString));
137                            } catch (Exception e) {
138                            }
139    
140                    }
141    
142                    final PriceValue priceStep = newPrice(priceStepMantissa,
143                                    frac.decimalExponent);
144                    
145                    /* Build Lifetime, currently only have last month/year of instrument from ddf.extras */
146                    TimeInterval lifetime; 
147                    if(expire == null || expire == Time.NULL) { // Was isNull()
148                            lifetime = TimeInterval.NULL;
149                    } else {
150                            lifetime = factory.newTimeInterval(0, expire.millisecond());
151                    }
152                    
153                    Fraction f = factory.newFraction(1, (int) frac.decimalDenominator);
154                    
155                    return build(guid, symbolReal, symbolComment, codeCFI, 
156                                    exchange, priceStep, pricePoint, f, lifetime, zone);
157                    
158            }

160            public static InstrumentDDF decodeSAX(final Attributes ats) throws Exception {
161                    
162                    // lookup status
163                    final String statusCode = xmlStringDecode(ats, STATUS, XML_STOP);
164                    final StatusXML status = StatusXML.fromCode(statusCode);
165                    if (!status.isFound()) {
166                            final String lookup = xmlStringDecode(ats, LOOKUP, XML_STOP);
167                            throw new SymbolNotFoundException(lookup);
168                    }
169                    
170                    // decode SAX
171                    final TextValue guid = ValueBuilder.newText(xmlStringDecode(ats, GUID, XML_STOP));
172                    final TextValue symbolReal = ValueBuilder.newText(
173                                    xmlStringDecode(ats, SYMBOL_REALTIME, XML_STOP));
174                    final byte exchCode = xmlByteDecode(ats, EXCHANGE_DDF, XML_PASS); 
175                    final byte baseCode = xmlByteDecode(ats, BASE_CODE_DDF, XML_STOP);
176                    final String codeCFI = xmlStringDecode(ats, SYMBOL_CODE_CFI, XML_PASS);
177                    final String zoneCode = xmlStringDecode(ats, TIME_ZONE_DDF, XML_STOP);
178                    final String symbolComment = xmlStringDecode(ats, SYMBOL_COMMENT,
179                                    XML_PASS);
180                    final Time expire = xmlTimeDecode(ats, SYMBOL_EXPIRE, XML_PASS);
181    
182                    //
183    
184                    final DDF_TimeZone zone = DDF_TimeZone.fromCode(zoneCode);
185                    final DDF_Exchange exchange = DDF_Exchange.fromCode(exchCode);
186                    final DDF_Fraction frac = DDF_Fraction.fromBaseCode(baseCode);
187                    final long priceStepMantissa = xmlDecimalDecode(frac, ats,
188                                    PRICE_TICK_INCREMENT, XML_STOP);
189                    final String pricePointString = xmlStringDecode(ats, PRICE_POINT_VALUE,
190                                    XML_PASS);
191    
192                    PriceValue pricePoint = ValueBuilder.newPrice(0);
193                    if (pricePointString != null) {
194                            try {
195                                    pricePoint = ValueBuilder.newPrice(Double
196                                                    .valueOf(pricePointString));
197                            } catch (Exception e) {
198                            }
199                    }
200    
201                    final PriceValue priceStep = newPrice(priceStepMantissa,
202                                    frac.decimalExponent);
203                    
204                    /* Build Lifetime, currently only have last month/year of instrument from ddf.extras */
205                    TimeInterval lifetime; 
206                    if(expire == null || expire == com.barchart.util.value.impl.ValueConst.NULL_TIME) { // Was isNull()
207                            lifetime = com.barchart.util.value.impl.ValueConst.NULL_TIME_INTERVAL;
208                    } else {
209                            lifetime = factory.newTimeInterval(0, expire.millisecond());
210                    }
211                    
212                    Fraction f = factory.newFraction(1, (int) frac.decimalDenominator);
213                    
214                    return build(guid, symbolReal, symbolComment, codeCFI, 
215                                    exchange, priceStep, pricePoint, f, lifetime, zone);
216                    
217            }

219            @SuppressWarnings("rawtypes")
220            private static final InstrumentDDF build(final TextValue guid,
221                            final TextValue symbolReal, final String symbolComment,
222                            final String codeCFI, final DDF_Exchange exchange,
223                            final PriceValue priceStep, final PriceValue pricePoint,
224                            final Fraction fraction, final TimeInterval lifetime,
225                            final DDF_TimeZone zone) {
226                    
227                    final Map<Tag, Object> map = new HashMap<Tag, Object>();
228                    
229                    map.put(InstrumentField.GUID, new InstrumentGUID(guid));
230                    map.put(MARKET_GUID, guid);
231                    map.put(SECURITY_TYPE, Instrument.SecurityType.NULL_TYPE);
232                    map.put(BOOK_LIQUIDITY, Instrument.BookLiquidityType.NONE);
233                    map.put(BOOK_STRUCTURE, Instrument.BookStructureType.NONE);
234                    map.put(BOOK_DEPTH, ValueConst.NULL_SIZE);
235                    map.put(VENDOR, newText("Barchart"));
236                    map.put(SYMBOL, symbolReal);
237                    map.put(DESCRIPTION, newText(symbolComment));
238                    map.put(CFI_CODE, newText(codeCFI));
239                    map.put(EXCHANGE, exchange.asExchange());
240                    map.put(EXCHANGE_CODE, newText(new byte[]{(exchange.code)}));
241                    map.put(TICK_SIZE, priceStep);
242                    map.put(POINT_VALUE, pricePoint);
243                    map.put(DISPLAY_FRACTION, fraction);
244                    map.put(LIFETIME, lifetime);
245                    map.put(MARKET_HOURS, factory.newSchedule(new TimeInterval[0]));
246                    map.put(TIME_ZONE_OFFSET, newSize(zone.getUTCOffset()));
247                    map.put(TIME_ZONE_NAME, newText(zone.name()));
248    
249                    return ObjectMapFactory.build(InstrumentDDF.class, map);
250                    
251            }

253            public static InstrumentDefinition decodeSAXProto(final Attributes ats) throws Exception {
254    
255                    InstrumentDefinition.Builder builder = InstrumentDefinition.newBuilder();
256                    
257                    // lookup status
258                    final String statusCode = xmlStringDecode(ats, STATUS, XML_STOP);
259                    final StatusXML status = StatusXML.fromCode(statusCode);
260                    if (!status.isFound()) {
261                            final String lookup = xmlStringDecode(ats, LOOKUP, XML_STOP);
262                            throw new SymbolNotFoundException(lookup);
263                    }
264                    
265                    /* market identifier; must be globally unique; */
266                    try {
267                            builder.setMarketId(Long.parseLong(xmlStringDecode(ats, GUID, XML_STOP)));
268                    } catch(Exception e) {
269                            return InstrumentDefinition.getDefaultInstance();
270                    }
271                    /* type of security, Forex, Equity, etc. */
272                    builder.setInstrumentType(InstrumentType.NO_TYPE_INST);
273                    
274                    /* liquidy type, default / implied / combined */
275                    builder.setBookLiquidity(BookLiquidity.NO_BOOK_LIQUIDITY);
276                    
277                    /* structure of book */
278                    builder.setBookStructure(BookStructure.NO_BOOK_STRUCTURE);
279                    
280                    /* book depth */
281                    builder.setBookDepth(0);
282                    
283                    /* vendor */
284                    builder.setVendorId("Barchart");
285                    
286                    /* market symbol; can be non unique; */
287                    builder.setSymbol(xmlStringDecode(ats, SYMBOL_REALTIME, XML_STOP));
288                    
289                    /* market free style description; can be used in full text search */
290                    builder.setDescription(String.valueOf(xmlStringDecode(ats, SYMBOL_COMMENT,
291                                    XML_PASS)));
292                    
293                    /* stock vs future vs etc. */
294                    builder.setCfiCode(xmlStringDecode(ats, SYMBOL_CODE_CFI, XML_PASS));
295                    
296                    /* price currency */
297                    builder.setCurrencyCode("USD");
298                    
299                    /* market originating exchange identifier */
300                    final DDF_Exchange exchange = DDF_Exchange.fromCode( 
301                                    xmlByteDecode(ats, EXCHANGE_DDF, XML_PASS));
302                    String eCode = new String(new byte[] {exchange.code});
303                    if(eCode == null || eCode.isEmpty()) {
304                            eCode = Exchanges.NULL_CODE;
305                    }
306                    builder.setExchangeCode(eCode);
307                    
308                    final DDF_Fraction frac = DDF_Fraction.fromBaseCode(
309                                    xmlByteDecode(ats, BASE_CODE_DDF, XML_STOP));
310                    
311                    /* price step / increment size / tick size */
312                    try {
313                            final long priceStepMantissa = xmlDecimalDecode(frac, ats,
314                                    PRICE_TICK_INCREMENT, XML_STOP);
315                            builder.setMinimumPriceIncrement(buildDecimal(priceStepMantissa, frac.decimalExponent));
316                    } catch (Exception e) {
317                            builder.setMinimumPriceIncrement(buildDecimal(0,0));
318                    }
319                    
320                    /* value of a future contract / stock share */
321                    final String pricePointString = xmlStringDecode(ats, PRICE_POINT_VALUE, XML_PASS);
322                    if(pricePointString == null || pricePointString.isEmpty()) {
323                            builder.setContractPointValue(buildDecimal(0,0));
324                    } else {
325                            PriceValue pricePoint = ValueBuilder.newPrice(Double
326                                            .valueOf(pricePointString));
327                            builder.setContractPointValue(buildDecimal(
328                                            pricePoint.mantissa(), pricePoint.exponent()));
329                    }
330                    
331                    /* display fraction base : decimal(10) vs binary(2), etc. */
332                    builder.setDisplayBase((int)frac.fraction.base());
333                    builder.setDisplayExponent(frac.fraction.exponent());
334                    
335                    /* Calendar */
336                    final Time expire = xmlTimeDecode(ats, SYMBOL_EXPIRE, XML_PASS);
337                    final Calendar.Builder calBuilder = Calendar.newBuilder();
338                    final Interval.Builder intBuilder = Interval.newBuilder();
339                    
340                    intBuilder.setTimeStart(0);
341                    if(expire == null) {
342                            intBuilder.setTimeFinish(0);
343                    } else {
344                            intBuilder.setTimeFinish(expire.millisecond());
345                    }
346                    
347                    calBuilder.setLifeTime(intBuilder.build());
348                    builder.setCalendar(calBuilder.build());
349    
350                    //
351                    final DDF_TimeZone zone = DDF_TimeZone.fromCode(xmlStringDecode(
352                                    ats, TIME_ZONE_DDF, XML_STOP));
353                    
354                    /* timezone represented as offset in minutes from utc */
355                    builder.setTimeZoneOffset(zone.getUTCOffset());
356                    
357                    /* time zone name as text */
358                    builder.setTimeZoneName(zone.name());
359                    
360                    return builder.build();
361                    
362            }
363            
364            public static Decimal buildDecimal(final long mantissa, final int exponent) {
365                    
366                    final Decimal.Builder builder = Decimal.newBuilder();
367                    builder.setMantissa(mantissa);
368                    builder.setExponent(exponent);
369                    return builder.build();
370                    
371            }
