008    package com.barchart.feed.ddf.settings.api;
009    
010    import java.util.Set;
011    
012    import com.barchart.util.anno.NotMutable;
013    
014    /**
015     * Encapsulates all necessary data for a user login.
016     * 
017     */
018    @NotMutable
019    public interface DDF_Login {
020    
021            /**
022             * 
023             * @return The Barchart given user name used to populate login data.
024             */
025            String getUsername();
026    
027            /**
028             * Call to determine the status of a login attempt.
029             * 
030             * @return String "ok" if user name is valid.
031             */
032            String getStatus();
033    
034            /**
035             * Call to determine the status of a login attempt.
036             * 
037             * @return String "ok" if both user name and password are valid.
038             */
039            String getCredentials();
040    
041            /**
042             * What is the serviceId? How is it used?
043             * 
044             * @return
045             */
046            String getServiceId();
047    
048            /**
049             * 
050             * @return The maximum number of symbols the user is allowed to listen to on
051             *         this connection.
052             */
053            int getMaxSymbols();
054    
055            /**
056             * 
057             * @return The set of exchanges available to the user from this connection.
058             */
059            Set<String> getExchangeSet();
060    
061            /**
062             * 
063             * @return Returns true if both getStatus() and getCredentials() return
064             *         "ok".
065             */
066            boolean isValid();
067    
068    }
