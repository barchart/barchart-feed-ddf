001    package com.barchart.feed.ddf.instrument.provider;
002    
003    import org.openfeed.proto.inst.Decimal;
004    import org.openfeed.proto.inst.InstrumentDefinition;
005    
006    import com.barchart.feed.api.model.meta.Instrument;
007    import com.barchart.feed.inst.participant.InstrumentState;
008    import com.barchart.feed.inst.provider.InstrumentFactory;
009    import com.barchart.util.value.api.Fraction;
010    import com.barchart.util.value.api.Price;
011    
012    public class InstrumentStateFactory {
013            
014            public static InstrumentState newInstrument(final String symbol) {
015                    
016                    InstrumentDefinition.Builder builder = InstrumentDefinition.newBuilder();
017                    
018                    builder.setSymbol(symbol);
019                    
020                    return InstrumentFactory.instrumentState(builder.buildPartial());
021                    
022            }
023            
024            public static InstrumentState newInstrumentFromStub(final Instrument inst) {
025                    
026                    InstrumentDefinition.Builder builder = InstrumentDefinition.newBuilder();
027                    
028                    builder.setSymbol(inst.symbol());
029                    builder.setDescription(inst.description());
030                    
031                    builder.setRecordCreateTime(System.currentTimeMillis());
032                    builder.setRecordUpdateTime(System.currentTimeMillis());
033                    
034                    final Price tickSize = inst.tickSize();
035                    
036                    builder.setMinimumPriceIncrement(Decimal.newBuilder()
037                                    .setMantissa(tickSize.mantissa())
038                                    .setExponent(tickSize.exponent()));
039                    
040                    final Fraction frac = inst.displayFraction();
041                    
042                    builder.setDisplayBase((int) frac.base());
043                    builder.setDisplayExponent(frac.exponent());
044                    
045                    builder.setExchangeCode(inst.exchangeCode());
046                    
047                    return InstrumentFactory.instrumentState(builder.buildPartial());
048                    
049            }
050            
051    }
