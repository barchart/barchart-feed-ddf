001    package com.barchart.feed.ddf.market.provider;
002    
003    import org.slf4j.Logger;
004    import org.slf4j.LoggerFactory;
005    
006    import com.barchart.feed.api.model.meta.Instrument;
007    import com.barchart.feed.base.market.api.MarketDo;
008    import com.barchart.feed.base.market.api.MarketFactory;
009    import com.barchart.feed.base.provider.MarketProviderBase;
010    import com.barchart.feed.base.sub.SubscriptionHandler;
011    import com.barchart.feed.ddf.instrument.provider.DDF_MetadataServiceWrapper;
012    import com.barchart.feed.ddf.market.api.DDF_MarketProvider;
013    import com.barchart.feed.ddf.message.api.DDF_MarketBase;
014    import com.barchart.feed.ddf.message.api.DDF_MessageVisitor;
015    
016    public class DDF_ConsumerMarketProvider extends MarketProviderBase<DDF_MarketBase> 
017                    implements DDF_MarketProvider {
018    
019            protected static final Logger log = LoggerFactory.getLogger(
020                            DDF_ConsumerMarketProvider.class);
021                            
022            protected DDF_ConsumerMarketProvider(final MarketFactory factory,
023                            final SubscriptionHandler handler) {
024                    super(factory, new DDF_MetadataServiceWrapper(), handler);
025            }
026                            
027            private final DDF_MessageVisitor<Void, MarketDo> visitor = new MapperDDF();
028            
029            public static final DDF_ConsumerMarketProvider newInstance(
030                            final SubscriptionHandler handler) {
031                    
032                    return new DDF_ConsumerMarketProvider(new MarketFactory() {
033                            
034                            @Override
035                            public MarketDo newMarket(final Instrument instrument) {
036                                    return new VarMarketEntityDDF(instrument);
037                            }
038                            
039                    }, handler);
040            }
041    
042            @Override
043            protected void make(final DDF_MarketBase message, final MarketDo market) {
044                    message.accept(visitor, market);
045            }
046            
047    }
