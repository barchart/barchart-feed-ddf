001    package com.barchart.feed.ddf.instrument.provider;
002    
003    import java.util.ArrayList;
004    import java.util.Collection;
005    import java.util.Collections;
006    import java.util.HashMap;
007    import java.util.List;
008    import java.util.Map;
009    import java.util.concurrent.Callable;
010    import java.util.concurrent.ConcurrentHashMap;
011    import java.util.concurrent.ConcurrentMap;
012    import java.util.concurrent.ExecutorService;
013    
014    import org.slf4j.Logger;
015    import org.slf4j.LoggerFactory;
016    
017    import com.barchart.feed.api.model.meta.Instrument;
018    import com.barchart.feed.ddf.instrument.api.DDF_DefinitionService;
019    import com.barchart.feed.inst.InstrumentFuture;
020    import com.barchart.feed.inst.InstrumentFutureMap;
021    import com.barchart.feed.inst.provider.InstrumentFactory;
022    import com.barchart.proto.buf.inst.InstrumentDefinition;
023    import com.barchart.util.values.provider.ValueBuilder;
024    
025    public class ServiceDatabaseDDF implements DDF_DefinitionService {
026            
027            static final Logger log = LoggerFactory.getLogger(ServiceDatabaseDDF.class);
028            
029            private final ConcurrentMap<CharSequence, List<Instrument>> cache = 
030                            new ConcurrentHashMap<CharSequence, List<Instrument>>();
031            
032            private final LocalInstrumentDBMap db;
033            private final ExecutorService executor;
034            
035            public ServiceDatabaseDDF(final LocalInstrumentDBMap map, 
036                            final ExecutorService executor) {
037                    
038                    this.db = map;
039                    this.executor = executor;
040            }
041    
042            @Override
043            public List<Instrument> lookup(final CharSequence symbol) {
044                    
045                    try {
046                            return retrieve(symbol);
047                    } catch (final Throwable t) {
048                            log.error("Lookup failed", t);
049                            return Collections.emptyList();
050                    }
051                    
052            }
053    
054            @Override
055            public InstrumentFuture lookupAsync(final CharSequence symbol) {
056                    
057                    InstrumentFuture future = new InstrumentFuture();
058                    
059                    executor.submit(new LookupCallable(symbol, future));
060                    
061                    return future;
062                    
063            }
064    
065            @Override
066            public Map<CharSequence, List<Instrument>> lookup(
067                            final Collection<? extends CharSequence> symbols) {
068                    
069                    try {
070                            return retrieveMap(symbols);
071                    } catch (final Throwable t) {
072                            log.error("Lookup failed", t);
073                            return Collections.emptyMap();
074                    }
075                    
076            }
077    
078            @Override
079            public InstrumentFutureMap<CharSequence> lookupAsync(
080                            final Collection<? extends CharSequence> symbols) {
081                    
082                    InstrumentFutureMap<CharSequence> future = 
083                                    new InstrumentFutureMap<CharSequence>();
084                    
085                    executor.submit(new LookupMapCallable(symbols, future));
086                    
087                    return future;
088                    
089            }
090            
091            private List<Instrument> retrieve(CharSequence symbol) {
092                    
093                    symbol = ValueBuilder.newText(symbol.toString());
094                    
095                    if(symbol == null || symbol.length() == 0) {
096                            return Collections.emptyList();
097                    }
098                    
099                    List<Instrument> instrument = cache.get(symbol);
100                    
101                    if(instrument != null) {
102                            return instrument;
103                    }
104                    
105                    final InstrumentDefinition instDef = db.get(symbol.toString());
106                    
107                    if(cache.size() % 10000 == 0) {
108                            log.debug("Cache size = {}", cache.size());
109                    }
110                    
111                    if(instDef == null) {
112                            //log.debug("Symbol {} not in db", symbol);
113                            cache.put(symbol, new ArrayList<Instrument>(0));
114                            return Collections.emptyList();
115                    } else {
116                            instrument = Collections.singletonList(
117                                            InstrumentFactory.buildFromProtoBuf(instDef));
118                            cache.put(symbol, instrument);
119                            return instrument;
120                    }
121            }
122            
123            private class LookupCallable implements Callable<List<Instrument>> {
124    
125                    private final CharSequence symbol;
126                    private final InstrumentFuture future;
127                    
128                    public LookupCallable(final CharSequence symbol,
129                                    final InstrumentFuture future) {
130                            this.symbol = symbol;
131                            this.future = future;
132                    }
133                    
134                    @Override
135                    public List<Instrument> call() throws Exception {
136                            
137                            List<Instrument> inst;
138                            
139                            try {
140                            
141                                    inst = retrieve(symbol);
142                                    future.succeed(inst);
143                                    
144                            } catch (final Throwable t) {
145                                    future.fail(t);
146                                    return Collections.emptyList();
147                            }
148                            
149                            return inst;
150                            
151                    }
152                    
153            }
154            
155            private final class LookupMapCallable implements 
156                            Callable<Map<CharSequence, List<Instrument>>> {
157    
158                    private final Collection<? extends CharSequence> symbols;
159                    private final InstrumentFutureMap<CharSequence> future;
160    
161                    LookupMapCallable(final Collection<? extends CharSequence> symbols,
162                                    final InstrumentFutureMap<CharSequence> future) {
163                            this.symbols = symbols;
164                            this.future = future;
165                    }
166    
167                    @Override
168                    public Map<CharSequence, List<Instrument>> call() throws Exception {
169                            
170                            Map<CharSequence, List<Instrument>> map;
171                            
172                            try {
173                                    
174                                    map = retrieveMap(symbols);
175                                    future.succeed(map);
176                                    
177                            } catch (final Throwable t) {
178                                    future.fail(t);
179                                    return Collections.emptyMap();
180                            }
181                            
182                            return map;
183                    }
184    
185            }
186            
187            private Map<CharSequence, List<Instrument>> retrieveMap(
188                            final Collection<? extends CharSequence> symbols) {
189                    
190                    if (symbols == null || symbols.size() == 0) {
191                            log.warn("Lookup called with empty collection");
192                            return Collections.emptyMap(); 
193                    }
194                    
195                    final Map<CharSequence, List<Instrument>> result = 
196                                    new HashMap<CharSequence, List<Instrument>>();
197    
198                    for (final CharSequence symbol : symbols) {
199                            try {
200                                    result.put(symbol.toString(), retrieve(symbol));
201                            } catch (final Exception e) {
202                                    result.put(symbol.toString(), new ArrayList<Instrument>(0));
203                            }
204                    }
205                    
206                    return result;
207                    
208            }
209    
210    }
