025    public class ServiceDatabaseDDF implements DDF_DefinitionService {
026            
027            static final Logger log = LoggerFactory.getLogger(ServiceDatabaseDDF.class);
028            
029            private static final List<Instrument> EMPTY_LIST = Collections.emptyList();
030            
031            private final ConcurrentMap<CharSequence, List<Instrument>> cache = 
032                            new ConcurrentHashMap<CharSequence, List<Instrument>>();
033            
034            private final ConcurrentMap<CharSequence, Boolean> failedCache = 
035                            new ConcurrentHashMap<CharSequence, Boolean>();
036            
037            private final LocalInstrumentDBMap db;
038            private final ExecutorService executor;
039            private final ServiceMemoryDDF remoteInstService;
040            
041            public ServiceDatabaseDDF(final LocalInstrumentDBMap map, 
042                            final ExecutorService executor) {
043                    
044                    this.db = map;
045                    this.executor = executor;
046                    remoteInstService = new ServiceMemoryDDF(executor);
047                    
048            }
049    
050            @Override
051            public List<Instrument> lookup(final CharSequence symbol) {
052                    
053                    try {
054                            return retrieve(symbol);
055                    } catch (final Throwable t) {
056                            log.error("Lookup failed", t);
057                            return Collections.emptyList();
058                    }
059                    
060            }
061    
062            @Override
063            public InstrumentFuture lookupAsync(final CharSequence symbol) {
064                    
065                    InstrumentFuture future = new InstrumentFuture();
066                    
067                    executor.submit(new LookupCallable(symbol, future));
068                    
069                    return future;
070                    
071            }
072    
073            @Override
074            public Map<CharSequence, List<Instrument>> lookup(
075                            final Collection<? extends CharSequence> symbols) {
076                    
077                    try {
078                            return retrieveMap(symbols);
079                    } catch (final Throwable t) {
080                            log.error("Lookup failed", t);
081                            return Collections.emptyMap();
082                    }
083                    
084            }
085    
086            @Override
087            public InstrumentFutureMap<CharSequence> lookupAsync(
088                            final Collection<? extends CharSequence> symbols) {
089                    
090                    InstrumentFutureMap<CharSequence> future = 
091                                    new InstrumentFutureMap<CharSequence>();
092                    
093                    executor.submit(new LookupMapCallable(symbols, future));
094                    
095                    return future;
096                    
097            }
098            
099            private List<Instrument> retrieve(CharSequence symbol) {
100                    
101                    symbol = ValueBuilder.newText(symbol.toString());
102                    
103                    if(symbol == null || symbol.length() == 0) {
104                            return Collections.emptyList();
105                    }
106                    
107                    if(failedCache.containsKey(symbol)) {
108                            return EMPTY_LIST;
109                    }
110                    
111                    /*
112                     * FIXME HACK
113                     * Currently need to filter options because of symbology
114                     */
115                    if(symbol.charAt(symbol.length() - 1) == 'P' ||
116                            symbol.charAt(symbol.length() - 1) == 'C' ||
117                            symbol.charAt(symbol.length() - 1) == 'Q' ||
118                            symbol.charAt(symbol.length() - 1) == 'D') {
119                            return EMPTY_LIST;
120                    }
121                    
122                    List<Instrument> instrument = cache.get(symbol);
123                    
124                    if(instrument != null) {
125                            return instrument;
126                    }
127                    
128                    final InstrumentDefinition instDef = db.get(symbol.toString());
129                    
130                    if(cache.size() % 10000 == 0) {
131                            log.debug("Cache size = {}", cache.size());
132                    }
133                    
134                    if(instDef != null) {
135                            
136                            instrument = Collections.singletonList(
137                                            InstrumentFactory.buildFromProtoBuf(instDef));
138                            cache.put(symbol, instrument);
139                            
140                            return instrument;
141                            
142                    } 
143                    
144                    //log.debug("Symbol {} not found in DB, using remote", symbol);
145                    instrument = remoteInstService.lookup(symbol);
146                    
147                    if(instrument != null && instrument.size() > 0) {
148                            cache.put(symbol, instrument);
149                            return instrument;
150                    } else {
151                            failedCache.put(symbol, false);
152                            return EMPTY_LIST;
153                    }
154                    
155            }
156            
157            private class LookupCallable implements Callable<List<Instrument>> {
158    
159                    private final CharSequence symbol;
160                    private final InstrumentFuture future;
161                    
162                    public LookupCallable(final CharSequence symbol,
163                                    final InstrumentFuture future) {
164                            this.symbol = symbol;
165                            this.future = future;
166                    }
167                    
168                    @Override
169                    public List<Instrument> call() throws Exception {
170                            
171                            List<Instrument> inst;
172                            
173                            try {
174                            
175                                    inst = retrieve(symbol);
176                                    future.succeed(inst);
177                                    
178                            } catch (final Throwable t) {
179                                    future.fail(t);
180                                    return Collections.emptyList();
181                            }
182                            
183                            return inst;
184                            
185                    }
186                    
187            }
188            
189            private final class LookupMapCallable implements 
190                            Callable<Map<CharSequence, List<Instrument>>> {
191    
192                    private final Collection<? extends CharSequence> symbols;
193                    private final InstrumentFutureMap<CharSequence> future;
194    
195                    LookupMapCallable(final Collection<? extends CharSequence> symbols,
196                                    final InstrumentFutureMap<CharSequence> future) {
197                            this.symbols = symbols;
198                            this.future = future;
199                    }
200    
201                    @Override
202                    public Map<CharSequence, List<Instrument>> call() throws Exception {
203                            
204                            Map<CharSequence, List<Instrument>> map;
205                            
206                            try {
207                                    
208                                    map = retrieveMap(symbols);
209                                    future.succeed(map);
210                                    
211                            } catch (final Throwable t) {
212                                    future.fail(t);
213                                    return Collections.emptyMap();
214                            }
215                            
216                            return map;
217                    }
218    
219            }
220            
221            private Map<CharSequence, List<Instrument>> retrieveMap(
222                            final Collection<? extends CharSequence> symbols) {
223                    
224                    if (symbols == null || symbols.size() == 0) {
225                            log.warn("Lookup called with empty collection");
226                            return Collections.emptyMap(); 
227                    }
228                    
229                    final Map<CharSequence, List<Instrument>> result = 
230                                    new HashMap<CharSequence, List<Instrument>>();
231    
232                    for (final CharSequence symbol : symbols) {
233                            try {
234                                    result.put(symbol.toString(), retrieve(symbol));
235                            } catch (final Exception e) {
236                                    result.put(symbol.toString(), new ArrayList<Instrument>(0));
237                            }
238                    }
239                    
240                    return result;
241                    
242            }
