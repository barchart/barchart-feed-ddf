008    package com.barchart.feed.ddf.market.provider;
009    
010    import com.barchart.feed.base.market.api.MarketDo;
011    import com.barchart.feed.base.market.api.MarketFactory;
012    import com.barchart.feed.base.provider.MakerBase;
013    import com.barchart.feed.ddf.market.api.DDF_MarketProvider;
014    import com.barchart.feed.ddf.message.api.DDF_MarketBase;
015    import com.barchart.feed.ddf.message.api.DDF_MessageVisitor;
016    import com.barchart.util.anno.ThreadSafe;
017    
018    /**
019     * The DDF extension of MakerBase from the feed-base library. Sets the Market
020     * objects used by the market maker to DDF specific VarMarket.
021     */
022    @ThreadSafe
023    public class DDF_MarketService extends MakerBase<DDF_MarketBase> implements
024                    DDF_MarketProvider {
025    
026            /**
027             * @param factory
028             */
029            protected DDF_MarketService(final MarketFactory factory) {
030                    super(factory);
031    
032            }
033    
034            private final DDF_MessageVisitor<Void, MarketDo> visitor = new MapperDDF();
035    
036            public static final DDF_MarketProvider newInstance() {
037                    return new DDF_MarketService(new MarketFactory() {
038    
039                            @Override
040                            public MarketDo newMarket() {
041                                    return new VarMarketDDF();
042                            }
043    
044                    });
045            }
046    
047            
048            
049            @Override
050            protected void make(final DDF_MarketBase message, final MarketDo market) {
051                    message.accept(visitor, market);
052            }
053    
054    }
