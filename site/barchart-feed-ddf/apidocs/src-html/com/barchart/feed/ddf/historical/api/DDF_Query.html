011    import static com.barchart.feed.ddf.historical.enums.DDF_QueryType.MINUTES;
012    import static com.barchart.feed.ddf.historical.enums.DDF_QueryType.MINUTES_FORM_T;
013    import static com.barchart.feed.ddf.historical.enums.DDF_QueryType.MINUTES_NEARBY;
014    import static com.barchart.feed.ddf.historical.enums.DDF_QueryType.TICKS;
015

017    import org.joda.time.DateTimeZone;
018

021    import com.barchart.feed.ddf.historical.enums.DDF_QueryEodVolume;

023    import com.barchart.feed.ddf.historical.enums.DDF_QueryType;

027    // TODO: Auto-generated Javadoc
028    /**
029     * ddf historical market data query builder.
030     * 
031     * @param <E>
032     *            the element type
033     */
034    @Mutable
035    public final class DDF_Query<E extends DDF_Entry> implements
036                    PublicCloneable<DDF_Query<E>> {
037    
038            /**
039             * Instantiates a new dD f_ query.
040             * 
041             * @param queryType
042             *            the query type
043             */
044            public DDF_Query(final DDF_QueryType<E> queryType) {
045                    this.type = queryType;
046            }
047    
048            /** The type. */
049            public DDF_QueryType<E> type;
050    
051            /** The instrument. */
052            public Instrument instrument;
053    
054            /**
055             * this parameter should be set to the desired START date/time for  the
056             * query; the result set will include records back to, AND including, this
057             * value.
058             */
059            public DateTime timeStart;
060    
061            /**
062             * this parameter should be set to the desired END date/time for the  query;
063             * the result set will include records up to, but NOT including, this 
064             * value.
065             */
066            public DateTime timeEnd;
067    
068            /** query result sort order. */
069            public DDF_QueryOrder resultOrder = DDF_QueryOrder.ASCENDING;
070    
071            /** positive limit of records to return; zero for no limit. */
072            public int maxRecords = 0;
073    
074            /**
075             * An optional interval can be specified in order to aggregate sets of 
076             * contiguous
077             * minute records (the number of minute records in each set is equal
078             * to the specified interval) into one record. If the interval is omitted, 
079             * then a 1 minute interval (no aggregation) will be the  default.
080             */
081            public int groupBy = 1;
082    
083            /** The eod type. */
084            public DDF_QueryEodType eodType;
085    
086            /** The eod volume. */
087            public DDF_QueryEodVolume eodVolume;
088    
089            /*
090             * (non-Javadoc)
091             * 
092             * @see java.lang.Object#clone()
093             */
094            // public boolean isTrend;
095    
096            @SuppressWarnings("unchecked")
097            @Override
098            public final DDF_Query<E> clone() {
099                    try {
100                            return (DDF_Query<E>) super.clone();
101                    } catch (final Exception e) {
102                            // should not happen
103                            e.printStackTrace();
104                            return null;
105                    }
106            }
107    
108            private final CharSequence renderId() {
109                    return instrument == null ? "NONE" : instrument.symbol();
110            }
111    
112            private final CharSequence renderDescription() {
113                    return instrument == null ? "NONE" : instrument.description();
114            }
115    
116            private final static DateTime NULL_TIME = new DateTime(0, DateTimeZone.UTC);
117    
118            private final CharSequence renderTime(/* local */DateTime time) {
119                    if (time == null) {
120                            time = NULL_TIME;
121                    }
122                    if (instrument == null) {
123                            return time.toString();
124                    } else {
125                            final DateTimeZone zone = DateTimeZone.forOffsetMillis((int)(instrument.timeZoneOffset()));
126                            return time.withZone(zone).toString();
127                    }
128            }
129    
130            private final CharSequence renderOrder(/* local */DDF_QueryOrder order) {
131                    if (order == null) {
132                            order = DDF_QueryOrder.ASCENDING;
133                    }
134                    return order.toString();
135            }
136    
137            private final CharSequence renderEodType(/* local */DDF_QueryEodType type) {
138                    if (type == null) {
139                            type = DDF_QueryEodType.DAILY;
140                    }
141                    return type.toString();
142            }
143    
144            private final CharSequence renderEodVolume(
145            /* local */DDF_QueryEodVolume volume) {
146                    if (volume == null) {
147                            volume = DDF_QueryEodVolume.CONTRACT;
148                    }
149                    return volume.toString();
150            }
151    
152            /*
153             * (non-Javadoc)
154             * 
155             * @see java.lang.Object#toString()
156             */
157            @Override
158            public final String toString() {
159                    return "" + //
160                                    "\n queryType   " + type + //
161                                    "\n instrument  " + renderId() + //
162                                    "\n timeStart   " + timeStart + //
163                                    "\n timeEnd     " + timeEnd + //
164                                    "\n resultOrder " + resultOrder + //
165                                    "\n maxRecords  " + maxRecords + //
166                                    "\n groupBy     " + groupBy + //
167                                    "\n eodType     " + eodType + //
168                                    "\n eodVolume   " + eodVolume + //
169                                    "";
170            }
171    
172            /**
173             * Description.
174             * 
175             * @return the string
176             */
177            public final String description() {
178                    final StringBuilder text = new StringBuilder(128);
179                    build: {
180                            if (type == null) {
181                                    text.append("<invalid query>");
182                                    break build;
183                            }
184                            //
185                            text.append(type);
186                            text.append(" ");
187                            text.append(renderId());
188                            text.append("(");
189                            text.append(renderDescription());
190                            text.append(")");
191                            text.append(" from ");
192                            text.append(renderTime(timeStart));
193                            text.append(" upto ");
194                            text.append(renderTime(timeEnd));
195                            text.append(" order ");
196                            text.append(renderOrder(resultOrder));
197                            //
198                            if (type.is(TICKS)) {
199                                    break build;
200                            }
201                            if (type.isIn(MINUTES, MINUTES_NEARBY, MINUTES_FORM_T)) {
202                                    text.append(" group ");
203                                    text.append(groupBy);
204                                    break build;
205                            }
206                            if (type.is(END_OF_DAY)
207                                            && instrument != null
208                                            && instrument.CFICode().charAt(0) == 'F') {
209                                    text.append(" eodType ");
210                                    text.append(renderEodType(eodType));
211                                    text.append(" eodVolume ");
212                                    text.append(renderEodVolume(eodVolume));
213                                    break build;
214                            }
215                    }
216                    return text.toString();
217            }
218    
219            /**
220             * Inits the from.
221             * 
222             * @param that
223             *            the that
224             */
225            public final void initFrom(final DDF_Query<E> that) {
226                    this.type = that.type;
227                    this.instrument = that.instrument;
228                    this.timeStart = that.timeStart;
229                    this.timeEnd = that.timeEnd;
230                    this.groupBy = that.groupBy;
231                    this.maxRecords = that.maxRecords;
232                    this.eodType = that.eodType;
233                    this.eodVolume = that.eodVolume;
234            }
