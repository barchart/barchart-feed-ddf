034                    super(instrument);

036            
037            
038            /* ***** ***** ***** Update State Methods ***** ***** ***** */
039            
040            
041            @SuppressWarnings("unchecked")
042            private <V extends MarketData<V>> void fireCallbacks(
043                            final Class<V> clazz) {
044                    
045                    for(final FrameworkAgent agent : agentMap.get(clazz)) {
046                            
047                            if(agent.isActive()) {
048                                    agent.callback().onNext(agent.data(this));
049                            }
050                            
051                    }
052                    
053                    for(final FrameworkAgent agent : agentMap.get(Market.class)) {
054                            
055                            if(agent.isActive()) {
056                                    agent.callback().onNext(agent.data(this));
057                            }
058                            
059                    }
060                    
061            }
062            
063            @Override
064            public void setInstrument(final Instrument newSymbol) {
065                    super.setInstrument(newSymbol);
066                    
067                    // Currently not firing on instruments
068            }
069            
070            @Override
071            public void setBookSnapshot(final MarketDoBookEntry[] entries,
072                            final TimeValue time) {
073                    super.setBookSnapshot(entries, time);
074                    
075                    //log.debug("Set book snapshot, firing callbacks");
076                    
077                    fireCallbacks(Book.class);
078            }
079            
080            @Override
081            public void setBookUpdate(final MarketDoBookEntry entry,
082                            final TimeValue time) {
083                    super.setBookUpdate(entry, time);
084                    
085                    log.debug("Set book update, firing callbacks");
086                    
087                    // Book updates aren't being fired when GO ESU3=bB, 
088                    // only when all interests are subbed to
089                    fireCallbacks(Book.class);
090            }
091            
092            @Override
093            public void setCuvolUpdate(final MarketDoCuvolEntry entry,
094                            final TimeValue time) {
095                    super.setCuvolUpdate(entry, time);
096                    
097                    //log.debug("Set cuvol update, firing callbacks");
098                    
099                    fireCallbacks(Cuvol.class);
100            }
101

104                            final TimeValue time) {
105                    super.setCuvolSnapshot(entries, time);
106                    
107                    //log.debug("Set cuvol snapshot, firing callbacks");
108    
109                    fireCallbacks(Cuvol.class);
110            }
111            
112            @Override
113            public void setTrade(final MarketTradeType type,
114                            final MarketTradeSession session,
115                            final MarketTradeSequencing sequencing, final PriceValue price,
116                            final SizeValue size, final TimeValue time, final TimeValue date) {
117                    super.setTrade(type, session, sequencing, price, size, time, date);
118                    
119                    //log.debug("Set trade, firing callbacks");
120                    
121                    fireCallbacks(Trade.class);
122                    fireCallbacks(Cuvol.class);  // HACK FOR DDF
123                    fireCallbacks(Session.class);  // HACK FOR DDF
124            }
125

127            public void setBar(final MarketBarType type, final MarketDoBar bar) {
128                    super.setBar(type, bar);
129                    
130                    //log.debug("Set bar, firing callbacks");

132                    fireCallbacks(Session.class);
133            }
134

137                    super.setState(entry, isOn);
138                    
139                    //log.debug("Set state");

141                    // Currently not firing on state
142            }
