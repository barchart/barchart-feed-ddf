001    package com.barchart.feed.ddf.market.provider;
002    
003    import com.barchart.feed.api.model.meta.Instrument;
004    import com.barchart.feed.base.market.api.MarketDo;
005    import com.barchart.feed.base.market.api.MarketFactory;
006    import com.barchart.feed.base.provider.MarketplaceBase;
007    import com.barchart.feed.base.sub.SubscriptionHandler;
008    import com.barchart.feed.ddf.instrument.provider.InstrumentProviderWrapper;
009    import com.barchart.feed.ddf.market.api.DDF_MarketProvider;
010    import com.barchart.feed.ddf.message.api.DDF_MarketBase;
011    import com.barchart.feed.ddf.message.api.DDF_MessageVisitor;
012    import com.barchart.feed.inst.InstrumentService;
013    
014    public class DDF_Marketplace extends MarketplaceBase<DDF_MarketBase> implements 
015                    DDF_MarketProvider {
016    
017            protected DDF_Marketplace(MarketFactory factory,
018                            InstrumentService<String> instLookup,
019                            SubscriptionHandler handler) {
020                    super(factory, instLookup, handler);
021            }
022            
023            private final DDF_MessageVisitor<Void, MarketDo> visitor = new MapperDDF();
024            
025            public static final DDF_Marketplace newInstance(
026                            final SubscriptionHandler handler) {
027                    
028                    return new DDF_Marketplace(new MarketFactory() {
029    
030                            @Override
031                            public MarketDo newMarket(final Instrument instrument) {
032                                    return new VarMarketEntityDDF(instrument);
033                            }
034    
035                    }, new InstrumentProviderWrapper(), handler);
036                    
037            }
038    
039            @Override
040            protected void make(DDF_MarketBase message, MarketDo market) {
041                    message.accept(visitor, market);
042            }
043    
044    }
