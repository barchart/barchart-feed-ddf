008    package com.barchart.feed.ddf.historical.api;
009    
010    import java.util.List;
011    
012    import com.barchart.feed.ddf.historical.enums.DDF_ResultStatus;
013    import com.barchart.util.thread.Runner;
014    import com.barchart.util.thread.RunnerLoop;
015    
016    /**
017     * The Interface DDF_Result.
018     *
019     * @param <E> the element type
020     */
021    public interface DDF_Result<E extends DDF_Entry> extends RunnerLoop<E> {
022    
023            
024            /** clone of original query */
025            DDF_Query<E> getQuery();
026    
027            String getQueryURL();
028            
029            //
030    
031            DDF_ResultStatus getStatus();
032    
033            String getStatusComment();
034    
035            //
036    
037            /** size of result data set */
038            int size();
039    
040            /** entry at given index */
041            E get(int index) throws IndexOutOfBoundsException;
042    
043            //
044    
045            /** convert result data set into comma separated value lines */
046            String asCSV() throws DDF_ResultInterruptedException;
047    
048            @Override
049            String toString() throws DDF_ResultInterruptedException;
050    
051            //
052    
053            @Override
054            <R> void runLoop(Runner<R, E> task, List<R> list)
055                            throws DDF_ResultInterruptedException;
056    
057    }
