163            private static ExecutorService getDefault() {
164                    return Executors.newCachedThreadPool( 
165                                    
166                                    new ThreadFactory() {
167    
168                            final AtomicLong counter = new AtomicLong(0);
169                            
170                            @Override
171                            public Thread newThread(final Runnable r) {
172                                    
173                                    final Thread t = new Thread(r, "Feed thread " + 
174                                                    counter.getAndIncrement()); 
175                                    
176                                    t.setDaemon(true);
177                                    
178                                    return t;
179                            }
180                            
181                    });
182            }

208            /**
209             * Starts the data feed asynchronously. Notification of login success is
210             * reported by FeedStateListeners which are bound to this object.
211             * <p>
212             * Constructs a new feed client with the user's user name and password. The
213             * transport protocol defaults to TCP and a default executor are used.
214             * 
215             * @param username
216             * @param password
217             */
218            
219            private final AtomicBoolean isStartingup = new AtomicBoolean(false);
220            private final AtomicBoolean isShuttingdown = new AtomicBoolean(false);
221            
222            @Override
223            public synchronized void startup() {
224                    
225                    if(isStartingup.get()) {
226                            throw new IllegalStateException("Startup called while already starting up");
227                    }
228                    
229                    if(isShuttingdown.get()) {
230                            throw new IllegalStateException("Startup called while shutting down");
231                    }
232                    
233                    isStartingup.set(true);
234                    
235                    executor.execute(new StartupRunnable());
236                    
237                    /** Currently just letting it run */
238                    
239            }
240            
241            private volatile Future<?> dbUpdater = null;
242            
243            private final class StartupRunnable implements Runnable {
244    
245                    /*
246                     * Ensures instrument database is installed/updated before
247                     * user is able to send subscriptions to JERQ
248                     */
249                    @Override
250                    public void run() {
251                            
252                            try {
253                            
254                                    log.debug("Startup Runnable starting");
255                                    
256                                    connection.startup();
257                            
258                            } catch (final Throwable t) {
259                                    
260                                    log.error("Exception starting up marketplace {}", t);
261                                    isStartingup.set(false);
262                                    
263                                    return;
264                            }
265                            
266                            isStartingup.set(false);
267                            
268                    }
269                    
270            }
271            
272            @Override
273            public synchronized void shutdown() {
274    
275                    if(isStartingup.get()) {
276                            throw new IllegalStateException(
277                                            "Shutdown called while starting up");
278                    }
279                    
280                    if(isShuttingdown.get()) {
281                            throw new IllegalStateException(
282                                            "Shutdown called while already shutting down");
283                    }
284                    
285                    isShuttingdown.set(true);
286                    
287                    executor.execute(new ShutdownRunnable());
288                    
289                    /** Currently just letting it run */
290    
291            }
292            
293            private final class ShutdownRunnable implements Runnable {
294    
295                    @Override
296                    public void run() {
297                            
298                            try {
299                                    
300                                    if(maker != null) {
301                                            maker.clearAll();
302                                    }
303                                    
304                                    //dbUpdater.cancel(true);
305            
306                                    connection.shutdown();
307    
308                                    log.debug("Barchart Feed shutdown");
309                                    
310                            } catch (final Throwable t) {
311                                    
312                                    log.error("Error {}", t);
313                                    
314                                    isShuttingdown.set(false);
315                                    
316                                    return;
317                            }
318                            
319                            
320                            isShuttingdown.set(false);
321                            
322                            log.debug("Barchart Feed shutdown succeeded");
323                            
324                    }
325                    
326            }

328            /*
329             * This is the default message listener. Users wishing to handle raw
330             * messages will need to implement their own feed client.
331             */
332            private final DDF_MessageListener msgListener = new DDF_MessageListener() {
333    
334                    @Override
335                    public void handleMessage(final DDF_BaseMessage message) {
336    
337                            if (message instanceof DDF_ControlTimestamp) {
338                                    for (final TimestampListener listener : timeStampListeners) {
339                                            listener.listen(factory.newTime(((DDF_ControlTimestamp) message)
340                                                            .getStampUTC().asMillisUTC(), ""));
341                                    }
342                            }
343    
344                            if (message instanceof DDF_MarketBase) {
345                                    final DDF_MarketBase marketMessage = (DDF_MarketBase) message;
346                                    
347                                    maker.make(marketMessage);
348                            }
349    
350                    }
351    
352            };
353            
354            @Override
355            public void bindConnectionStateListener(final Connection.Monitor listener) {
356    
357                    stateListener = listener;
358    
359                    if (connection != null) {
360                            connection.bindStateListener(listener);
361                    } else {
362                            throw new RuntimeException("Connection state listener already bound");
363                    }
364    
365            }
366            
367            @Override
368            public void bindTimestampListener(final TimestampListener listener) {
369                    
370                    if(listener != null) {
371                            timeStampListeners.add(listener);
372                    }
373                    
374            }
