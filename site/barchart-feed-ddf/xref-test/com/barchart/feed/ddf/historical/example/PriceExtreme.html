8   package com.barchart.feed.ddf.historical.example;
9   
10  import com.barchart.feed.api.model.meta.Instrument;
11  import com.barchart.feed.base.market.api.MarketDisplay;
12  import com.barchart.feed.base.provider.MarketDisplayBaseImpl;
13  import com.barchart.util.values.api.PriceValue;
14  import com.barchart.util.values.provider.ValueBuilder;
15  
16  /**
17   * helper class to demonstrate batch processing of historical data
18   */
19  class PriceExtreme {
20  
21  	Instrument inst;
22  	
23  	static final MarketDisplay display = new MarketDisplayBaseImpl();
24  
25  	PriceExtreme(final Instrument instrument) {
26  		this.inst = instrument;
27  	}
28  
29  	long mantissaMin = Long.MAX_VALUE;
30  
31  	long mantissaMax = Long.MIN_VALUE;
32  
33  	/*
34  	 * (non-Javadoc)
35  	 * 
36  	 * @see java.lang.Object#toString()
37  	 */
38  	@Override
39  	public String toString() {
40  
41  		final int exponent = (int)inst.displayFraction().decimalExponent();
42  
43  		final PriceValue priceMin = ValueBuilder
44  				.newPrice(mantissaMin, exponent);
45  		final PriceValue priceMax = ValueBuilder
46  				.newPrice(mantissaMax, exponent);
47  
48  		final String stringMin = display.priceText(priceMin, 
49  				inst.displayFraction());
50  		final String stringMax = display.priceText(priceMax, 
51  				inst.displayFraction());
52  
53  		return String
54  				.format("minimum : %s  maximum : %s", stringMin, stringMax);
55  
56  	}
57  
58  }
