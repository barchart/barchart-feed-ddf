443 	/**
444 	 * via feed XQ snapshot or web URL lookup.
445 	 * 
446 	 * @param message
447 	 *            the message
448 	 * @param market
449 	 *            the market
450 	 * @return the void
451 	 */
452 	@Override
453 	public Void visit(final DDF_MarketQuote message, final MarketDo market) {
454 
455 		market.clearChanges();
456 		// ### process quote
457 
458 		// TODO part of instrument; should update definition?
459 		final TextValue symbolName = message.getSymbolName();
460 		final PriceValue priceStep = message.getPriceStep();
461 		final PriceValue pointValue = message.getPointValue();
462 
463 		// TODO add more complete flag support?
464 		final DDF_Condition condition = message.getCondition();
465 		final DDF_QuoteState state = message.getState();
466 		final DDF_QuoteMode mode = message.getMode();
467 
468 		/** Handle quote publish state */
469 		{
470 			switch (mode) {
471 			case END_OF_DAY:
472 			case DELAYED:
473 			case SNAPSHOT:
474 			case UNKNOWN:
475 				market.setState(MarketStateEntry.IS_PUBLISH_REALTIME, false);
476 				break;
477 			case REALTIME:
478 				market.setState(MarketStateEntry.IS_PUBLISH_REALTIME, true);
479 				break;
480 			default:
481 				break;
482 			}
483 		}
484 
485 		/** Handle settle state */
486 		{
487 			switch (state) {
488 			case GOT_CLOSE:
489 				break;
490 			case GOT_SETTLE:
491 				log.debug("Set State IS_SETTLED true inside visit MarketQuote by GOT_SETTLE state");
492 				market.setState(MarketStateEntry.IS_SETTLED, true);
493 				break;
494 			case PRE_MARKET:
495 				log.debug("Set State IS_SETTLED false inside visit MarketQuote by PRE_MARKET state");
496 				market.setState(MarketStateEntry.IS_SETTLED, false);
497 				break;
498 			}
499 		}
500 
501 		/** Process Sessions */
502 		{
503 			final DDF_MarketSession[] sessionArray = message.sessions();
504 
505 			for (final DDF_MarketSession session : sessionArray) {
506 				final DDF_Indicator indicator = session.getIndicator();
507 				log.debug("DDF_Indicator for session {} : {} ", session
508 						.getSession().name(), indicator.name());
509 				switch (indicator) {
510 				case CURRENT:
511 					
512 					/* Update changed comonents */
513 					market.setChange(Component.DEFAULT_CURRENT);
514 					
515 					visit(session, market);
516 					
517 					break;
518 				case PREVIOUS:
519 					
520 					/* Update changed comonents */
521 					market.setChange(Component.DEFAULT_PREVIOUS);
522 					
523 					visit(session, market);
524 					
525 					break;
526 				default:
527 					log.error("@@@ wrong indicator : {}", indicator);
528 					break;
529 				}
530 			}
531 		}
532 
533 		/** Process top of book */
534 		{
535 			
536 			/* Update changed comonents */
537 			market.setChange(Component.BOOK_COMBINED);
538 			
539 			final DDF_MarketBookTop bookTop = message;
540 			visit(bookTop, market);
541 			
542 		}
543 
544 		return null;
545 	}
