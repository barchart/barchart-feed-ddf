126 	private DDF_Fraction(final byte baseCode, final int unitCode,
127 			final Fraction fraction) {
128 
129 		this.baseCode = baseCode;
130 		this.unitCode = unitCode;
131 
132 		this.ord = (byte) ordinal();
133 
134 		this.fraction = fraction;
135 		this.decimalExponent = fraction.decimalExponent();
136 		this.decimalDenominator = fraction.decimalDenominator();
137 
138 		if(fraction.base() == 2) {
139 			this.isBinary = true;
140 		} else {
141 			this.isBinary = false;
142 		}
143 
144 		this.nativeExponent = fraction.exponent();
145 		this.nativeDenominator = fraction.denominator();
146 
147 		this.spaces = fraction.places();
148 
149 		this.spacesDenomPlus = (long) pow(10, spaces);
150 		this.spacesDenomMinus = (long) pow(10, -decimalExponent - spaces);
151 
152 	}
