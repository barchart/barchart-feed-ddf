57  	private void progressStart() {
58  		progressCount = 0;
59  		if (progressListener == null) {
60  			return;
61  		}
62  		progressListener.onProgressEvent(progressCount);
63  	}
64  
65  	private void progressUpdate(final String message) {
66  		// final Thread thread = Thread.currentThread();
67  		if (Thread.interrupted()) {
68  			throw new DDF_ResultInterruptedException(message);
69  		}
70  		if (progressListener == null) {
71  			return;
72  		}
73  		if (progressCount % DDF_ResultListener.PROGRESS_SIZE == 0) {
74  			progressListener.onProgressEvent(progressCount);
75  		}
76  		progressCount++;
77  	}
78  
79  	private void progressFinish() {
80  		if (progressListener == null) {
81  			return;
82  		}
83  		progressListener.onProgressEvent(progressCount);
84  	}

109 	@Override
110 	public <R> void runLoop(final Runner<R, E> task, final List<R> list) {
111 		progressStart();
112 		final int lenght = entryList.size();
113 		for (int index = 0; index < lenght; index++) {
114 			final E entry = entryList.get(index);
115 			final R result = task.run(entry);
116 			progressUpdate("run loop interrupted");
117 			if (list == null || result == null) {
118 				continue;
119 			}
120 			list.add(result);
121 		}
122 		progressFinish();
123 	}
