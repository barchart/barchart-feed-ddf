124 		if (time == null) {

178 	/**
179 	 * Description.
180 	 * 
181 	 * @return the string
182 	 */
183 	public final String description() {
184 		final StringBuilder text = new StringBuilder(128);
185 		build: {
186 			if (type == null) {
187 				text.append("<invalid query>");
188 				break build;
189 			}
190 			//
191 			text.append(type);
192 			text.append(" ");
193 			text.append(renderId());
194 			// text.append("(");
195 			// text.append(renderDescription());
196 			// text.append(")");
197 			text.append(" from ");
198 			text.append(renderTime(timeStart));
199 			text.append(" upto ");
200 			text.append(renderTime(timeEnd));
201 			text.append(" order ");
202 			text.append(renderOrder(resultOrder));
203 			//
204 			if (type.is(TICKS)) {
205 				break build;
206 			}
207 			if (type.isIn(MINUTES, MINUTES_NEARBY, MINUTES_FORM_T)) {
208 				text.append(" group ");
209 				text.append(groupBy);
210 				break build;
211 			}
212 			if (type.is(END_OF_DAY)
213 					&& instrument != null
214 					&& instrument.get(DDF_InstrumentField.DDF_EXCHANGE).kind == DDF_ExchangeKind.FUTURE) {
215 				text.append(" eodType ");
216 				text.append(renderEodType(eodType));
217 				text.append(" eodVolume ");
218 				text.append(renderEodVolume(eodVolume));
219 				break build;
220 			}
221 		}
222 		return text.toString();
223 	}
