46  	final static String SPACE = " ";

48  	private void addSpreadComponents(final StringBuilder text) {
49  
50  		String id = get(InstrumentField.SYMBOL).toString();
51  
52  		/** ddf prefix in spread symbology */
53  		if (id.startsWith("_S_")) {
54  
55  			id = id.replaceFirst("_S_", "");
56  
57  			id = id.replaceAll("_", " ");
58  
59  			text.append(id);
60  
61  		}
62  
63  	}
64  
65  	/*
66  	 * (non-Javadoc)
67  	 * 
68  	 * @see com.barchart.feed.ddf.instrument.api.DDF_Instrument#fullText()
69  	 */
70  	public String fullText() {
71  
72  		final StringBuilder text = new StringBuilder(256);
73  
74  		text.append(get(VENDOR));
75  		text.append(SPACE);
76  
77  		text.append(get(SYMBOL));
78  		text.append(SPACE);
79  
80  		text.append(get(DESCRIPTION));
81  		text.append(SPACE);
82  
83  		text.append(get(EXCHANGE_CODE));
84  		text.append(SPACE);
85  
86  		addSpreadComponents(text);
87  
88  		final TimeValue expire = ValueBuilder.newTime(get(LIFETIME).stop().millisecond());
89  		if (!expire.isNull()) {
90  
91  			text.append(display.timeMonthFull(expire));
92  			text.append(SPACE);
93  
94  			text.append(display.timeYearFull(expire));
95  			text.append(SPACE);
96  
97  			text.append(display.timeYearShort(expire));
98  			text.append(SPACE);
99  
100 		}
101 
102 		return text.toString();
103 
104 	}
