297 	static DDF_Instrument remoteLookup(CharSequence symbol) throws Exception {
298 
299 		if (overrideURL) {
300 			symbol = symbol + "&bats=1";
301 		}
302 
303 		final String symbolURI = urlInstrumentLookup(symbol);
304 
305 		log.debug("SINGLE symbolURI");
306 		// log.debug("SINGLE symbolURI={}", symbolURI);
307 
308 		final Element root = xmlDocumentDecode(symbolURI);
309 
310 		final Element tag = xmlFirstChild(root, XmlTagExtras.TAG, XML_STOP);
311 
312 		final InstrumentDOM instrument = new InstrumentDOM();
313 
314 		instrument.decodeXML(tag);
315 
316 		return instrument;
317 
318 	}

320 	static String concatenate(final List<String> symbolList) {
321 
322 		final StringBuilder text = new StringBuilder(1024);
323 
324 		int count = 0;
325 
326 		for (final String symbol : symbolList) {
327 			text.append(symbol);
328 			count++;
329 			if (count != symbolList.size()) {
330 				text.append(",");
331 			}
332 		}
333 
334 		return text.toString();
335 
336 	}

338 	static List<DDF_Instrument> remoteLookup(final List<String> symbolList)
339 			throws Exception {
340 
341 		final List<DDF_Instrument> list =
342 				new ArrayList<DDF_Instrument>(symbolList.size());
343 
344 		final String symbolString = concatenate(symbolList);
345 
346 		final String symbolURI = urlInstrumentLookup(symbolString);
347 
348 		log.debug("BATCH symbolURI");
349 		// log.debug("BATCH symbolURI : {}", symbolURI);
350 
351 		final URL url = new URL(symbolURI);
352 
353 		final InputStream input = url.openStream();
354 
355 		final BufferedInputStream stream = new BufferedInputStream(input);
356 
357 		try {
358 
359 			final SAXParserFactory factory = SAXParserFactory.newInstance();
360 
361 			final SAXParser parser = factory.newSAXParser();
362 
363 			final DefaultHandler handler = new DefaultHandler() {
364 
365 				int count;
366 
367 				@Override
368 				public void startElement(final String uri,
369 						final String localName, final String qName,
370 						final Attributes attributes) throws SAXException {
371 
372 					if (XmlTagExtras.TAG.equals(qName)) {
373 
374 						try {
375 
376 							final InstrumentSAX instruement =
377 									new InstrumentSAX();
378 
379 							instruement.decodeSAX(attributes);
380 
381 							list.add(instruement);
382 
383 						} catch (final SymbolNotFoundException e) {
384 
385 							log.warn("symbol not found : {}", e.getMessage());
386 
387 						} catch (final Exception e) {
388 
389 							log.error("decode failure", e);
390 							HelperXML.log(attributes);
391 
392 						}
393 
394 						count++;
395 
396 						if (count % 1000 == 0) {
397 							log.debug("fetch count : {}", count);
398 						}
399 
400 					}
401 
402 				}
403 			};
404 
405 			parser.parse(stream, handler);
406 
407 		} catch (final SAXParseException e) {
408 
409 			log.warn("parse failed : {} ", symbolURI);
410 
411 		} finally {
412 
413 			input.close();
414 
415 		}
416 
417 		return list;
418 
419 	}
