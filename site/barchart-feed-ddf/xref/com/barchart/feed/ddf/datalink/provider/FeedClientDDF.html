151 	private void startup(final String username, final String password,
152 			final Executor exec, final DDF_SocksProxy proxy) {
153 
154 		this.username = username;
155 		this.password = password;
156 		this.executor = exec;
157 		
158 		this.proxySettings = proxy;
159 
160 		timer = new HashedWheelTimer();
161 		
162 		channelFactory = new NioClientSocketChannelFactory(
163 				executor, 1, new NioWorkerPool(executor, DEFAULT_IO_THREADS), timer);
164 
165 		boot = new ClientBootstrap(channelFactory);
166 
167 		if (proxySettings == null) {
168 
169 			/*
170 			 * The vector for data leaving the netty channel and entering the
171 			 * business application logic.
172 			 */
173 			final SimpleChannelHandler ddfHandler = new ChannelHandlerDDF(
174 					eventQueue, messageQueue);
175 
176 			final ChannelPipelineFactory pipelineFactory = new PipelineFactoryDDF(
177 					ddfHandler);
178 
179 			boot.setPipelineFactory(pipelineFactory);
180 
181 		} else {
182 
183 			final ChannelPipelineFactory socksPipelineFactory = new PipelineFactorySocks(
184 					executor, this, proxy);
185 
186 			boot.setPipelineFactory(socksPipelineFactory);
187 			boot.setOption("child.tcpNoDelay", true);
188 			boot.setOption("child.keepAlive", true);
189 			boot.setOption("child.reuseAddress", true);
190 			boot.setOption("readWriteFair", true);
191 
192 		}
193 
194 		boot.setOption(TIMEOUT_OPTION, TIMEOUT);
195 
196 		/* Initialize event policy with event policies that do nothing. */
197 		for (final DDF_FeedEvent event : DDF_FeedEvent.values()) {
198 			eventPolicy.put(event, new EventPolicy() {
199 
200 				@Override
201 				public void newEvent(DDF_FeedEvent event) {
202 					/* Do nothing */
203 				}
204 			});
205 		}
206 
207 		/* Add DefaultReloginPolicy to selected events */
208 		eventPolicy.put(DDF_FeedEvent.LOGIN_FAILURE, reconnectionPolicy);
209 
210 		eventPolicy.put(DDF_FeedEvent.LINK_DISCONNECT, reconnectionPolicy);
211 
212 		eventPolicy.put(DDF_FeedEvent.SETTINGS_RETRIEVAL_FAILURE,
213 				reconnectionPolicy);
214 
215 		eventPolicy.put(DDF_FeedEvent.CHANNEL_CONNECT_FAILURE,
216 				reconnectionPolicy);
217 		eventPolicy.put(DDF_FeedEvent.CHANNEL_CONNECT_TIMEOUT,
218 				reconnectionPolicy);
219 		eventPolicy.put(DDF_FeedEvent.LINK_CONNECT_PROXY_TIMEOUT,
220 				reconnectionPolicy);
221 
222 		/* Add SubscribeAfterLogin to LOGIN_SUCCESS */
223 		eventPolicy.put(DDF_FeedEvent.LOGIN_SUCCESS, new SubscribeAfterLogin());
224 
225 		/* Add HeartbeatPolicy to HEART_BEAT */
226 		eventPolicy.put(DDF_FeedEvent.HEART_BEAT, new HeartbeatPolicy());
227 
228 	}

232 	private boolean loginProxy(String username, String password,
233 			DDF_Server feedServers) {
234 
235 		terminate();
236 
237 		initialize();
238 
239 		// do socks connection
240 
241 		log.info("connect to proxy - address {} port {}",
242 				proxySettings.getProxyAddress(), proxySettings.getProxyPort());
243 
244 		final InetSocketAddress address = new InetSocketAddress(
245 				proxySettings.getProxyAddress(), proxySettings.getProxyPort());
246 
247 		final ChannelFuture futureConnect = boot.connect(address);
248 
249 		channel = futureConnect.getChannel();
250 
251 		final boolean isGoodConnect = futureConnect
252 				.awaitUninterruptibly(TIMEOUT);
253 
254 		if (!isGoodConnect) {
255 			log.error("proxy connect error {}", futureConnect.getCause());
256 			log.error("proxy; {}:{} ", proxySettings.getProxyAddress(),
257 					proxySettings.getProxyPort());
258 
259 			postEvent(DDF_FeedEvent.LINK_CONNECT_PROXY_TIMEOUT);
260 
261 			channel.close();
262 			return false;
263 		}
264 
265 		log.info("server = {}", feedServers.getPrimary());
266 
267 		// set the ddf servers
268 
269 		proxySettings.setFeedServer(feedServers);
270 
271 		// block until we get proxy_connect_command result
272 
273 		Boolean proxyResult = false;
274 
275 		try {
276 			proxyResult = socksConnectResult.take();
277 		} catch (InterruptedException e) {
278 		}
279 
280 		if (proxyResult == false) {
281 
282 			log.error("Socks connect error");
283 			// postEvent(DDF_FeedEvent.LINK_CONNECT_PROXY_TIMEOUT);
284 
285 			return false;
286 		}
287 
288 		/* Send login command to JERQ */
289 		DDF_FeedEvent writeEvent = blockingWrite(FeedDDF.tcpLogin(username,
290 				password));
291 
292 		if (writeEvent == DDF_FeedEvent.COMMAND_WRITE_FAILURE) {
293 			log.error("error sending login command to jerq");
294 			return false;
295 		}
296 
297 		/* Send VERSION # command to JERQ */
298 		writeEvent = blockingWrite(FeedDDF.tcpVersion(VERSION));
299 
300 		if (writeEvent == DDF_FeedEvent.COMMAND_WRITE_FAILURE) {
301 			log.error("error sending VERSION 3 command to jerq");
302 			return false;
303 		}
304 
305 		/* Send timestamp command to JERQ */
306 		writeEvent = blockingWrite(FeedDDF.tcpGo(FeedDDF.SYMBOL_TIMESTAMP));
307 
308 		if (writeEvent == DDF_FeedEvent.COMMAND_WRITE_FAILURE) {
309 			log.error("error sending login GO TIMESTAMP to jerq");
310 			return false;
311 		}
312 
313 		// all is good
314 
315 		return true;
316 
317 	}

367 	private final RunnerDDF eventTask = new RunnerDDF() {
368 
369 		@Override
370 		protected void runCore() {
371 
372 			final int threadNumber = eventTaskNumber.getAndIncrement();
373 
374 			Thread.currentThread().setName("# DDF EVENT TASK " + threadNumber);
375 
376 			log.warn("# started DDF-EventTask {}", threadNumber);
377 
378 			startupLatch.countDown();
379 
380 			while (!Thread.currentThread().isInterrupted()) {
381 
382 				try {
383 
384 					final DDF_FeedEvent event = eventQueue.take();
385 
386 					if (DDF_FeedEvent.isConnectionError(event)) {
387 
388 						log.info("Setting feed state to logged out");
389 						updateFeedStateListeners(Connection.State.DISCONNECTED);
390 
391 					} else if (event == DDF_FeedEvent.LOGIN_SUCCESS) {
392 
393 						log.info("Login success, feed state updated");
394 						updateFeedStateListeners(Connection.State.CONNECTED);
395 
396 					} else if (event == DDF_FeedEvent.LOGOUT) {
397 
398 						log.info("Setting feed state to logged out");
399 						updateFeedStateListeners(Connection.State.DISCONNECTED);
400 
401 					}
402 
403 					log.trace("Enacting policy for :{}", event.name());
404 
405 					eventPolicy.get(event).newEvent(event);
406 
407 				} catch (final InterruptedException e) {
408 
409 					log.warn("# DDF-EventTask InterruptedException {}",
410 							threadNumber);
411 
412 					log.info("Setting feed state to logged out");
413 					updateFeedStateListeners(Connection.State.DISCONNECTED);
414 
415 					Thread.currentThread().interrupt();
416 
417 				} catch (final Throwable e) {
418 					log.error("event delivery failed", e);
419 				}
420 			}
421 
422 			log.error("# DDF-EventTask death {}", threadNumber);
423 		}
424 	};

484 	/*
485 	 * the calls to initialize() and terminate were causing the threading issue,
486 	 * as the runnables were never getting shutdown...
487 	 */
488 	private void initialize() {
489 
490 		startupLatch = new CountDownLatch(3);
491 
492 		log.warn("# initialize start");
493 
494 		final StackTraceElement[] trace = Thread.currentThread()
495 				.getStackTrace();
496 
497 		for (final StackTraceElement e : trace) {
498 			log.debug(e.getClassName() + ":" + e.getLineNumber());
499 		}
500 
501 		try {
502 			executor.execute(heartbeatTask);
503 		} catch (Exception e) {
504 			log.error("error starting DDF_Heartbeat Thread: {} ", e);
505 
506 			try {
507 				executor.execute(new Thread(new Disconnector(
508 						"DDF_Heartbeat Thread Start Exception")));
509 			} catch (Exception e1) {
510 			}
511 
512 			return;
513 		}
514 
515 		try {
516 			executor.execute(eventTask);
517 		} catch (Exception e) {
518 			log.error("error starting DDF_Event Thread: {} ", e);
519 
520 			try {
521 				executor.execute(new Thread(new Disconnector(
522 						"DDF_Event Thread Start Exception")));
523 			} catch (Exception e1) {
524 			}
525 
526 			return;
527 		}
528 
529 		try {
530 			executor.execute(messageTask);
531 		} catch (Exception e) {
532 			log.error("error starting DDF_Message Thread: {} ", e);
533 
534 			try {
535 				executor.execute(new Thread(new Disconnector(
536 						"DDF_Message Thread Start Exception")));
537 			} catch (Exception e1) {
538 			}
539 
540 			return;
541 		}
542 
543 		try {
544 			startupLatch.await();
545 		} catch (InterruptedException e1) {
546 			log.warn("# initialize interrupted, returning");
547 			return;
548 		}
549 
550 		log.warn("# initialize complete");
551 
552 	}

554 	private void terminate() {
555 
556 		log.warn("## terminate start");
557 		
558 		eventQueue.clear();
559 		messageQueue.clear();
560 
561 		/* Interrupts login thread if login is active */
562 
563 		loginHandler.disableLogins();
564 		loginHandler.interruptLogin();
565 
566 		// kill all threads
567 		// blocking order with join()
568 
569 		if (heartbeatTask != null) {
570 			heartbeatTask.interrupt();
571 
572 			log.warn("# terminate: DDF-heartbeat killed");
573 
574 		}
575 
576 		if (messageTask != null) {
577 			messageTask.interrupt();
578 
579 			log.warn("# terminate: DDF-MesssageTask killed");
580 
581 		}
582 
583 		if (eventTask != null) {
584 			eventTask.interrupt();
585 
586 			log.warn("# terminate DDF-EventTask killed");
587 
588 		}
589 
590 		log.warn("# terminate: closing channel.");
591 
592 		if (channel != null) {
593 
594 			final ChannelFuture cf = channel.close();
595 
596 			try {
597 				cf.await();
598 			} catch (final InterruptedException e) {
599 				log.warn("# terminate: channel.close() channel interrupted");
600 			} finally {
601 				log.warn("# terminate: channel.close() complete");
602 				channel = null;
603 			}
604 
605 		}
606 		
607 		log.warn("## terminate complete");
608 
609 	}

641 		loginHandler.login(0);

664 	private DDF_FeedEvent blockingWrite(final CharSequence message) {
665 		final ChannelFuture futureWrite = channel.write(message);
666 
667 		futureWrite.awaitUninterruptibly(TIMEOUT, TIME_UNIT);
668 
669 		if (futureWrite.isSuccess()) {
670 			return DDF_FeedEvent.COMMAND_WRITE_SUCCESS;
671 		} else {
672 			return DDF_FeedEvent.COMMAND_WRITE_FAILURE;
673 		}
