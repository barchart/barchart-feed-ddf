69  	public static final Set<SubscriptionType> fromEvents(
70  			final Set<MarketEvent> eventSet) {
71  		
72  		final Set<SubscriptionType> result = 
73  				EnumSet.noneOf(SubscriptionType.class);
74  		
75  		if(eventSet == null || eventSet.isEmpty()) {
76  			return result;
77  		}
78  		
79  		for(final MarketEvent event : eventSet) {
80  			switch(event) {
81  			
82  			case MARKET_UPDATED:
83  				result.addAll(DDF_FeedInterest.setValues());
84  				break;
85  			case NEW_TRADE:
86  				result.add(QUOTE_UPDATE);
87  				break;
88  				
89  			case NEW_BAR_CURRENT:
90  			case NEW_BAR_PREVIOUS:
91  			case NEW_OPEN:
92  			case NEW_HIGH:
93  			case NEW_LOW:
94  			case NEW_CLOSE:
95  			case NEW_SETTLE:
96  			case NEW_VOLUME:
97  			case NEW_INTEREST:
98  				result.add(QUOTE_SNAPSHOT);
99  				result.add(QUOTE_UPDATE);
100 				break;
101 				
102 			case NEW_BOOK_ERROR:
103 				// debug use only
104 				break;
105 
106 			case NEW_BOOK_SNAPSHOT:
107 				result.add(BOOK_SNAPSHOT);
108 				break;
109 
110 			case NEW_BOOK_UPDATE:
111 			case NEW_BOOK_TOP:
112 				result.add(BOOK_UPDATE);
113 				result.add(BOOK_SNAPSHOT);
114 				break;
115 
116 			//NEW_CUVOL_UPDATE not supported by ddf
117 				
118 			case NEW_CUVOL_SNAPSHOT:
119 				result.add(CUVOL_SNAPSHOT);
120 				break;
121 
122 			default:
123 				result.add(QUOTE_UPDATE);
124 				break;
125 			}
126 		}
127 		
128 		return result;
129 	}
