377 		final DDF_MessageType type = getMessageType();
378 		//

426 	/*
427 	 * <soh><rec><symbol>,<subrec><stx><base><exch>(<delay>)(,)(<spread>)||
428 	 */
429 	protected final void decodeHead(final ByteBuffer buffer) {
430 		check(buffer.get(), SOH); // <soh>
431 		final byte record = buffer.get(); // <rec>
432 		final byte[] symbolMain = read(buffer, COMMA); // <symbol>,
433 		final byte subRecord = buffer.get(); // <subrec>
434 		check(buffer.get(), STX); // <stx>
435 		final byte baseCode = buffer.get(); // <base code>
436 		final byte exchCode = buffer.get(); // <exch code>
437 		decodeDelay(buffer); // optional (<delay>)(,)
438 		final byte[][] symbolLegs = decodeSpread(buffer); // optional (<spread>)
439 		//
440 		setSymbol(symbolMain, symbolLegs);
441 		setMessageType(DDF_MessageType.fromPair(record, subRecord));
442 		ordFraction = DDF_Fraction.fromBaseCode(baseCode).ord;
443 		ordExchange = DDF_Exchange.fromCode(exchCode).ord;
444 	}

446 	private final void setSymbol(final byte[] symbolMain,
447 			final byte[][] symbolLegs) {
448 		if (symbolMain == null) {
449 			symbolArray = null;
450 			return;
451 		}
452 		if (symbolLegs == null) {
453 			symbolArray = new byte[1][];
454 			symbolArray[0] = symbolMain;
455 			return;
456 		}
457 		final int legCount = symbolLegs.length;
458 		symbolArray = new byte[1 + legCount][];
459 		symbolArray[0] = symbolMain;
460 		System.arraycopy(symbolLegs, 0, symbolArray, 1, legCount);
461 	}

473 	protected void decodeTail(final ByteBuffer buffer) {
474 		final DateTimeZone zone = getExchange().kind.time.zone;
475 		//
476 		final byte dayCode = buffer.get(); // <day>
477 		final byte sessCode = buffer.get(); // <session>
478 		check(buffer.get(), ETX); // <etx>
479 		millisUTC = decodeFeedTimeStamp(zone, buffer); // <time stamp>
480 		//
481 		ordTradeDay = DDF_TradeDay.fromCode(dayCode).ord;
482 		ordSession = DDF_Session.fromPair(ordExchange, sessCode).ord;
483 	}

487 	/* <spread type> <num of legs> <symbol2>, <symbol3>, <symbol4>, */
488 
489 	private final void encodeSpread(final ByteBuffer buffer) {
490 		final DDF_SpreadType spread = getSpreadType();
491 		if (!spread.isKnown() || symbolArray == null || symbolArray.length == 1) {
492 			return;
493 		}
494 		final int legCount = symbolArray.length - 1;
495 		buffer.putChar(spread.code); // <spread type>
496 		encodeUnsigned_1(legCount, buffer); // <num of legs>
497 		for (int k = 1; k <= legCount; k++) {
498 			buffer.put(symbolArray[k]); // <symbol N>
499 			buffer.put(ASCII.COMMA); // ,
500 		}
501 	}
502 
503 	private final byte[][] decodeSpread(final ByteBuffer buffer) {
504 		buffer.mark();
505 		final char spreadCode = buffer.getChar(); // <spread type>
506 		final DDF_SpreadType spread = DDF_SpreadType.fromCode(spreadCode);
507 		setSpread(spread);
508 		if (spread.isKnown()) {
509 			final int legCount = decodeUnsigned_1(buffer); // <num of legs>
510 			final byte[][] symbolLegs = new byte[legCount][];
511 			for (int k = 0; k < legCount; k++) {
512 				symbolLegs[k] = read(buffer, COMMA); // <symbol N>,
513 			}
514 			return symbolLegs;
515 		} else {
516 			buffer.reset();
517 			return null;
518 		}
519 	}

521 	protected final void updateSpread() {
522 		if (symbolArray == null) {
523 			return;
524 		}
525 		if (symbolArray.length == 1) {
526 			setSpread(DDF_SpreadType.UNKNOWN);
527 		} else {
528 			setSpread(DDF_SpreadType.DEFAULT);
529 		}
