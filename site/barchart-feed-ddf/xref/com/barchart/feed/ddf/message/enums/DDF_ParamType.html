212 	/**
213 	 * From code.
214 	 *
215 	 * @param code the code
216 	 * @return the dD f_ param type
217 	 */
218 	public final static DDF_ParamType fromCode(final char code) {
219 		for (final DDF_ParamType known : ENUM_VALUES) {
220 			if (known.code == code) {
221 				return known;
222 			}
223 		}
224 		log.debug("UNKNOWN PARAM CODE = " + code);
225 		return UNKNOWN;
226 	}
