140 
141 		final long[] array = sizeArray;
142 		final int length = array.length;
143 
144 		final int entryCount = entryCount();
145 
146 		final int exponent = getFraction().decimalExponent;
147 
148 		final MarketDoCuvolEntry[] entries = new MarketDoCuvolEntry[entryCount];
149 		int entryIndex = 0;
150 
151 		for (int index = 0; index < length; index++) {
152 			final long sizeValue = array[index];
153 			if (sizeValue > 0) {
154 				final int place = entryIndex + 1;
155 				final long mantissa = priceFirst + index * priceStep;
156 				final PriceValue price = ValueBuilder.newPrice(mantissa,
157 						exponent);
158 				final SizeValue size = ValueBuilder.newSize(sizeValue);
159 				final MarketDoCuvolEntry entry = new DefCuvolEntry(place,
160 						price, size);
161 				entries[entryIndex++] = entry;
162 			}
163 		}
164 
165 		assert entryIndex == entryCount;
166 
167 		return entries;

246 	public final void decodeXML(final Element tag) {
247 
248 		xmlCheckTagName(tag, TAG);
249 
250 		symbolArray = xmlDecSymbol(tag, SYMBOL, XML_STOP);
251 
252 		updateSpread();
253 
254 		final byte baseCode = xmlByteDecode(tag, FRACTION_DDF, XML_STOP);
255 		final DDF_Fraction frac = DDF_Fraction.fromBaseCode(baseCode);
256 		setFraction(frac);
257 
258 		priceStep = xmlDecimalDecode(frac, tag, PRICE_TICK_INCREMENT, XML_STOP);
259 
260 		priceTrade = xmlDecimalDecode(frac, tag, PRICE_LAST, XML_STOP);
261 		sizeTrade = xmlLongDecode(tag, SIZE_LAST, XML_STOP);
262 		sizeTradeCuvol = xmlLongDecode(tag, SIZE_LAST_CUVOL, XML_STOP);
263 
264 		//
265 
266 		final int entryCount = xmlIntegerDecode(tag, ENTRY_COUNT, XML_STOP);
267 
268 		if (entryCount > 0) {
269 
270 			final String stringEntries = xmlStringDecode(tag, ENTRY_ARRAY,
271 					XML_STOP);
272 
273 			final String[] stringArray = stringEntries.split(STRING_COLON);
274 
275 			long priceMin = Long.MAX_VALUE;
276 			long priceMax = Long.MIN_VALUE;
277 
278 			final long[] prices = new long[entryCount];
279 			final int[] sizes = new int[entryCount];
280 
281 			int index = 0;
282 
283 			for (final String stringEntry : stringArray) {
284 				assert ValueBuilder.isPureAscii(stringEntry);
285 				final ByteBuffer buffer = ByteBuffer.wrap(stringEntry
286 						.getBytes(ASCII_CHARSET));
287 				final long price = HelperDDF.decimalDecode(frac, buffer, COMMA);
288 				final long size = HelperDDF.longDecode(buffer, NUL);
289 				priceMin = Math.min(priceMin, price);
290 				priceMax = Math.max(priceMax, price);
291 				prices[index] = price;
292 				sizes[index] = MathExtra.castLongToInt(size);
293 				index++;
294 			}
295 
296 			assert index == entryCount;
297 
298 			priceFirst = priceMin;
299 
300 			final long range = priceMax - priceMin + priceStep;
301 			final int length = MathExtra.castLongToInt(range / priceStep);
302 			sizeArray = new long[length];
303 
304 			for (index = 0; index < entryCount; index++) {
305 				final long price = prices[index];
306 				final int size = sizes[index];
307 				final int offset = (int) ((price - priceFirst) / priceStep);
308 				sizeArray[offset] = size;
309 			}
310 
311 			assert index == entryCount;
312 
313 		}
314 
315 		// hack to work around for lack of "exchange_ddf" attribute
316 
317 		final DDF_Instrument instrument = getInstrument();
318 		setExchange(instrument.get(DDF_InstrumentField.DDF_EXCHANGE));
319 
320 		final long millisUTC = xmlTimeDecode(getExchange().kind.time.zone, tag,
321 				TIME_LAST, XML_PASS);
322 		setDecodeDefaults(millisUTC);
323 
324 	}

333 	@Override
334 	public final void encodeXML(final Element tag) {
335 
336 		xmlCheckTagName(tag, TAG);
337 
338 		xmlAsciiEncode(getSymbolFull(), tag, SYMBOL);
339 
340 		final DDF_Fraction frac = getFraction();
341 		xmlByteEncode(frac.baseCode, tag, FRACTION_DDF);
342 
343 		xmlDecimalEncode(priceStep, frac, tag, PRICE_TICK_INCREMENT);
344 
345 		xmlDecimalEncode(priceTrade, frac, tag, PRICE_LAST);
346 		xmlLongEncode(sizeTrade, tag, SIZE_LAST);
347 		xmlLongEncode(sizeTradeCuvol, tag, SIZE_LAST_CUVOL);
348 
349 		//
350 
351 		final int entryCount = entryCount();
352 		xmlIntegerEncode(entryCount, tag, ENTRY_COUNT);
353 
354 		final StringBuilder text = new StringBuilder(512);
355 
356 		final ByteBuffer buffer = ByteBuffer.allocate(32);
357 
358 		if (entryCount > 0) {
359 
360 			final long[] array = sizeArray;
361 			final int length = array.length;
362 
363 			int entryIndex = 0;
364 
365 			for (int index = 0; index < length; index++) {
366 				final long size = array[index];
367 				if (size > 0) {
368 					if (entryIndex > 0) {
369 						text.append(STRING_COLON);
370 					}
371 					final long price = priceFirst + index * priceStep;
372 					buffer.clear();
373 					HelperDDF.decimalEncode(price, frac, buffer, COMMA);
374 					HelperDDF.longEncode(size, buffer, NUL);
375 					final String stringEntry = new String(buffer.array(), 0,
376 							buffer.position(), ASCII_CHARSET);
377 					text.append(stringEntry);
378 					entryIndex++;
379 				}
380 			}
381 
382 			assert entryCount == entryIndex;
383 
384 		}
385 
386 		xmlTextEncode(text, tag, ENTRY_ARRAY);
387 
388 		// see decodeXML() above
389 
390 		xmlTimeEncode(millisUTC, getExchange().kind.time.zone, tag, TIME_LAST);
391 
392 	}
