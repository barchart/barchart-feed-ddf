8   package com.barchart.feed.client.provider;

10  import java.util.ArrayList;
11  import java.util.HashSet;
12  import java.util.List;
13  import java.util.Set;
14  
15  import com.barchart.feed.base.instrument.values.MarketInstrument;
16  import com.barchart.feed.base.market.api.MarketTaker;
17  import com.barchart.feed.base.market.enums.MarketEvent;
18  import com.barchart.feed.base.market.enums.MarketField;
19  import com.barchart.feed.ddf.instrument.provider.DDF_InstrumentProvider;
20  import com.barchart.util.values.api.Value;
21  
22  public class MarketTakerFactory<V extends Value<V>> {
23  
24  	private Set<MarketEvent> events = new HashSet<MarketEvent>();
25  	private Set<MarketInstrument> instruments = new HashSet<MarketInstrument>();
26  	private List<MarketEventCallback<V>> eventCallbacks = new ArrayList<MarketEventCallback<V>>();
27  	private MarketField<V> field = null;
28  	
29  	public MarketTakerFactory<V> addMarketField(final MarketField<V> field) {
30  		this.field = field;
31  		return this;
32  	}
33  	
34  	public MarketTakerFactory<V> addEvent(final MarketEvent event) {
35  		events.add(event);
36  		return this;
37  	}
38  	
39  	public MarketTakerFactory<V> addInstrument(final MarketInstrument inst) {
40  		instruments.add(inst);
41  		return this;
42  	}
43  	
44  	public MarketTakerFactory<V> addSymbol(final String symbol) {
45  		instruments.add(DDF_InstrumentProvider.find(symbol));
46  		return this;
47  	}
48  	
49  	public MarketTakerFactory<V> addInstruments(final List<MarketInstrument> insts) {
50  		instruments.addAll(insts);
51  		return this;
52  	}
53  	
54  	public MarketTakerFactory<V> addSymbols(final List<String> symbols) {
55  		instruments.addAll(DDF_InstrumentProvider.find(symbols));
56  		return this;
57  	}
58  	
59  	public MarketTakerFactory<V> addEventCallback(final MarketEventCallback<V> eventCallback) {
60  		eventCallbacks.add(eventCallback);
61  		return this;
62  	}
63  	
64  	public MarketTaker<V> build() {	
65  		
66  		final MarketTaker<V> taker =  new MarketTaker<V>() {
67  		
68  			final MarketEvent[] eventArray = events.toArray(new MarketEvent[0]);
69  			final MarketInstrument[] instArray = instruments.toArray(new MarketInstrument[0]);
70   			final MarketEventCallback<V>[] callbackArray = eventCallbacks.toArray(new MarketEventCallback[0]);
71  			final MarketField<V> boundField = field;
72   			
73  			@Override
74  			public MarketField<V> bindField() {
75  				return boundField;
76  			}
77  	
78  			@Override
79  			public MarketEvent[] bindEvents() {
80  				return eventArray;
81  			}
82  	
83  			@Override
84  			public MarketInstrument[] bindInstruments() {
85  				return instArray;
86  			}
87  	
88  			@Override
89  			public void onMarketEvent(MarketEvent event,
90  					MarketInstrument instrument, V value) {
91  				for(final MarketEventCallback<V> e : callbackArray) {
92  					e.onMarketEvent(event, instrument, value);
93  				}
94  			}
95  			
96  		};
97  		
98  		events.clear();
99  		instruments.clear();
100 		eventCallbacks.clear();
101 		
102 		return taker;
103 		
104 	}
