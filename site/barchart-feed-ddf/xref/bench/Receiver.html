29  public class Receiver {

33  	/**
34  	 * 
35  	 * @param args
36  	 * @throws Exception
37  	 */
38  	public static void main(final String... args) throws Exception {
39  
40  		if (args == null || args.length < 3) {
41  			throw new RuntimeException("Bad arguments passed to main");
42  		}
43  		
44  		final Marketplace feed = new BarchartMarketplace(args[0], args[1]);
45  
46  		final MarketObserver<Market> observer = new MarketObserver<Market>() {
47  
48  			@Override
49  			public void onNext(Market value) {
50  				
51  				final StringBuilder sb = new StringBuilder(value.instrument().marketGUID())
52  					.append(" ")
53  					.append(" EventTime=")
54  					.append(value.updated());
55  
56  				final Top top = value.book().top();
57  				
58  				Entry entry = top.ask();
59  				
60  				if (!entry.isNull()) {
61  					sb.append(" ASK TOP").append(" price=")
62  							.append(entry.price().asDouble())
63  							.append(" qty=").append(entry.size().asDouble());
64  				}
65  				
66  				entry = top.bid();
67  				
68  				if (!entry.isNull()) {
69  					sb.append(" BID TOP").append(" price=")
70  							.append(entry.price().asDouble())
71  							.append(" qty=").append(entry.size().asDouble());
72  				}
73  
74  				System.out.println(sb.toString());
75  
76  				
77  			}
78  
79  		};
80  		
81  		feed.startup();
82  		
83  		final Agent myAgent = feed.newAgent(Market.class, observer);
84  		
85  		for(int i = 2; i < args.length; i++) {
86  			myAgent.include(args[i]);
87  		}
88  
89  		System.in.read();
90  		feed.shutdown();
91  		System.exit(0);
92  
93  	}
94
