008    package com.barchart.feed.ddf.resovler.provider;
009    
010    import java.util.concurrent.Callable;
011    import java.util.concurrent.ExecutorService;
012    import java.util.concurrent.Executors;
013    import java.util.concurrent.Future;
014    
015    import com.barchart.feed.base.thread.ExecutorCallable;
016    import com.barchart.feed.ddf.resolver.api.DDF_Resolver;
017    import com.barchart.feed.ddf.resolver.api.DDF_Resolver.Mode;
018    import com.barchart.feed.ddf.resolver.provider.DDF_ResolverProvider;
019    
020    // TODO: Auto-generated Javadoc
021    /**
022     * The Class MainResolverIndex.
023     */
024    public class MainResolverIndex {
025    
026            /**
027             * The main method.
028             *
029             * @param args the arguments
030             * @throws Exception the exception
031             */
032            public static void main(final String... args) throws Exception {
033    
034                    final String folder = MainConst.FOLDER;
035                    final int limit = MainConst.LIMIT;
036    
037                    final ExecutorService service = Executors.newFixedThreadPool(3);
038    
039                    final ExecutorCallable caller = new ExecutorCallable() {
040                            @Override
041                            public <V> Future<V> submit(final Callable<V> task) {
042                                    return service.submit(task);
043                            }
044                    };
045    
046                    final DDF_Resolver resolver = DDF_ResolverProvider.newInstance(caller,
047                                    folder, limit);
048    
049                    resolver.open(Mode.REINDEX);
050    
051                    Thread.sleep(10 * 1000);
052    
053                    resolver.close();
054    
055                    Thread.sleep(1000 * 1000);
056    
057            }
058    
059    }
