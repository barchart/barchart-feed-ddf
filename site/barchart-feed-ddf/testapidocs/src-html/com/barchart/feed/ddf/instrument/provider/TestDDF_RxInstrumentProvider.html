041            static Observer<Result<Instrument>> obs() {
042                    return new Observer<Result<Instrument>>() {
043    
044                            @Override
045                            public void onCompleted() {
046                                    log.debug("On Completed Called");
047                            }
048    
049                            @Override
050                            public void onError(Throwable e) {
051                                    log.debug("On Error Called /n{}", e);
052                            }
053    
054                            @Override
055                            public void onNext(Result<Instrument> args) {
056                                    log.debug("On Next Called");
057                                    
058                                    final Map<String, List<Instrument>> insts = args.results();
059                                    
060                                    for(final Entry<String, List<Instrument>> e : insts.entrySet()) {
061                                            
062                                            log.debug("Query : {}", e.getKey());
063                                            
064                                            final List<Instrument> is = e.getValue();
065                                            
066                                            for(final Instrument i : is) {
067                                                    log.debug("Result : \n{}", i);
068                                                    log.debug("Inst Type : {}", i.securityType());
069                                            }
070                                            
071                                    }
072                                    
073                            }
074                            
075                    };
076            }
