026    public class TestUpdate {
027    
028            /**
029             * @param args
030             */
031            public static void main(final String[] args) throws Exception {
032    
033                    final String username = System.getProperty("barchart.username");
034                    final String password = System.getProperty("barchart.password");
035    
036                    final BarchartFeedClient client = new BarchartFeedClient();
037    
038                    final MarketInstrument[] initInsts = new MarketInstrument[3];
039    
040                    initInsts[0] = client.lookup("GOOG");
041                    initInsts[1] = client.lookup("AAPL");
042                    initInsts[2] = client.lookup("FB");
043    
044                    final MarketInstrument[] newInsts = new MarketInstrument[3];
045    
046                    newInsts[0] = client.lookup("AAPL");
047                    newInsts[1] = client.lookup("FB");
048                    newInsts[2] = client.lookup("ORCL");
049    
050                    final Switch taker = new Switch(initInsts, newInsts);
051    
052                    final FeedStateListener feedListener = new FeedStateListener() {
053    
054                            @Override
055                            public void stateUpdate(final FeedState state) {
056    
057                                    if (state == FeedState.LOGGED_IN) {
058                                            client.addTaker(taker);
059                                    }
060                            }
061                    };
062    
063                    client.login(username, password);
064                    client.bindFeedStateListener(feedListener);
065    
066                    Thread.sleep(5000);
067    
068                    System.out
069                                    .println("**********************************************************");
070                    System.out
071                                    .println("**********************************************************");
072                    System.out
073                                    .println("**********************************************************");
074                    System.out
075                                    .println("**********************************************************");
076                    System.out
077                                    .println("**********************************************************");
078    
079                    taker.switchInsts();
080                    client.updateTaker(taker);
081    
082                    Thread.sleep(5000);
083    
084                    client.shutdown();
085    
086            }
087    
088            public static class Switch implements MarketTaker<Market> {
089    
090                    MarketInstrument[] old;
091                    MarketInstrument[] newI;
092                    MarketInstrument[] curr;
093    
094                    Switch(final MarketInstrument[] old, final MarketInstrument[] newI) {
095                            this.old = old;
096                            this.newI = newI;
097                            this.curr = old;
098                    }
099    
100                    public void switchInsts() {
101                            curr = newI;
102                    }
103    
104                    @Override
105                    public MarketEvent[] bindEvents() {
106                            return MarketEvent.in(MarketEvent.values());
107                    }
108    
109                    @Override
110                    public MarketField<Market> bindField() {
111                            return MarketField.MARKET;
112                    }
113    
114                    @Override
115                    public MarketInstrument[] bindInstruments() {
116                            return curr;
117                    }
118    
119                    @Override
120                    public void onMarketEvent(final MarketEvent event,
121                                    final MarketInstrument instrument, final Market value) {
122                            System.out.println(value.get(MarketField.BOOK_TOP).toString());
123    
124                    }
125    
126            }
127    }
