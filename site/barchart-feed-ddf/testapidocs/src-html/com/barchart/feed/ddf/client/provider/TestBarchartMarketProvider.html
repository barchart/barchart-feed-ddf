001    package com.barchart.feed.ddf.client.provider;
002    
003    import java.util.concurrent.CountDownLatch;
004    
005    import org.slf4j.Logger;
006    import org.slf4j.LoggerFactory;
007    
008    import rx.Observer;
009    
010    import com.barchart.feed.api.MarketObserver;
011    import com.barchart.feed.api.consumer.MetadataService.Result;
012    import com.barchart.feed.api.connection.Connection;
013    import com.barchart.feed.api.connection.Connection.Monitor;
014    import com.barchart.feed.api.connection.Connection.State;
015    import com.barchart.feed.api.consumer.ConsumerAgent;
016    import com.barchart.feed.api.consumer.MarketService;
017    import com.barchart.feed.api.model.data.Book;
018    import com.barchart.feed.api.model.data.Market;
019    import com.barchart.feed.api.model.meta.Instrument;
020    import com.barchart.feed.client.provider.BarchartMarketProvider;
021    
022    public class TestBarchartMarketProvider {
023    
024            private static final Logger log = LoggerFactory.getLogger(
025                            TestBarchartMarketProvider.class);
026            
027            private static final String[] insts = {"ESF2014|550C"};
028            
029            public static void main(final String[] args) throws Exception {
030                    
031                    final String username = System.getProperty("barchart.username");
032                    final String password = System.getProperty("barchart.password");
033                    
034                    final MarketService market = new BarchartMarketProvider(username, password);
035                    
036                    final CountDownLatch lock = new CountDownLatch(1);
037                    
038                    market.bindConnectionStateListener(listener(lock));
039                    market.startup();
040                    
041                    lock.await();
042                    
043                    final ConsumerAgent agent1 = market.register(marketObs(), Market.class);
044                    
045                    agent1.include(insts).subscribe(instObs());
046                    Thread.sleep(20 * 1000);
047                    
048                    agent1.exclude(insts).subscribe(instObs());
049                    Thread.sleep(5 * 1000);
050                    
051                    log.debug("Shutting down");
052                    market.shutdown();
053                    
054                    Thread.sleep(5 * 1000);
055                    
056            }
057            
058            private static Monitor listener(final CountDownLatch lock) { 
059                    return new Monitor() {
060    
061                            @Override
062                            public void handle(State state, Connection connection) {
063                                    
064                                    log.debug("New State = {}", state);
065                                    
066                                    if(state == State.CONNECTED) {
067                                            lock.countDown();
068                                    }
069                            }
070                    };
071            }
072            
073            private static MarketObserver<Market> marketObs() {
074                    
075                    return new MarketObserver<Market>() {
076    
077                            @Override
078                            public void onNext(final Market v) {
079                                    final Book book = v.book();
080                                    log.debug("UPDATE: \n{}", book);
081                            }
082                    };
083            }
084            
085            private static Observer<Result<Instrument>> instObs() {
086                    
087                    return new Observer<Result<Instrument>>() {
088    
089                            @Override
090                            public void onCompleted() {
091                                    log.debug("Lookup and registration complete");
092                            }
093    
094                            @Override
095                            public void onError(Throwable e) {
096                                    log.error("Exception in lookup and registration \n{}", e);
097                            }
098    
099                            @Override
100                            public void onNext(Result<Instrument> args) {
101                                    log.debug("New Instrument Lookup and Registration");
102                                    //
103                            }
104                            
105                    };
106            }
107            
108            
109    }
