008    /**
009     * 
010     * The core DDF class which encapsulates all the core functionality a new user
011     * will need to get started.
012     * <p>
013     * Instances are created using the public constructor and require a
014     * valid user name and password. Optional parameters include specifying the
015     * transport protocol and providing an executor.
016     * <p>
017     * The price feed is started and stopped using the startup() and shutdown()
018     * methods. Note that these are non-blocking calls. Applications requiring
019     * actions upon successful login should instantiate and bind a
020     * FeedStatusListener to the client.
021     * <p>
022     * 
023     * 
024     */

048    /**
049     * The entry point for Barchart data feed services.
050     */
051    public class BarchartFeedClient extends BarchartFeedClientBase {
052    
053            private static final Logger log = LoggerFactory
054                            .getLogger(BarchartFeedClient.class);
055    
056            private volatile DDF_FeedClient feed = null;
057    
058            private Executor executor = null;
059    
060            public BarchartFeedClient() {
061    
062                    this(new Executor() {
063    
064                            private final AtomicLong counter = new AtomicLong(0);
065    
066                            final String name = "# DDF Client - " + counter.getAndIncrement();
067    
068                            @Override
069                            public void execute(final Runnable task) {
070                                    log.debug("executing new runnable = " + task.toString());
071                                    new Thread(task, name).start();
072                            }
073    
074                    });
075    
076            }
077    
078            public BarchartFeedClient(final Executor ex) {
079                    maker.add(instrumentSubscriptionListener);
080                    executor = ex;
081            }
082    
083            /**
084             * Starts the data feed asynchronously. Notification of login success is
085             * reported by FeedStateListeners which are bound to this object.
086             * <p>
087             * Constructs a new feed client with the user's user name and password and a
088             * Socks5 proxy. The transport protocol defaults to TCP and a default
089             * executor are used.
090             * 
091             * @param username
092             * @param password
093             */
094            public void login(final String username, final String password,
095                            final DDF_SocksProxy proxySettings) {
096    
097                    loginProxy(username, password, DDF_Transport.TCP, executor, proxySettings);
098                    
099            }
100    
101            /**
102             * Starts the data feed asynchronously. Notification of login success is
103             * reported by FeedStateListeners which are bound to this object.
104             * <p>
105             * Constructs a new feed client with the user's user name and password. The
106             * transport protocol defaults to TCP and a default executor are used.
107             * 
108             * @param username
109             * @param password
110             */
111            public void login(final String username, final String password) {
112                    
113                    loginMain(username, password, DDF_Transport.TCP, executor);
114                    
115            }
116    
117            /**
118             * Starts the data feed asynchronously. Notification of login success is
119             * reported by FeedStateListeners which are bound to this object.
120             * <p>
121             * Constructs a new feed client with the user's user name, password, and
122             * desired transport protocol. A default executor is used.
123             * 
124             * @param username
125             * @param password
126             * @param tp
127             */
128            public void login(final String username, final String password, final DDF_Transport tp) {
129    
130                    loginMain(username, password, tp, executor);
131    
132            }
133    
134            /**
135             * Starts the data feed asynchronously. Notification of login success is
136             * reported by FeedStateListeners which are bound to this object.
137             * <p>
138             * Constructs a new feed client with the user's user name, password, desired
139             * transport protocol, and framework executor.
140             * 
141             * @param username
142             * @param password
143             * @param tp
144             * @param executor
145             */
146            public void login(final String username, final String password,
147                            final DDF_Transport tp, final Executor executor) {
148    
149                    loginMain(username, password, tp, executor);
150    
151            }
152    
153            /*
154             * Handles login. Non-blocking.
155             */
156            private void loginMain(final String username, final String password,
157                            final DDF_Transport tp, final Executor executor) {
158    
159                    maker.clearAll();
160    
161                    feed = DDF_FeedClientFactory.newConnectionClient(tp, username,
162                                    password, executor);
163    
164                    setClient(feed, false);
165    
166            }
167    
168            private void loginProxy(final String username, final String password,
169                            final DDF_Transport tp, final Executor executor,
170                            final DDF_SocksProxy proxySettings) {
171    
172                    maker.clearAll();
173    
174                    feed = DDF_FeedClientFactory.newConnectionClient(tp, username,
175                                    password, executor, proxySettings);
176    
177                    setClient(feed, true);
178            }
179    
180            /*
181             * This is where the instruments are registered and unregistered as needed
182             * by the market maker. Subscribe events are sent when the instrument has
183             * not been bound by a previously registered market taker. Unsubscribe
184             * events are sent only when the instrument is not needed by any previously
185             * registered market takers.
186             */
187            private final MarketRegListener instrumentSubscriptionListener = 
188                            new MarketRegListener() {
189    
190                    @Override
191                    public void onRegistrationChange(final Map<Instrument, Set<MarketEvent>> 
192                                    instMap) {
193    
194                            final Set<Sub> subs = new HashSet<Sub>();
195                            final Set<Sub> unsubs = new HashSet<Sub>();
196    
197                            for(final Entry<Instrument, Set<MarketEvent>> e: instMap.entrySet()) {
198    
199                                    /*
200                                     * The market maker denotes 'unsubscribe' with an empty event set
201                                     */
202                                    if (e.getValue().isEmpty()) {
203                                            log.debug("Unsubscribing to "
204                                                            + e.getKey().symbol());
205                                            unsubs.add(new DDF_Subscription(e.getKey(), e.getValue(), Type.INSTRUMENT));
206                                    } else {
207                                            log.debug("Subscribing to "
208                                                            + e.getKey().symbol() + " Events: "
209                                                            + printEvents(e.getValue()));
210                                            subs.add(new DDF_Subscription(e.getKey(), e.getValue(), Type.INSTRUMENT));
211                                    }
212    
213                            }
214    
215                            if(!unsubs.isEmpty()) {
216                                    feed.unsubscribe(unsubs);
217                            }
218                            feed.subscribe(subs);
219    
220                    }
221            };
222    
223            private String printEvents(final Set<MarketEvent> events) {
224                    final StringBuffer sb = new StringBuffer();
225    
226                    for (final MarketEvent me : events) {
227                            sb.append(me.name() + ", ");
228                    }
229    
230                    return sb.toString();
231            }
232
