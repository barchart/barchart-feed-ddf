001    package com.barchart.feed.ddf.instrument.provider;
002    
003    import java.util.ArrayList;
004    import java.util.Collections;
005    import java.util.List;
006    
007    import javax.xml.parsers.DocumentBuilder;
008    import javax.xml.parsers.DocumentBuilderFactory;
009    
010    import org.joda.time.DateTime;
011    import org.joda.time.format.DateTimeFormat;
012    import org.joda.time.format.DateTimeFormatter;
013    import org.w3c.dom.Element;
014    import org.w3c.dom.NodeList;
015    
016    public class TestVersionParser {
017    
018            private static final String S3_URL = "https://s3.amazonaws.com/instrument-def/";
019            
020            private static final DateTimeFormatter TIMESTAMP_FORMATTER = 
021                            DateTimeFormat.forPattern("yyyy-MM-dd'T'HH:mm:ss");
022            
023            public static void main(final String[] args) throws Exception {
024                    
025                    
026                    System.out.println(getLatestVersion());
027                    
028            }
029            
030            private static String getLatestVersion() throws Exception {
031                    
032                    DocumentBuilder db = DocumentBuilderFactory.newInstance().newDocumentBuilder();
033                    
034                    Element element = db.parse(S3_URL).getDocumentElement();
035                    NodeList nodeList = element.getChildNodes();
036                    
037                    List<DateTime> versionTimes = new ArrayList<DateTime>();
038                    
039                    for(int i = 0; i < nodeList.getLength(); i++) {
040                            
041                            if(nodeList.item(i).getNodeName().equals("Contents")) {
042                                    
043                                    NodeList contentNodes = nodeList.item(i).getChildNodes();
044                                    for(int n = 0; n < contentNodes.getLength(); n++) {
045                                            
046                                            if(contentNodes.item(n).getNodeName().equals("LastModified")) {
047                                                    versionTimes.add(TIMESTAMP_FORMATTER.parseDateTime(
048                                                                    contentNodes.item(n).getTextContent().split("\\.")[0]));
049                                            }
050                                            
051                                    }
052                                    
053                            }
054                    }
055                    
056                    Collections.sort(versionTimes);
057                    
058                    return TIMESTAMP_FORMATTER.print(versionTimes.get(versionTimes.size() - 1));
059                    
060            }
061            
062    }
