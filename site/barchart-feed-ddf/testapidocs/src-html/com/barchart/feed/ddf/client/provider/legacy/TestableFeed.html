077            public TestableFeed(final String username, final String password) {
078                    
079                    executor = Executors.newFixedThreadPool(100, 
080                                    
081                                    new ThreadFactory() {
082    
083                            final AtomicLong counter = new AtomicLong(0);
084                            
085                            @Override
086                            public Thread newThread(final Runnable r) {
087                                    
088                                    final Thread t = new Thread(r, "Feed thread " + 
089                                                    counter.getAndIncrement()); 
090                                    
091                                    t.setDaemon(true);
092                                    
093                                    return t;
094                            }
095                            
096                    });
097                    
098                    dbFolder = getTempFolder();
099                    
100                    connection = DDF_FeedClientFactory.newConnectionClient(
101                                    DDF_Transport.TCP, username, password, executor);
102                    
103                    connection.bindMessageListener(msgListener);
104                    
105                    maker = TestableMarketplace.newTestableInstance(connection);
106                    maker.add(instrumentSubscriptionListener);
107                    
108            }

110            private final MarketRegListener instrumentSubscriptionListener = new MarketRegListener() {
111    
112                    @Override
113                    public void onRegistrationChange(final Map<Instrument, Set<MarketEvent>> 
114                                    instMap) {
115    
116                            final Set<Subscription> subs = new HashSet<Subscription>();
117                            final Set<Subscription> unsubs = new HashSet<Subscription>();
118    
119                            for(final Entry<Instrument, Set<MarketEvent>> e: instMap.entrySet()) {
120    
121                                    /*
122                                     * The market maker denotes 'unsubscribe' with an empty event set
123                                     */
124                                    if (e.getValue().isEmpty()) {
125                                            log.debug("Unsubscribing to "
126                                                            + e.getKey().symbol());
127                                            unsubs.add(new DDF_Subscription(e.getKey(), e.getValue()));
128                                    } else {
129                                            log.debug("Subscribing to "
130                                                            + e.getKey().symbol() + " Events: "
131                                                            + printEvents(e.getValue()));
132                                            subs.add(new DDF_Subscription(e.getKey(), e.getValue()));
133                                    }
134    
135                            }
136    
137                            if(!unsubs.isEmpty()) {
138                                    connection.unsubscribe(unsubs);
139                            }
140                            connection.subscribe(subs);
141    
142                    }
143            };
144            
145            private String printEvents(final Set<MarketEvent> events) {
146                    final StringBuffer sb = new StringBuffer();
147    
148                    for (final MarketEvent me : events) {
149                            sb.append(me.name() + ", ");
150                    }
151    
152                    return sb.toString();
153            }

155            /*
156             * This is the default message listener. Users wishing to handle raw
157             * messages will need to implement their own feed client.
158             */
159            private final DDF_MessageListener msgListener = new DDF_MessageListener() {
160    
161                    @Override
162                    public void handleMessage(final DDF_BaseMessage message) {
163    
164                            if (message instanceof DDF_ControlTimestamp) {
165                                    for (final TimestampListener listener : timeStampListeners) {
166                                            listener.listen(factory.newTime(((DDF_ControlTimestamp) message)
167                                                            .getStampUTC().asMillisUTC(), ""));
168                                    }
169                            }
170    
171                            if (message instanceof DDF_MarketBase) {
172                                    final DDF_MarketBase marketMessage = (DDF_MarketBase) message;
173                                    maker.make(marketMessage);
174                            }
175    
176                    }
177    
178            };

180            private final AtomicBoolean isStartingup = new AtomicBoolean(false);
181            private final AtomicBoolean isShuttingdown = new AtomicBoolean(false);
182            
183            @Override
184            public void startup() {
185                    
186                    // Consider dummy future?
187                    if(isStartingup.get()) {
188                            throw new IllegalStateException("Startup called while already starting up");
189                    }
190                    
191                    if(isShuttingdown.get()) {
192                            throw new IllegalStateException("Startup called while shutting down");
193                    }
194                    
195                    isStartingup.set(true);
196                    
197                    executor.execute(new StartupRunnable());
198                    
199            }
200            
201            private final class StartupRunnable implements Runnable {
202    
203                    /*
204                     * Ensures instrument database is installed/updated before
205                     * user is able to send subscriptions to JERQ
206                     */
207                    @Override
208                    public void run() {
209                            
210                            try {
211                            
212                                    dbMap = InstrumentDBProvider.getMap(dbFolder);
213                                    
214                                    DDF_InstrumentProvider.bindDatabaseMap(dbMap);
215                                    
216                                    final Future<Boolean> dbUpdate = executor.submit(
217                                                    InstrumentDBProvider.updateDBMap(dbFolder, dbMap));
218                                    
219                                    dbUpdate.get(DB_UPDATE_TIMEOUT, TimeUnit.SECONDS);
220                                    
221                                    connection.startup();
222                            
223                            } catch (final Throwable t) {
224                                    
225                                    isStartingup.set(false);
226                                    
227                                    return;
228                            }
229                            
230                            isStartingup.set(false);
231                            
232                    }
233                    
234            }
235    
236            @Override
237            public void shutdown() {
238                    
239                    // Consider dummy future?
240                    if(isStartingup.get()) {
241                            throw new IllegalStateException("Shutdown called while shutting down");
242                    }
243                    
244                    if(isShuttingdown.get()) {
245                            throw new IllegalStateException("Shutdown called while already shutting down");
246                    }
247                    
248                    isShuttingdown.set(true);
249                    
250                    executor.execute(new ShutdownRunnable());
251                    
252            }
253            
254            private final class ShutdownRunnable implements Runnable {
255    
256                    @Override
257                    public void run() {
258                            
259                            try {
260                                    
261                                    if(maker != null) {
262                                            maker.clearAll();
263                                    }
264                                    
265                                    if(dbMap != null) {
266                                            dbMap.close();
267                                    }
268            
269                                    connection.shutdown();
270                                    
271                                    log.debug("Barchart Feed shutdown");
272                                    
273                            } catch (final Throwable t) {
274                                    
275                                    log.error(t.getMessage());
276                                    
277                                    isShuttingdown.set(false);
278                                    
279                                    return;
280                            }
281                            
282                            
283                            isShuttingdown.set(false);
284                            
285                            log.debug("Barchart Feed shutdown succeeded");
286                            
287                    }
288                    
289            }
