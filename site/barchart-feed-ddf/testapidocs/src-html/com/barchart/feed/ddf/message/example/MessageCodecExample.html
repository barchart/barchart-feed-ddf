073            static void decodeMessage(final byte[] array) {
074    
075                    final DDF_BaseMessage message;
076                    try {
077                            message = DDF_MessageService.decode(array);
078                    } catch (final Exception e) {
079                            log.error("decode failed", e);
080                            return;
081                    }
082    
083                    log.info("message : {}", message.toString());
084
