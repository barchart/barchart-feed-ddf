001    package com.barchart.feed.ddf.instrument.provider;
002    
003    import static org.junit.Assert.assertTrue;
004    
005    import java.util.ArrayList;
006    import java.util.List;
007    import java.util.Map;
008    import java.util.concurrent.ConcurrentHashMap;
009    import java.util.concurrent.ConcurrentMap;
010    
011    import org.junit.Test;
012    
013    import com.barchart.feed.api.model.meta.Instrument;
014    import com.barchart.feed.api.util.InstrumentGUID;
015    
016    public class TestRemoteSymbologyContextDDF {
017    
018            public static final String INST_1 = "GOOG";
019            public static final InstrumentGUID GUID_1 = new InstrumentGUID("1261904");
020            
021            public static final String INST_2 = "A6H2012";
022            public static final InstrumentGUID GUID_2 = new InstrumentGUID("1004331");
023            
024            public static final String FAIL_1 = "XXXGGG";
025            
026            @Test
027            public void testRemoteLookup() {
028                    
029                    final ConcurrentMap<InstrumentGUID, Instrument> guidMap = 
030                                    new ConcurrentHashMap<InstrumentGUID, Instrument>();
031                    
032                    final RemoteSymbologyContextDDF ctx = new RemoteSymbologyContextDDF(guidMap);
033                    
034                    final InstrumentGUID guid1 = ctx.lookup(INST_1);
035                    
036                    assertTrue(guid1.equals(GUID_1));
037                    assertTrue(guidMap.containsKey(guid1));
038                    assertTrue(ctx.symbolMap.containsKey(INST_1));
039                    
040                    final InstrumentGUID guid2 = ctx.lookup(INST_2);
041                    
042                    assertTrue(guid2.equals(GUID_2));
043                    assertTrue(guidMap.containsKey(guid2));
044                    assertTrue(ctx.symbolMap.containsKey(INST_2));
045                    
046                    final InstrumentGUID guidFail = ctx.lookup(FAIL_1);
047                    
048                    assertTrue(guidFail.equals(InstrumentGUID.NULL));
049                    assertTrue(ctx.failedMap.containsKey(FAIL_1));
050                    
051            }
052            
053            @Test
054            public void testBatchRemoteLookup() {
055                    
056                    final ConcurrentMap<InstrumentGUID, Instrument> guidMap = 
057                                    new ConcurrentHashMap<InstrumentGUID, Instrument>();
058                    
059                    final RemoteSymbologyContextDDF ctx = new RemoteSymbologyContextDDF(guidMap);
060                    final List<CharSequence> symbols = new ArrayList<CharSequence>();
061                    
062                    symbols.add(INST_1);
063                    symbols.add(INST_2);
064                    symbols.add(FAIL_1);
065                    
066                    final Map<CharSequence, InstrumentGUID> symbolMap = ctx.lookup(symbols);
067                    
068                    assertTrue(symbolMap.containsKey(INST_1));
069                    assertTrue(symbolMap.containsKey(INST_2));
070                    assertTrue(symbolMap.containsKey(FAIL_1));
071                    
072                    assertTrue(symbolMap.get(FAIL_1).equals(InstrumentGUID.NULL));
073                    
074            }
075            
076    }
