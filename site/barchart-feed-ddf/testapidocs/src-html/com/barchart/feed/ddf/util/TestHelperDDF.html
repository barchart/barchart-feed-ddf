201            /**
202             * Test decode binary to decimal.
203             */
204            @Test
205            public void testDecodeBinaryToDecimal() {
206    
207                    long value;
208                    com.barchart.feed.ddf.util.enums.DDF_Fraction frac;
209                    long result;
210    
211                    // 10-1/4
212                    value = 101;
213                    frac = DDF_Fraction.Q4;
214                    result = HelperDDF.fromBinaryToDecimal(value, frac);
215                    // System.out.println(" value=" + value + " exp2=" + frac.nativeExponent
216                    // + " result=" + result + " exp10=" + frac.decimalExponent);
217                    assertEquals(1025, result);
218    
219                    // (-)10-1/4
220                    value = -101;
221                    frac = DDF_Fraction.Q4;
222                    result = HelperDDF.fromBinaryToDecimal(value, frac);
223                    // System.out.println(" value=" + value + " exp2=" + frac.nativeExponent
224                    // + " result=" + result + " exp10=" + frac.decimalExponent);
225                    assertEquals(-1025, result);
226    
227                    // 10-11/32
228                    value = 1011;
229                    frac = DDF_Fraction.Q32;
230                    result = HelperDDF.fromBinaryToDecimal(value, frac);
231                    // System.out.println(" value=" + value + " exp2=" + frac.nativeExponent
232                    // + " result=" + result + " exp10=" + frac.decimalExponent);
233                    assertEquals(1034375, result);
234    
235                    // (-)10-11/32
236                    value = -1011;
237                    frac = DDF_Fraction.Q32;
238                    result = HelperDDF.fromBinaryToDecimal(value, frac);
239                    // System.out.println(" value=" + value + " exp2=" + frac.nativeExponent
240                    // + " result=" + result + " exp10=" + frac.decimalExponent);
241                    assertEquals(-1034375, result);
242    
243                    // 12342-111/256
244                    value = 12342111;
245                    frac = DDF_Fraction.Q256;
246                    result = HelperDDF.fromBinaryToDecimal(value, frac);
247                    // System.out.println(" value=" + value + " exp2=" + frac.nativeExponent
248                    // + " result=" + result + " exp10=" + frac.decimalExponent);
249                    assertEquals(1234243359375L, result);
250    
251            }
