001    package com.barchart.feed.ddf.instrument.provider;
002    
003    import static com.barchart.feed.ddf.util.HelperXML.XML_STOP;
004    import static com.barchart.feed.ddf.util.HelperXML.xmlFirstChild;
005    import static org.junit.Assert.assertTrue;
006    
007    import java.io.ByteArrayInputStream;
008    
009    import javax.xml.parsers.DocumentBuilder;
010    import javax.xml.parsers.DocumentBuilderFactory;
011    
012    import org.junit.Test;
013    import org.w3c.dom.Document;
014    import org.w3c.dom.Element;
015    
016    import com.barchart.feed.api.model.meta.Instrument;
017    import com.barchart.feed.inst.provider.InstrumentFactory;
018    import com.barchart.util.value.ValueFactoryImpl;
019    import com.barchart.util.value.api.ValueFactory;
020    import com.barchart.util.value.api.Size;
021    import com.barchart.util.value.api.TimeInterval;
022    
023    public class TestInstrumentXML {
024    
025            private static final ValueFactory factory = new ValueFactoryImpl();
026            
027            private static final String IBM = "<instruments status=\"200\" count=\"1\">       <instrument lookup=\"IBM\" status=\"200\" guid=\"IBM\" id=\"1298146\" symbol_realtime=\"IBM\" symbol_ddf=\"IBM\" symbol_historical=\"IBM\" "+
028                    "symbol_description=\"International Business Machines Corp.\" symbol_cfi=\"EXXXXX\" exchange=\"XNYS\" exchange_channel=\"NYSE\" exchange_description=\"New York Stock Exchange\" exchange_ddf=\"N\" time_zone_ddf=\"America/New_York\" " + 
029                    "tick_increment=\"1\" unit_code=\"2\" base_code=\"A\" point_value=\"1\"/> </instruments>";
030            
031            @Test
032            public void testXML() throws Exception {
033                    
034                    final DocumentBuilderFactory fac = DocumentBuilderFactory.newInstance();
035                    final DocumentBuilder builder = fac.newDocumentBuilder();
036                    
037                    final Document document = builder.parse(new ByteArrayInputStream(IBM.getBytes()));
038                    final Element root = document.getDocumentElement();
039                    final Element tag = xmlFirstChild(root, XmlTagExtras.TAG, XML_STOP);
040                    final Instrument IBMInst = InstrumentFactory.instrument(InstrumentXML.decodeXML(tag));
041                    
042                    assertTrue(IBMInst.marketGUID().equals("IBM"));
043                    assertTrue(IBMInst.securityType() == Instrument.SecurityType.EQUITY);
044                    assertTrue(IBMInst.liquidityType() == Instrument.BookLiquidityType.NONE);
045                    assertTrue(IBMInst.bookStructure() == Instrument.BookStructureType.NONE);
046                    assertTrue(IBMInst.maxBookDepth() == Size.NULL);
047                    assertTrue(IBMInst.instrumentDataVendor().equals("Barchart"));
048                    assertTrue(IBMInst.symbol().equals("IBM"));
049                    assertTrue(IBMInst.description().equals("International Business Machines Corp."));
050                    assertTrue(IBMInst.CFICode().equals("EXXXXX"));
051                    assertTrue(IBMInst.exchangeCode().equals("N"));
052                    assertTrue(IBMInst.tickSize().equals(factory.newPrice(1, -2)));
053                    assertTrue(IBMInst.pointValue().equals(factory.newPrice(1, 0)));
054                    assertTrue(IBMInst.displayFraction().equals(factory.newFraction(10, -2)));
055                    assertTrue(IBMInst.lifetime() == TimeInterval.NULL);
056                    assertTrue(IBMInst.marketHours().size() == 0);
057                    assertTrue(IBMInst.timeZoneOffset() == -18000000);
058                    assertTrue(IBMInst.timeZoneName().equals("NEW_YORK"));
059                    
060            }
061

064            
065    }
