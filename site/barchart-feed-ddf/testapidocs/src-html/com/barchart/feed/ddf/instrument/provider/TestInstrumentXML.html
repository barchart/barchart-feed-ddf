028    
029    public class TestInstrumentXML {
030    
031            private static final Factory factory = FactoryLoader.load();
032            
033            private static final String IBM = "<instruments status=\"200\" count=\"1\">       <instrument lookup=\"IBM\" status=\"200\" guid=\"IBM\" id=\"1298146\" symbol_realtime=\"IBM\" symbol_ddf=\"IBM\" symbol_historical=\"IBM\" "+
034                    "symbol_description=\"International Business Machines Corp.\" symbol_cfi=\"EXXXXX\" exchange=\"XNYS\" exchange_channel=\"NYSE\" exchange_description=\"New York Stock Exchange\" exchange_ddf=\"N\" time_zone_ddf=\"America/New_York\" " + 
035                    "tick_increment=\"1\" unit_code=\"2\" base_code=\"A\" point_value=\"1\"/> </instruments>";
036            
037            @Test
038            public void testXML() throws Exception {
039                    
040                    ObjectMapFactory.install(new BarchartFeedInstManifest());
041                    final Manifest<ObjectMap> ddfManifest = new Manifest<ObjectMap>();
042                    ddfManifest.put(InstrumentDDF.class, new Tag<?>[0]);
043                    ObjectMapFactory.install(ddfManifest);
044                    
045                    final DocumentBuilderFactory fac = DocumentBuilderFactory.newInstance();
046                    final DocumentBuilder builder = fac.newDocumentBuilder();
047                    
048                    final Document document = builder.parse(new ByteArrayInputStream(IBM.getBytes()));
049                    final Element root = document.getDocumentElement();
050                    final Element tag = xmlFirstChild(root, XmlTagExtras.TAG, XML_STOP);
051                    final InstrumentDDF IBMInst = InstrumentXML.decodeXML(tag);
052                    
053                    System.out.println(IBMInst.toString());
054                    
055                    assertTrue(IBMInst.get(MARKET_GUID).equals("1298146"));
056                    assertTrue(IBMInst.get(SECURITY_TYPE) == Instrument.SecurityType.NULL_TYPE);
057                    assertTrue(IBMInst.get(BOOK_LIQUIDITY) == Instrument.BookLiquidityType.NONE);
058                    assertTrue(IBMInst.get(BOOK_STRUCTURE) == Instrument.BookStructureType.NONE);
059                    assertTrue(IBMInst.get(BOOK_DEPTH) == ValueConst.NULL_SIZE);
060                    assertTrue(IBMInst.get(VENDOR).equals(newText("Barchart")));
061                    assertTrue(IBMInst.get(SYMBOL).equals(newText("IBM")));
062                    assertTrue(IBMInst.get(DESCRIPTION).equals("International Business Machines Corp."));
063                    assertTrue(IBMInst.get(CFI_CODE).equals(newText("EXXXXX")));
064                    assertTrue(IBMInst.get(EXCHANGE_CODE).equals(newText("N")));
065                    assertTrue(IBMInst.get(TICK_SIZE).equals(newPrice(0.01)));
066                    assertTrue(IBMInst.get(POINT_VALUE).equals(newPrice(1)));
067                    assertTrue(IBMInst.get(DISPLAY_FRACTION).equals(factory.newFraction(10, -2)));
068                    assertTrue(IBMInst.get(LIFETIME) == TimeInterval.NULL);
069                    assertTrue(IBMInst.get(MARKET_HOURS).size() == 0);
070                    assertTrue(IBMInst.get(TIME_ZONE_OFFSET).equals(newSize(-18000000)));
071                    assertTrue(IBMInst.get(TIME_ZONE_NAME).equals(newText("NEW_YORK")));
072                    
073            }
