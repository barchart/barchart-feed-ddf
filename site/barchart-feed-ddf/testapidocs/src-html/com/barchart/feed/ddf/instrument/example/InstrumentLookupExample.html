022            private static final Logger log = LoggerFactory
023                            .getLogger(InstrumentLookupExample.class);
024    
025            static final void find(final String symbol) {
026    
027                    final Instrument instrument = DDF_InstrumentProvider.find(symbol).get(0);
028    
029                    if (instrument.isNull()) {
030                            log.error("instrument lookup failed for : {}", symbol);
031                            return;
032                    }
033    
034                    log.info("instrument : {} \n", instrument);
035    
036            }
