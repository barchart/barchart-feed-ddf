036            /**
037             * The main method.

039             * @param args

041             * @throws Exception

043             */
044            public static void main(final String[] args) throws Exception {
045    
046                    final String username = System.getProperty("barchart.username");
047                    final String password = System.getProperty("barchart.password");
048    
049                    final Executor runner = new Executor() {
050    
051                            private final AtomicLong counter = new AtomicLong(0);
052    
053                            final String name = "# DDF Feed Client - "
054                                            + counter.getAndIncrement();
055    
056                            @Override
057                            public void execute(final Runnable task) {
058                                    new Thread(task, name).start();
059                            }
060    
061                    };
062    
063                    final DDF_FeedClient client =
064                                    DDF_FeedClientFactory.newConnectionClient(DDF_Transport.TCP, username,
065                                                    password, runner);
066    
067                    final DDF_MessageListener handler = new LoggingHandler();
068    
069                    client.bindMessageListener(handler);
070    
071                    final Connection.Monitor stateListener = new ExampleFeedStateListener();
072    
073                    client.bindStateListener(stateListener);
074    
075                    client.startup();
076    
077                    // TODO Add feed handler
078                    // if (isLogin == DDF_FeedEvent.LOGIN_SUCCESS) {
079                    // log.error("invalid login");
080                    // return;
081                    // }
082    
083                    Thread.sleep(5000);
084    
085                    final CharSequence request = "" + //
086                                    "$OEX=BbSs" +
087                                    "";
088    
089                    Set<SubscriptionType> insts = new HashSet<SubscriptionType>();
090                    insts.add(SubscriptionType.BOOK_SNAPSHOT);
091                    insts.add(SubscriptionType.BOOK_UPDATE);
092                    insts.add(SubscriptionType.QUOTE_SNAPSHOT);
093                    insts.add(SubscriptionType.QUOTE_UPDATE);
094                    client.subscribe(new DDF_Subscription("$OEX", Type.INSTRUMENT, insts));
095    //               if (!isSent) {
096    //               log.error("invalid session");
097    //               return;
098    //               }
099                    
100                     Thread.sleep(1000 * 1000);
101                    
102                     client.shutdown();
103    
104            }
