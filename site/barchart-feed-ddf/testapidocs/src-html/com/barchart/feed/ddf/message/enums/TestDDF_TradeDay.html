031             * Test trade date from0.
032             */
033            @Test
034            public void testTradeDateFrom0() {
035    
036                    final DDF_TradeDay day = DDF_TradeDay.D15;
037    
038                    final DateTime today = new DateTime("2012-01-18T10:39:44.647-06:00");
039    
040                    final TimeValue value = DDF_TradeDay.tradeDateFrom(day, today);
041    
042                    final DateTime trade = value.asDateTime();
043    
044                    logger.debug("day=" + day);
045                    logger.debug("today=" + today);
046                    logger.debug("trade=" + trade);
047    
048                    assertEquals(trade.getYear(), 2012);
049                    assertEquals(trade.getMonthOfYear(), 01);
050                    assertEquals(trade.getDayOfMonth(), 15);
051    
052            }
