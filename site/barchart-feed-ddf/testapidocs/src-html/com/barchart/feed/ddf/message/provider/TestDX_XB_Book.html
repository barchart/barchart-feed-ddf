205             * Test decode encode xml.
206             */
207            @Test
208            public void testDecodeEncodeXML() {
209    
210                    final DX_XB_Book msg = new DX_XB_Book();
211    
212                    final ByteBuffer source = ByteBuffer.wrap(baXB);
213    
214                    source.get(); // position + 1
215    
216                    msg.decodeXML(source);
217    
218                    final ByteBuffer target = ByteBuffer.allocate(baXB.length);
219    
220                    target.put((byte) '%'); // position + 1
221    
222                    msg.encodeXML(target);
223    
224                    // System.out.println(new String(target.array()));
225    
226                    final byte[] arraySource = source.array();
227                    final byte[] arrayTarget = target.array();
228    
229                    // System.out.println("source=" + new String(arraySource));
230                    // System.out.println("target=" + new String(arrayTarget));
231    
232                    assertTrue(Arrays.equals(arraySource, arrayTarget));
233    
234            }
