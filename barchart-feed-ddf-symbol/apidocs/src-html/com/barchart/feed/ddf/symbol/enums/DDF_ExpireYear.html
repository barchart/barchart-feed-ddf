016    /**
017     * The Enum DDF_ExpireYear.
018     */
019    public enum DDF_ExpireYear {
020    
021            /*
022             * YEAR 1 Byte, Alphanumeric; Contains the last digit of the expiration
023             * year.
024             */
025    
026            Y0('0'), //
027            Y1('1'), //
028            Y2('2'), //
029            Y3('3'), //
030            Y4('4'), //
031            Y5('5'), //
032            Y6('6'), //
033            Y7('7'), //
034            Y8('8'), //
035            Y9('9'), //
036    
037            UNKNOWN('?'), //
038    
039            ;
040    
041            private static final Logger log = LoggerFactory
042                            .getLogger(DDF_ExpireYear.class);
043    
044            /** The code. */
045            public final char code;
046    
047            private DDF_ExpireYear(final char code) {
048                    this.code = code;
049            }
050    
051            /**
052             * From code.
053             *
054             * @param code the code
055             * @return the dD f_ expire year
056             */
057            public static final DDF_ExpireYear fromCode(final char code) {
058                    switch (code) {
059                    case '0':
060                            return Y0;
061                    case '1':
062                            return Y1;
063                    case '2':
064                            return Y2;
065                    case '3':
066                            return Y3;
067                    case '4':
068                            return Y4;
069                    case '5':
070                            return Y5;
071                    case '6':
072                            return Y6;
073                    case '7':
074                            return Y7;
075                    case '8':
076                            return Y8;
077                    case '9':
078                            return Y9;
079                    default:
080                            return UNKNOWN;
081                    }
082            }
083    
084            /**
085             * four digit year value for this enum and millisUTC; for example:
086             * 
087             * Y9.getYear("millisUTC-some-time-in-2009") -> 2009
088             * 
089             * Y9.getYear("millisUTC-some-time-in-2013") -> 2019
090             *
091             * @param millisUTC the millis utc
092             * @return the year this or next
093             */
094            public final int getYearThisOrNext(final long millisUTC) {
095    
096                    // say, 2
097                    final int expiYearDigit = code - '0';
098    
099                    // say, 2013
100                    final int testYear = new DateTime(millisUTC, DateTimeZone.UTC)
101                                    .getYear();
102    
103                    // then 3
104                    final int testYearDigit = testYear % 10;
105    
106                    // then 2010
107                    final int testYearDecade = testYear - testYearDigit;
108    
109                    if (expiYearDigit >= testYearDigit) {
110                            // same decade : 2012
111                            return testYearDecade + expiYearDigit;
112                    } else {
113                            // next decade : 2022
114                            return testYearDecade + 10 + expiYearDigit;
115                    }
116    
117            }
118    
119            /**
120             * Gets the year this or past.
121             *
122             * @param millisUTC the millis utc
123             * @return the year this or past
124             */
125            public final int getYearThisOrPast(final long millisUTC) {
126    
127                    // say 2
128                    final int expiYearDigit = code - '0';
129    
130                    // say, 2013
131                    final int testYear = new DateTime(millisUTC, DateTimeZone.UTC)
132                                    .getYear();
133    
134                    // then 3
135                    final int testYearDigit = testYear % 10;
136    
137                    // then 2010
138                    final int testYearDecade = testYear - testYearDigit;
139    
140                    if (expiYearDigit > testYearDigit) {
141                            // past decade : 2002
142                            return testYearDecade - 10 + expiYearDigit;
143                    } else {
144                            // this decade : 2012
145                            return testYearDecade + expiYearDigit;
146                    }
147    
148            }
149    
150            /**
151             * From date time.
152             *
153             * @param dateTime the date time
154             * @return the dD f_ expire year
155             */
156            public static final DDF_ExpireYear fromDateTime(final DateTime dateTime) {
157                    final int year = dateTime.getYear();
158                    final int yearOfDecade = year % 10;
159                    // NOTE: assumes ascii values
160                    final char code = (char) (yearOfDecade + '0');
161                    return fromCode(code);
162            }
163    
164            // NOTE: TIME_ZONE_CHICAGO
165            /**
166             * From millis utc.
167             *
168             * @param millisUTC the millis utc
169             * @return the dD f_ expire year
170             */
171            public static final DDF_ExpireYear fromMillisUTC(final long millisUTC) {
172                    final DateTime dateTime = new DateTime(millisUTC, DateTimeZone.UTC);
173                    return fromDateTime(dateTime);
174            }
175    
176            /**
177             * Checks if is valid.
178             *
179             * @param code the code
180             * @return true, if is valid
181             */
182            public static final boolean isValid(final char code) {
183                    final DDF_ExpireYear year = fromCode(code);
184                    return year != UNKNOWN;
185            }
186    
187            /**
188             * Offset.
189             *
190             * @return the int
191             */
192            public int offset() {
193                    return code - '0';
194            }

223                    case CALL_9:

225                    case PUT_0:
226                            return Y0;
227                    case PUT_1:
228                            return Y1;
229                    case PUT_2:
230                            return Y2;
231                    case PUT_3:
232                            return Y3;
233                    case PUT_4:
234                            return Y4;
235                    case PUT_5:
236                            return Y5;
237                    case PUT_6:

239                    case PUT_7:

241                    case PUT_8:

243                    case PUT_9:
244                            return Y9;
245                    default:
246                            return UNKNOWN;
247                    }
248            }
