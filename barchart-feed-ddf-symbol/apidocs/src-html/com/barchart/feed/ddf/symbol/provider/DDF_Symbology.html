073            /**
074             * Byte array from symbol array.
075             *
076             * @param symbolArray the symbol array
077             * @return the byte[]
078             */
079            public static final byte[] byteArrayFromSymbolArray(
080                            final byte[][] symbolArray) {
081                    if (symbolArray == null || symbolArray.length == 0) {
082                            return DDF_NO_NAME;
083                    }
084                    final int symbolCount = symbolArray.length;
085                    int fullSize = 0;
086                    for (final byte[] symbol : symbolArray) {
087                            fullSize += symbol.length;
088                    }
089                    fullSize += symbolCount - 1;
090                    final byte[] symbolFull = new byte[fullSize];
091                    int index = 0;
092                    for (int k = 0; k < symbolCount; k++) {
093                            if (k > 0) {
094                                    symbolFull[index++] = UNDER;
095                            }
096                            final byte[] symbol = symbolArray[k];
097                            final int symbolLenght = symbol.length;
098                            System.arraycopy(symbol, 0, symbolFull, index, symbolLenght);
099                            index += symbolLenght;
100                    }
101                    assert index == fullSize;
102                    return symbolFull;
103            }

105            /**
106             * from ddf symbol with legs "<symbol1>_<symbl2>_...<symbolN>" into 2D byte
107             * array
108             *
109             * @param string the string
110             * @return the byte[][]
111             */
112            public static final byte[][] symbolArrayFromSymbolString(final String string) {
113                    if (string == null) {
114                            return null;
115                    }
116                    final String[] stringArray = string.split(STRING_UNDER);
117                    final int symbolCount = stringArray.length;
118                    final byte[][] symbolArray = new byte[symbolCount][];
119                    for (int k = 0; k < symbolCount; k++) {
120                            final String symbol = stringArray[k];
121                            symbolArray[k] = symbol.getBytes(ASCII_CHARSET);
122                    }
123                    return symbolArray;
124            }

235             * Example: CMY or CCMY; WZ8 or CTH8
236             * 
237             * Futures Symbols are 3 or 4 bytes, where C or CC is commodity code from
238             * symbols list, M is contract month code and Y is contract year.
239             */
240            /**
241             * Checks if is future.
242             *
243             * @param symbol the symbol
244             * @return true, if is future
245             */
246            public static final boolean isFuture(final String symbol) {
247                    if (symbol == null) {
248                            return false;
249                    }
250                    if (!containsDigit(symbol)) {
251                            return false;
252                    }
253                    final int size = symbol.length();
254                    if (size < 3) {
255                            return false;
256                    }
257                    if (!DDF_ExpireYear.isValid(symbol.charAt(size - 1))) {
258                            return false;
259                    }
260                    if (!DDF_ExpireMonth.isValid(symbol.charAt(size - 2))) {
261                            return false;
262                    }
263                    return true;
264            }

306             * 
307             * CMSSSSC or CCMSSSP; SX5300P or ESU990C
308             * 
309             * 8 bytes, where C or CC is commodity code from symbols list SSSS or SSS is
310             * strike price C is Put or Call letter C=call this year, D=call next year,
311             * E=call two years out, F=call 3 years out; G=call 4 years out P=put this
312             * year, Q=put next year, R=put 2 years out, S=put 3 years out, T=put 4

316            /**
317             * Checks if is option basic.
318             *
319             * @param symbol the symbol
320             * @return true, if is option basic
321             */
322            public static final boolean isOptionBasic(final String symbol) {
323                    if (!containsDigit(symbol)) {
324                            return false;
325                    }
326                    if (containsOptionSeparator(symbol)) {
327                            return false;
328                    }
329                    final int size = symbol.length();
330                    if (!DDF_OptionYear.isValid(symbol.charAt(size - 1))) {
331                            return false;
332                    }
333                    return true;
334            }

336            /*
337             * Futures Options Symbols Overview - Extended Symbology
338             * =====================================================
339             * 
340             * For exchange traded spreads and select futures options, currently only
341             * NYMEX ClearPort Swaps, exchange code "j" DDF will start utilizing a new
342             * format for options contracts.
343             * 
344             * CCCMY|SSSSSC or CCCMY|SSSSSP
345             * 
346             * Example: JAOV7|10050P
347             * 
348             * Up to 12 bytes, where CCCMY is the futures symbol. CCC - commodity code,
349             * M - month, Y - year | - delimiter SSSSS is the strike price, left
350             * justified, zero padded to the right. C = Call, P = Put
351             */

356            /**
357             * Contains option separator.
358             *
359             * @param symbol the symbol
360             * @return true, if successful
361             */
362            public static final boolean containsOptionSeparator(final String symbol) {
363                    if (symbol == null) {
364                            return false;
365                    }
366                    if (!containsDigit(symbol)) {
367                            return false;
368                    }
369                    return symbol.contains(OPTION_SEPARATOR);
370            }

531            public static final boolean isEquityOther(final String symbol) {
532                    if (symbol == null) {
533                            return false;
534                    }
535                    if (containsDigit(symbol)) {
536                            // excludes futures, options
537                            return false;
538                    }
539                    if (isTest(symbol)) {
540                            return false;
541    
542                    }
543                    if (isForex(symbol)) {
544                            return false;
545    
546                    }
547                    if (isIndex(symbol)) {
548                            return false;
549    
550                    }
551                    if (isEquityLSE(symbol)) {
552                            return false;
553    
554                    }
555                    if (isEquityCAN(symbol)) {
556                            return false;
557    
558                    }
559                    if (isEquityNSE(symbol)) {
560                            return false;
561    
562                    }
563                    return true;
564            }

566            /**
567             * Checks if is future historical.
568             *
569             * @param symbol the symbol
570             * @return true, if is future historical
571             */
572            public static final boolean isFutureHistorical(final String symbol) {
573                    if (symbol == null) {
574                            return false;
575                    }
576                    final int size = symbol.length();
577                    if (size < 3) {
578                            return false;
579                    }
580                    final char digitLow = symbol.charAt(size - 1);
581                    final char digitHigh = symbol.charAt(size - 2);
582                    if (!isDigit(digitLow)) {
583                            return false;
584                    }
585                    if (!isDigit(digitHigh)) {
586                            return false;
587                    }
588                    return true;
589            }
